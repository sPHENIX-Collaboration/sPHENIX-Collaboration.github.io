524 pair<int, int>
525 ExampleAnalysisModule::find_max(RawTowerContainer *towers, int cluster_size)
526 {
527  const int clus_edge_size = (cluster_size - 1) / 2;
528  assert(clus_edge_size >= 0);
529 
530  pair<int, int> max(-1000, -1000);
531  double max_e = 0;
532 
533  for (int col = 0; col < n_size; ++col)
534  for (int row = 0; row < n_size; ++row)
535  {
536  double energy = 0;
537 
538  for (int dcol = col - clus_edge_size; dcol <= col + clus_edge_size;
539  ++dcol)
540  for (int drow = row - clus_edge_size; drow <= row + clus_edge_size;
541  ++drow)
542  {
543  if (dcol < 0 or drow < 0)
544  continue;
545 
546  RawTower *t = towers->getTower(dcol, drow);
547  if (t)
548  energy += t->get_energy();
549  }
550 
551  if (energy > max_e)
552  {
553  max = make_pair(col, row);
554  max_e = energy;
555  }
556  }
557 
558  return max;
559 }
