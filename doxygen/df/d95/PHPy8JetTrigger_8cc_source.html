53  // 12 == nu_e
54  // 13 == muons
55  // 14 == nu_mu
56  // 15 == taus

91

93  {
94  if (_minZ > 0.0)
95  {
96  // Loop over constituents, calculate the z of the leading particle
97 
98  double leading_Z = 0.0;
99 
100  double jet_ptot = sqrt(fastjets[ijet].px() * fastjets[ijet].px() +
101  fastjets[ijet].py() * fastjets[ijet].py() +
102  fastjets[ijet].pz() * fastjets[ijet].pz());
103 
104  for (unsigned int j = 0; j < constituents.size(); j++)
105  {
106  double con_ptot = sqrt(constituents[j].px() * constituents[j].px() +
107  constituents[j].py() * constituents[j].py() +
108  constituents[j].pz() * constituents[j].pz());
109 
110  double ctheta = (constituents[j].px() * fastjets[ijet].px() +
111  constituents[j].py() * fastjets[ijet].py() +
112  constituents[j].pz() * fastjets[ijet].pz()) /
113  (con_ptot * jet_ptot);
114

116 
117  if (z_constit > leading_Z) leading_Z = z_constit;
118  }
119 
120  if (leading_Z > _minZ)
121  {
122  jetFound = true;
123  break;
124  }
125  }

130  }
131  }
