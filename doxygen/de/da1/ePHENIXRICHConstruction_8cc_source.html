269 G4LogicalVolume *
270 ePHENIXRICHConstruction::Construct_HBD(G4LogicalVolume *RICHSecLog)
271 {
272  const double HBD_thickness = geom.get_HBD_thickness();
273  const int n_GEM_layers = geom.get_n_GEM_layers();
274 
275  assert(
276  HBD_thickness < geom.get_dR_frontwindow_shrink() - geom.get_dR_frontwindow());
277  // depth check
278 
279  G4VSolid *RICHHBDBox = new G4Cons("RICHHBDBox", //
280  0, geom.get_RZ_Seg1_HBD() + geom.get_RZ_Seg2_HBD(), // G4double pRmin1, G4double pRmax1,
281  0, geom.get_RZ_Seg1_HBD() + geom.get_RZ_Seg2_HBD(), // G4double pRmin2, G4double pRmax2,
282  HBD_thickness / 2, // G4double pDz,
283  -geom.get_half_angle_HBD() + pi / 2, 2 * geom.get_half_angle_HBD() // G4double pSPhi, G4double pDPhi
284  );
285 
286  G4LogicalVolume *RICHHBDLog = new G4LogicalVolume(RICHHBDBox,
287  G4Material::GetMaterial(geom.get_RICH_gas_mat()), "RICHHBDLog");
288  RegisterLogicalVolume(RICHHBDLog);
289 
290  G4Transform3D transform1 = G4Translate3D(0, geom.get_R_Tip_HBD(),
291  geom.get_Z_Tip_HBD()) *
292  G4RotateX3D(-geom.get_Rotation_HBD()) * G4RotateY3D(pi) * G4TranslateZ3D(HBD_thickness / 2);
293  // G4Transform3D transform1 = G4TranslateZ3D(HBD_thickness / 2);
294 
295  RegisterPhysicalVolume(new G4PVPlacement(transform1, RICHHBDLog, "RICHHBDPhysical", RICHSecLog,
296  false, 0, overlapcheck_rich));
297 
298  // Internal HBD structure
299  // From doi:10.1016/j.nima.2011.04.015
300  // Component Material X0 (cm) Thickness (cm) Area (%) Rad. Length (%)
301  double current_z = -HBD_thickness / 2;
302 
303  // Mesh SS 1.67 0.003 11.5 0.021
304  // exclude for now to let all potons pass to the top GEM foil
305  //G4LogicalVolume* HBDMeshLog = Construct_HBD_Layers(RICHHBDLog, "Mesh", "Steel", current_z,
306  // 0.003 * cm * 11.5e-2);
307  current_z += 0.003 * cm;
308 
309  // photon detector - photocathode is CsI coating on top layer (surface)
310  //
311  G4LogicalSkinSurface *RICHPhotocathodeSurface;
312 
313  // GEM frames FR4 17.1 0.15x4 6.5 0.228 ; 6.5 is the percentage of the area covered by material.
314  // exclude for now to let all potons pass to the top GEM foil
315  //Construct_HBD_Layers(RICHHBDLog, "Frame0", "G10", current_z,
316  // 0.15 * cm * 6.5e-2);
317  current_z += 0.15 * cm;
318 
319  for (int gem = 1; gem <= n_GEM_layers; gem++)
320  {
321  stringstream sid;
322  sid << gem;
323 
324  // GEM Copper 1.43 0.0005x6 64 0.134
325  G4LogicalVolume *GemTopLog = Construct_HBD_Layers(RICHHBDLog,
326  G4String("GEMFrontCu") + G4String(sid.str()), "G4_Cu",
327  current_z, 0.0005 * cm * 64e-2);
328  current_z += 0.0005 * cm;
329 
330  // top GEM has CsI coating for photon detection
331  if (gem == 1)
332  {
333  RICHPhotocathodeSurface = new G4LogicalSkinSurface("RICHPhotocathodeSurface", GemTopLog, geom.get_RICH_Photocathode_OpticalSurface());
334  RICHPhotocathodeSurface->GetName(); // avoid error due to unused object
335  }
336 
337  // GEM Kapton 28.6 0.005x3 64 0.034
338  Construct_HBD_Layers(RICHHBDLog,
339  G4String("GEMKapton") + G4String(sid.str()), "G4_KAPTON",
340  current_z, 0.005 * cm * 64e-2);
341  current_z += 0.005 * cm;
342 
343  // GEM Copper 1.43 0.0005x6 64 0.134
344  Construct_HBD_Layers(RICHHBDLog,
345  G4String("GEMBackCu") + G4String(sid.str()), "G4_Cu", current_z,
346  0.0005 * cm * 64e-2);
347  current_z += 0.0005 * cm;
348 
349  // GEM frames FR4 17.1 0.15x4 6.5 0.228
350  Construct_HBD_Layers(RICHHBDLog,
351  G4String("Frame") + G4String(sid.str()), "G10", current_z,
352  0.15 * cm * 6.5e-2);
353  current_z += 0.15 * cm;
354  }
355 
356  // PCB Kapton 28.6 0.005 100 0.017
357  Construct_HBD_Layers(RICHHBDLog, G4String("PCBKapton"), "G4_KAPTON",
358  current_z, 0.005 * cm * 100e-2);
359  current_z += 0.005 * cm;
360 
361  // PCB Copper 1.43 0.0005 80 0.028
362  Construct_HBD_Layers(RICHHBDLog, G4String("PCBCu"), "G4_Cu", current_z,
363  0.0005 * cm * 80e-2);
364  current_z += 0.0005 * cm;
365 
366  // Facesheet FR4 17.1 0.025x2 100 0.292
367  Construct_HBD_Layers(RICHHBDLog, "Facesheet", "G10", current_z,
368  0.025 * 2 * cm * 100e-2);
369  current_z += 0.025 * 2 * cm;
370 
371  // Panel core Honeycomb 8170 1.905 100 0.023 <- very thin-X0 stuff, ignore
372 
373  // Total vessel 0.82
374  // Readout
375  // Readout board FR4/copper 17.1/1.43 0.05/0.001 100 0.367
376  Construct_HBD_Layers(RICHHBDLog, G4String("ReadoutFR4"), "G10", current_z,
377  0.05 * cm * 100e-2);
378  current_z += 0.05 * cm;
379  Construct_HBD_Layers(RICHHBDLog, G4String("ReadoutCu"), "G4_Cu", current_z,
380  0.001 * cm * 100e-2);
381  current_z += 0.001 * cm;
382 
383  // Preamps + sockets Copper 1.43 0.0005 100 0.66
384  // Total readout 1.03
385  Construct_HBD_Layers(RICHHBDLog, G4String("SocketsCu"), "G4_Cu", current_z,
386  0.0005 * cm * 100e-2);
387  current_z += 0.0005 * cm;
388 
389  assert(current_z < HBD_thickness / 2);
390  //boundary check
391 
392  return RICHHBDLog;
393 }
