487 int QAG4SimulationCalorimeter::process_event_Tower(PHCompositeNode *topNode)
488 {

501  // Grab the towers

503  towernodename.c_str());
504  if (!towers)
505  {
506  std::cout << PHWHERE << ": Could not find node " << towernodename.c_str()

509  }

512  topNode, towergeomnodename.c_str());
513  if (!towergeom)
514  {

516  << endl;
517  return Fun4AllReturnCodes::ABORTRUN;
518  }

520  static const int max_size = 5;
521  map<int, string> size_label;
522  size_label[1] = "1x1";
523  size_label[2] = "2x2";
524  size_label[3] = "3x3";
525  size_label[4] = "4x4";
526  size_label[5] = "5x5";
527  map<int, double> max_energy;
528  map<int, TH1F *> energy_hist_list;
529  map<int, TH1F *> max_energy_hist_list;
530 
531  for (int size = 1; size <= max_size; ++size)
532  {
533  max_energy[size] = 0;
534 
535  TH1F *h = dynamic_cast<TH1F *>(hm->getHisto(
536  get_histo_prefix() + "_Tower_" + size_label[size]));
537  assert(h);
538  energy_hist_list[size] = h;
539  h = dynamic_cast<TH1F *>(hm->getHisto(
540  get_histo_prefix() + "_Tower_" + size_label[size] + "_max"));
541  assert(h);
542  max_energy_hist_list[size] = h;
543  }
544 
545  h_norm->Fill("Tower", towergeom->size()); // total tower count
546  h_norm->Fill("Tower Hit", towers->size());
547 
548  for (int binphi = 0; binphi < towergeom->get_phibins(); ++binphi)
549  {
550  for (int bineta = 0; bineta < towergeom->get_etabins(); ++bineta)
551  {
552  for (int size = 1; size <= max_size; ++size)
553  {
554  // for 2x2 and 4x4 use slide-2 window as implemented in DAQ
555  if ((size == 2 or size == 4) and ((binphi % 2 != 0) and (bineta % 2 != 0)))
556  continue;
557 
558  double energy = 0;
559 
560  // sliding window made from 2x2 sums
561  for (int iphi = binphi; iphi < binphi + size; ++iphi)
562  {
563  for (int ieta = bineta; ieta < bineta + size; ++ieta)
564  {
565  if (ieta > towergeom->get_etabins())
566  continue;
567 
568  // wrap around
569  int wrapphi = iphi;
570  assert(wrapphi >= 0);
571  if (wrapphi >= towergeom->get_phibins())
572  {
573  wrapphi = wrapphi - towergeom->get_phibins();
574  }
575 
576  RawTower *tower = towers->getTower(ieta, wrapphi);
577 
578  if (tower)
579  {
580  const double e_intput = tower->get_energy();
581 
582  energy += e_intput;
583  }
584  }
585  }
586 
587  energy_hist_list[size]->Fill(energy == 0 ? 9.1e-4 : energy); // trick to fill 0 energy tower to the first bin
588 
589  if (energy > max_energy[size])
590  max_energy[size] = energy;
591 
592  } // for (int size = 1; size <= 4; ++size)
593  }
594  }
595 
596  for (int size = 1; size <= max_size; ++size)
597  {
598  max_energy_hist_list[size]->Fill(max_energy[size]);
599  }
600  return Fun4AllReturnCodes::EVENT_OK;
601 }

621 int QAG4SimulationCalorimeter::process_event_Cluster(PHCompositeNode *topNode)
622 {
623  if (Verbosity() > 2)
624  cout << "QAG4SimulationCalorimeter::process_event_Cluster() entered"
625  << endl;
626 
627  Fun4AllHistoManager *hm = QAHistManagerDef::getHistoManager();
628  assert(hm);
629  string towergeomnodename = "TOWERGEOM_" + _calo_name;
630  RawTowerGeomContainer *towergeom = findNode::getClass<RawTowerGeomContainer>(
631  topNode, towergeomnodename.c_str());
632  if (!towergeom)
633  {
634  cout << PHWHERE << ": Could not find node " << towergeomnodename.c_str()
635  << endl;
636  return Fun4AllReturnCodes::ABORTRUN;
637  }
638 
639  //get a cluster count
640  TH1D *h_norm = dynamic_cast<TH1D *>(hm->getHisto(
641  get_histo_prefix() + "_Normalization"));
642  assert(h_norm);
643 
644  std::string nodename = "CLUSTER_" + _calo_name;
645  RawClusterContainer *clusters = findNode::getClass<RawClusterContainer>(
646  topNode, nodename.c_str());
647  assert(clusters);
648  h_norm->Fill("Cluster", clusters->size());
649 
650  // get primary
651  assert(_truth_container);
652  assert(not _truth_container->GetMap().empty());
653  PHG4Particle *last_primary = _truth_container->GetMap().rbegin()->second;
654  assert(last_primary);
655 
656  if (Verbosity() > 2)
657  {
658  cout
659  << "QAG4SimulationCalorimeter::process_event_Cluster() handle this truth particle"
660  << endl;
661  last_primary->identify();
662  }
663 
664  assert(_caloevalstack);
665  CaloRawClusterEval *clustereval = _caloevalstack->get_rawcluster_eval();
666  assert(clustereval);
667 
668  TH1F *h = dynamic_cast<TH1F *>(hm->getHisto(
669  get_histo_prefix() + "_Cluster_BestMatchERatio"));
670  assert(h);
671 
672  RawCluster *cluster = clustereval->best_cluster_from(last_primary);
673  if (cluster)
674  {
675  // has a cluster matched and best cluster selected
676 
677  if (Verbosity() > 3)
678  cout << "QAG4SimulationCalorimeter::process_event_Cluster::"
679  << _calo_name << " - get cluster with energy "
680  << cluster->get_energy() << " VS primary energy "
681  << last_primary->get_e() << endl;
682 
683  h->Fill(cluster->get_energy() / (last_primary->get_e() + 1e-9)); //avoids divide zero
684 
685  // now work on the projection:
686  const CLHEP::Hep3Vector hit(cluster->get_position());
687 
688  const PHG4VtxPoint *primary_vtx = //
689  _truth_container->GetPrimaryVtx(last_primary->get_vtx_id());
690  assert(primary_vtx);
691  if (Verbosity() > 2)
692  {
693  cout
694  << "QAG4SimulationCalorimeter::process_event_Cluster() handle this vertex"
695  << endl;
696  primary_vtx->identify();
697  }
698 
699  const CLHEP::Hep3Vector vertex(primary_vtx->get_x(), primary_vtx->get_y(),
700  primary_vtx->get_z());
701 
702  // projection axis
703  CLHEP::Hep3Vector axis_proj(last_primary->get_px(), last_primary->get_py(),
704  last_primary->get_pz());
705  if (axis_proj.mag() == 0)
706  axis_proj.set(0, 0, 1);
707  axis_proj = axis_proj.unit();
708 
709  // azimuthal direction axis
710  CLHEP::Hep3Vector axis_azimuth = axis_proj.cross(CLHEP::Hep3Vector(0, 0, 1));
711  if (axis_azimuth.mag() == 0)
712  axis_azimuth.set(1, 0, 0);
713  axis_azimuth = axis_azimuth.unit();
714 
715  // polar direction axis
716  CLHEP::Hep3Vector axis_polar = axis_proj.cross(axis_azimuth);
717  assert(axis_polar.mag() > 0);
718  axis_polar = axis_polar.unit();
719 
720  TH2F *hlat = dynamic_cast<TH2F *>(hm->getHisto(
721  get_histo_prefix() + "_Cluster_LateralTruthProjection"));
722  assert(hlat);
723 
724  const double hit_azimuth = axis_azimuth.dot(hit - vertex);
725  const double hit_polar = axis_polar.dot(hit - vertex);
726  hlat->Fill(hit_polar, hit_azimuth);
727  }
728  else
729  {
730  if (Verbosity() > 3)
731  cout << "QAG4SimulationCalorimeter::process_event_Cluster::"
732  << _calo_name << " - missing cluster !";
733  h->Fill(0); // no cluster matched
734  }
735 
736  return Fun4AllReturnCodes::EVENT_OK;
737 }
