517 int PHParameters::ReadFromFile(const string &name, const string &extension, const int layer, const int issuper, const string &dir)
518 {
519  PHTimeStamp TSearch(10);
520  PdbBankID bankID(0);
521  ostringstream fnamestream;
522  fnamestream << name << "_geoparams"
523  << "-" << bankID.getInternalValue();
524  string fileprefix = fnamestream.str();
525  std::transform(fileprefix.begin(), fileprefix.end(), fileprefix.begin(),
526  ::tolower);
527  boost::filesystem::path targetDir(dir);
528 
529  boost::filesystem::recursive_directory_iterator iter(targetDir), eod;
530  boost::char_separator<char> sep("-.");
531  map<unsigned int, string> calibfiles;
532  BOOST_FOREACH (boost::filesystem::path const &i, make_pair(iter, eod))
533  {
534  if (is_regular_file(i))
535  {
536  // boost leaf() gives the filename without path,
537  // this checks if the filename starts with fileprefix
538  // (start pos of substring=0), if not coninue
539  string basename = i.filename().string();
540  if (basename.find(fileprefix) != 0)
541  {
542  continue;
543  }
544  // extension() contains the . - like .xml, so we
545  // just compare the extensions instead of !=
546  // and check that the size makes sense
547  if (i.extension().string().find(extension) == string::npos || i.extension().string().size() != extension.size() + 1)
548  {
549  continue;
550  }
551  boost::tokenizer<boost::char_separator<char> > tok(basename, sep);
552  boost::tokenizer<boost::char_separator<char> >::iterator iter =
553  tok.begin();
554  ++iter; // that skips the file prefix excluding bank id
555  ++iter; // that skips the bank id we checked already as part of the filename
556  PHTimeStamp TStart(ConvertStringToUint(*iter));
557  if (TSearch < TStart)
558  {
559  continue;
560  }
561  ++iter;
562  PHTimeStamp TStop(ConvertStringToUint(*iter));
563  if (TSearch >= TStop)
564  {
565  continue;
566  }
567  ++iter;
568  calibfiles[ConvertStringToUint(*iter)] = i.string();
569  }
570  }
571  if (calibfiles.empty())
572  {
573  cout << "No calibration file like " << dir << "/" << fileprefix << " found" << endl;
574  gSystem->Exit(1);
575  }
576  cout << "PHParameters::ReadFromFile - Reading from File: " << (calibfiles.rbegin())->second << " ... ";
577  string fname = (calibfiles.rbegin())->second;
578  TFile *f = TFile::Open(fname.c_str());
579  if (issuper)
580  {
581  PdbParameterMapContainer *myparm = static_cast<PdbParameterMapContainer *>(f->Get("PdbParameterMapContainer"));
582  assert(myparm);
583 
584  if (myparm->GetParameters(layer) == nullptr)
585  cout << "Missing PdbParameterMapContainer layer #" << layer << endl;
586  assert(myparm->GetParameters(layer));
587 
588  cout << "Received PdbParameterMapContainer layer #" << layer << " with (Hash = 0x" << std::hex << myparm->GetParameters(layer)->get_hash() << std::dec << ")" << endl;
589 
590  FillFrom(myparm, layer);
591  delete myparm;
592  }
593  else
594  {
595  PdbParameterMap *myparm = static_cast<PdbParameterMap *>(f->Get("PdbParameterMap"));
596  assert(myparm);
597  cout << "Received PdbParameterMap with (Hash = 0x" << std::hex << myparm->get_hash() << std::dec << ")" << endl;
598 
599  FillFrom(myparm);
600  delete myparm;
601  }
602  delete f;
603 
604  return 0;
605 }
