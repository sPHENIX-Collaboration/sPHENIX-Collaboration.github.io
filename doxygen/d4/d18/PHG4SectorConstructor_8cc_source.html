74  geom.set_max_polar_angle(geom.get_min_polar_angle());

76  }
77  if ((geom.get_min_polar_edge() == Sector_Geometry::kFlatEdge or geom.get_max_polar_edge() == Sector_Geometry::kFlatEdge) and geom.get_N_Sector() <= 2)

87  geom.get_normal_start() + geom.get_total_thickness() / 2);
88 
89  const G4Transform3D transform_Hall_to_Det(transform_Det_to_Hall.inverse());
90 
91  const double sph_min_polar_angle =
92  (geom.get_min_polar_edge() == Sector_Geometry::kConeEdge) ? geom.get_min_polar_angle() : 0;
93  const double sph_max_polar_angle =

97  geom.get_normal_start(), geom.get_max_R(), // G4double pRmin, G4double pRmax,
98  pi / 2 - pi / geom.get_N_Sector(), 2 * pi / geom.get_N_Sector(), // G4double pSPhi, G4double pDPhi,
99  sph_min_polar_angle, sph_max_polar_angle - sph_min_polar_angle //G4double pSTheta, G4double pDTheta
100  );
101 
102  G4VSolid *SecConeBoundary_Det = new G4DisplacedSolid("SecConeBoundary_Det",
103  SecConeBoundary_Hall, transform_Hall_to_Det);
104 
105  G4VSolid *Boundary_Det = SecConeBoundary_Det;
106 
107  if (geom.get_min_polar_edge() != Sector_Geometry::kConeEdge or geom.get_max_polar_edge() != Sector_Geometry::kConeEdge)
108  {

113  const double sph_max_polar_size =
114  (geom.get_max_polar_edge() == Sector_Geometry::kConeEdge) ? geom.get_max_R() : geom.get_normal_start() * tan(geom.get_max_polar_angle() - geom.get_normal_polar_angle());

116  G4VSolid *BoxBoundary_Det = new G4Box("BoxBoundary_Det",

119  G4VSolid *BoxBoundary_Det_Place = new G4DisplacedSolid(
120  "BoxBoundary_Det_Place", BoxBoundary_Det, 0,
121  G4ThreeVector(0, (sph_max_polar_size - sph_min_polar_size) / 2, 0));

334 double
335 Sector_Geometry::get_max_R() const
336 {
337  // Geometry check
338  assert(abs(min_polar_angle - normal_polar_angle) < pi / 2);
339  assert(abs(max_polar_angle - normal_polar_angle) < pi / 2);
340 
341  if (N_Sector <= 2)
342  {
343  // Geometry check
344  if (cos(min_polar_angle + normal_polar_angle) <= 0)
345  {
346  stringstream s;
347  s << "Geometry check failed. " << endl;
348  s << "normal_polar_angle = " << normal_polar_angle << endl;
349  s << "min_polar_angle = " << min_polar_angle << endl;
350  s << "max_polar_angle = " << max_polar_angle << endl;
351  s << "cos(min_polar_angle + normal_polar_angle) = "
352  << cos(min_polar_angle + normal_polar_angle) << endl;
353 
354  G4Exception("Sector_Geometry::get_max_R", __FILE__, FatalException,
355  s.str().c_str());
356  }
357  if (cos(max_polar_angle + normal_polar_angle) <= 0)
358  {
359  stringstream s;
360  s << "Geometry check failed. " << endl;
361  s << "normal_polar_angle = " << normal_polar_angle << endl;
362  s << "min_polar_angle = " << min_polar_angle << endl;
363  s << "max_polar_angle = " << max_polar_angle << endl;
364  s << "cos(max_polar_angle + normal_polar_angle) = "
365  << cos(max_polar_angle + normal_polar_angle) << endl;
366 
367  G4Exception("Sector_Geometry::get_max_R", __FILE__, FatalException,
368  s.str().c_str());
369  }
370 
371  const double max_tan_angle = std::max(
372  std::abs(tan(min_polar_angle + normal_polar_angle)),
373  std::abs(tan(max_polar_angle + normal_polar_angle)));
374 
375  return (get_normal_start() + get_total_thickness()) * sqrt(1 + max_tan_angle * max_tan_angle);
376  }
377  else
378  {
379  const double max_angle = std::max(
380  std::abs(min_polar_angle - normal_polar_angle),
381  std::abs(max_polar_angle - normal_polar_angle));
382 
383  return (get_normal_start() + get_total_thickness()) * sqrt(1 + pow(tan(max_angle), 2) + pow(tan(2 * pi / N_Sector), 2)) * 2;
384  }
385 }
