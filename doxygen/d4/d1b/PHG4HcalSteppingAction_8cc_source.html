61 bool PHG4HcalSteppingAction::UserSteppingAction(const G4Step* aStep, bool)
62 {
63  // get volume of the current step
64  G4VPhysicalVolume* volume = aStep->GetPreStepPoint()->GetTouchableHandle()->GetVolume();
65 
66  // collect energy and track length step by step
67  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
68  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
69  G4double light_yield = 0;
70 
71  const G4Track* aTrack = aStep->GetTrack();
72 
73  int layer_id = detector_->get_Layer();
74  // make sure we are in a volume
75  // IsInCylinderActive returns the number of the scintillator
76  // slat which has fired
77  int isactive = detector_->IsInCylinderActive(volume);
78  if (isactive > PHG4HcalDetector::INACTIVE)
79  {
80  bool geantino = false;
81  // the check for the pdg code speeds things up, I do not want to make
82  // an expensive string compare for every track when we know
83  // geantino or chargedgeantino has pid=0
84  if (aTrack->GetParticleDefinition()->GetPDGEncoding() == 0 &&
85  aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos)
86  {
87  geantino = true;
88  }
89  G4StepPoint* prePoint = aStep->GetPreStepPoint();
90  G4StepPoint* postPoint = aStep->GetPostStepPoint();
91  // cout << "track id " << aTrack->GetTrackID() << endl;
92  // cout << "time prepoint: " << prePoint->GetGlobalTime() << endl;
93  // cout << "time postpoint: " << postPoint->GetGlobalTime() << endl;
94  switch (prePoint->GetStepStatus())
95  {
96  case fGeomBoundary:
97  case fUndefined:
98 
99  if (!m_Hit)
100  {
101  m_Hit = new PHG4Hitv1();
102  }
103  m_Hit->set_layer((unsigned int) layer_id);
104  m_Hit->set_scint_id(isactive); // isactive contains the scintillator slat id
105  //here we set the entrance values in cm
106  m_Hit->set_x(0, prePoint->GetPosition().x() / cm);
107  m_Hit->set_y(0, prePoint->GetPosition().y() / cm);
108  m_Hit->set_z(0, prePoint->GetPosition().z() / cm);
109 
110  // time in ns
111  m_Hit->set_t(0, prePoint->GetGlobalTime() / nanosecond);
112  //set the track ID
113  m_Hit->set_trkid(aTrack->GetTrackID());
114  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
115  {
116  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
117  {
118  m_Hit->set_trkid(pp->GetUserTrackId());
119  m_Hit->set_shower_id(pp->GetShower()->get_id());
120  m_SaveShower = pp->GetShower();
121  }
122  }
123 
124  //set the initial energy deposit
125  m_Hit->set_edep(0);
126  m_Hit->set_eion(0); // only implemented for v5 otherwise empty
127  m_Hit->set_light_yield(0);
128  // m_Hit->print();
129  // Now add the hit
130  if (isactive >= 0) // the slat ids start with zero
131  {
132  m_SaveHitContainer = m_HitContainer;
133  // unsigned int shift_layer_id = layer_id << (phg4hitdefs::keybits - 3);
134  }
135  else
136  {
137  m_SaveHitContainer = m_AbsorberHits;
138  }
139  // m_SaveHitContainer->AddHit(layer_id, m_Hit);
140 
141  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
142  {
143  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
144  {
145  pp->GetShower()->add_g4hit_id(m_SaveHitContainer->GetID(), m_Hit->get_hit_id());
146  }
147  }
148  if (m_Hit->get_z(0) > zmax || m_Hit->get_z(0) < zmin)
149  {
150  cout << "PHG4HcalSteppingAction: hit outside acceptance, layer: " << layer_id << endl;
151  m_Hit->identify();
152  }
153  break;
154  default:
155  break;
156  }
157  // here we just update the exit values, it will be overwritten
158  // for every step until we leave the volume or the particle
159  // ceases to exist
160  m_Hit->set_x(1, postPoint->GetPosition().x() / cm);
161  m_Hit->set_y(1, postPoint->GetPosition().y() / cm);
162  m_Hit->set_z(1, postPoint->GetPosition().z() / cm);
163 
164  m_Hit->set_t(1, postPoint->GetGlobalTime() / nanosecond);
165 
166  if (isactive >= 0) // the slat ids start with zero
167  {
168  if (light_scint_model_)
169  {
170  light_yield = GetVisibleEnergyDeposition(aStep);
171  }
172  else
173  {
174  light_yield = eion;
175  }
176 
177  if (ValidCorrection())
178  {
179  light_yield = light_yield * GetLightCorrection(postPoint->GetPosition().x() / cm, (postPoint->GetPosition().y() / cm));
180  }
181  }
182 
183  //sum up the energy to get total deposited
184  m_Hit->set_edep(m_Hit->get_edep() + edep);
185  m_Hit->set_eion(m_Hit->get_eion() + eion);
186  m_Hit->set_light_yield(m_Hit->get_light_yield() + light_yield);
187  m_Hit->set_path_length(aTrack->GetTrackLength() / cm);
188 
189  if (m_Hit->get_z(1) > zmax || m_Hit->get_z(1) < zmin)
190  {
191  cout << "PHG4HcalSteppingAction: hit outside acceptance zmin " << zmin << ", zmax " << zmax << " at exit" << endl;
192  m_Hit->identify();
193  }
194  if (geantino)
195  {
196  m_Hit->set_edep(-1); // only energy=0 g4hits get dropped, this way geantinos survive the g4hit compression
197  m_Hit->set_eion(-1);
198  }
199  if (edep > 0)
200  {
201  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
202  {
203  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
204  {
205  pp->SetKeep(1); // we want to keep the track
206  }
207  }
208  }
209  // if any of these conditions is true this is the last step in
210  // this volume and we need to save the hit
211  // postPoint->GetStepStatus() == fGeomBoundary: track leaves this volume
212  // postPoint->GetStepStatus() == fWorldBoundary: track leaves this world
213  // (happens when your detector goes outside world volume)
214  // postPoint->GetStepStatus() == fAtRestDoItProc: track stops (typically
215  // aTrack->GetTrackStatus() == fStopAndKill is also set)
216  // aTrack->GetTrackStatus() == fStopAndKill: track ends
217  if (postPoint->GetStepStatus() == fGeomBoundary ||
218  postPoint->GetStepStatus() == fWorldBoundary ||
219  postPoint->GetStepStatus() == fAtRestDoItProc ||
220  aTrack->GetTrackStatus() == fStopAndKill)
221  {
222  // save only hits with energy deposit (or -1 for geantino)
223  if (m_Hit->get_edep())
224  {
225  m_SaveHitContainer->AddHit(layer_id, m_Hit);
226  if (m_SaveShower)
227  {
228  m_SaveShower->add_g4hit_id(m_SaveHitContainer->GetID(), m_Hit->get_hit_id());
229  }
230  // ownership has been transferred to container, set to null
231  // so we will create a new hit for the next track
232  m_Hit = nullptr;
233  }
234  else
235  {
236  // if this hit has no energy deposit, just reset it for reuse
237  // this means we have to delete it in the dtor. If this was
238  // the last hit we processed the memory is still allocated
239  m_Hit->Reset();
240  }
241  }
242  // return true to indicate the hit was used
243  // hit->identify();
244  // return true to indicate the hit was used
245  return true;
246  }
247  else
248  {
249  return false;
250  }
