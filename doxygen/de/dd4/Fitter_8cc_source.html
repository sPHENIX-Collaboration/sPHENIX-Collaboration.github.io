96 int Fitter::processTrack(PHGenFit::Track* track, const bool save_to_evt_disp)
97 {
98  genfit::Track* fitTrack = track->getGenFitTrack();
99 
100 #if ROOT_VERSION_CODE >= ROOT_VERSION(6, 00, 0)
101  try
102  {
103  fitTrack->checkConsistency();
104  }
105  catch (genfit::Exception& e)
106  {
107  if (verbosity >= 2)
108  {
109  std::cerr << "genfit::Track::checkConsistency() failed!" << std::endl;
110  std::cerr << e.what();
111  }
112  return -1;
113  }
114 #else
115  if (!fitTrack->checkConsistency())
116  {
117  if (verbosity >= 2) LogWARNING("genfit::Track::checkConsistency() failed!");
118  return -1;
119  }
120 #endif
121  try
122  {
123  _fitter->processTrack(fitTrack);
124  }
125  catch (genfit::Exception& e)
126  {
127  if (verbosity >= 1)
128  {
129  std::cerr << "PHGenFit::Fitter::processTrack::Exception: \n";
130  std::cerr << e.what();
131  std::cerr << "Exception, next track" << std::endl;
132  }
133  return -1;
134  }
135 #if ROOT_VERSION_CODE >= ROOT_VERSION(6, 00, 0)
136  try
137  {
138  fitTrack->checkConsistency();
139  }
140  catch (genfit::Exception& e)
141  {
142  if (verbosity >= 2)
143  {
144  std::cerr << "genfit::Track::checkConsistency() failed!" << std::endl;
145  std::cerr << e.what();
146  }
147  return -1;
148  }
149 #else
150 
151  if (!fitTrack->checkConsistency())
152  {
153  if (verbosity >= 2) LogWARNING("genfit::Track::checkConsistency() failed!");
154  return -1;
155  }
156 #endif
157  genfit::AbsTrackRep* rep = fitTrack->getCardinalRep();
158  if (!fitTrack->getFitStatus(rep)->isFitConverged())
159  {
160  if (verbosity >= 2) LogWARNING("Track could not be fitted successfully! Fit is not converged!");
161  return -1;
162  }
163 
164  if (_display and save_to_evt_disp)
165  _display->addEvent(track->getGenFitTrack());
166 
167  return 0;
168 }
