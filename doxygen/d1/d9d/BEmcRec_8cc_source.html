264 int BEmcRec::FindClusters()
265 {
266  // Cluster search algorithm based on Lednev's one developed for GAMS.
267  // Returns -1 if fgMaxLen parameter is too small (just increase it)
268 
269  int nhit, nCl;
270  int LenCl[fgMaxLen];
271  int next, ib, ie, iab, iae, last, LastCl, leng, ich;
272  int ia = 0;
273 
274  EmcModule* vv;
275  EmcModule *vhit, *vt;
276  EmcCluster Clt(this);
277  vector<EmcModule>::iterator ph;
278  vector<EmcModule> hl;
279 
280  (*fClusters).clear();
281  nhit = (*fModules).size();
282 
283  if (nhit <= 0) return 0;
284  if (nhit == 1)
285  {
286  Clt.ReInitialize((*fModules));
287  fClusters->push_back(Clt);
288  return 1;
289  }
290 
291  vt = new EmcModule[nhit];
292  vhit = new EmcModule[nhit];
293 
294  ph = (*fModules).begin();
295  vv = vhit;
296  while (ph != (*fModules).end()) *vv++ = *ph++;
297 
298  qsort(vhit, nhit, sizeof(EmcModule), HitNCompare);
299 
300  nCl = 0;
301  next = 0;
302  for (ich = 1; ich < nhit + 1; ich++)
303  {
304  if (ich < nhit) ia = vhit[ich].ich;
305 
306  // New subcluster
307  //
308  if ((ia - vhit[ich - 1].ich > 1) // the beginning of new subcluster
309  || (ich >= nhit) // just finish defining last sub-cluster
310  || (ia - ia / fNx * fNx == 0))
311  { // new raw -> new subcluster
312 
313  ib = next;
314  ie = ich - 1;
315  next = ich;
316  if (nCl >= fgMaxLen)
317  {
318  delete[] vhit;
319  delete[] vt;
320  return -1;
321  }
322  nCl++;
323  LenCl[nCl - 1] = next - ib;
324  if (nCl > 1)
325  {
326  // Job to glue the subclusters with common edge
327  //
328  iab = vhit[ib].ich; // The last subcl begin
329  iae = vhit[ie].ich; // The last subcl end
330  last = ib - 1; // The prelast subcl end
331  LastCl = nCl - 2;
332  for (int iCl = LastCl; iCl >= 0; iCl--)
333  {
334  leng = LenCl[iCl];
335 
336  if (iab - vhit[last].ich > fNx) goto new_ich;
337  for (int ichc = last; ichc >= last - leng + 1; ichc--)
338  {
339  // if( iab-vhit[ichc].ich > fNx ) goto new_icl; // From PHENIX version !!! This may be not right for complicated clusters, where tower ordering is not conserved
340 
341  // if( iae-vhit[ichc].ich >= fNx // From PHENIX version
342  if ((vhit[ichc].ich + fNx <= iae && vhit[ichc].ich + fNx >= iab) || (bCYL && (iae % fNx == fNx - 1) && (iae - vhit[ichc].ich == fNx - 1)) // Only for CYLinder geom !!!!
343  )
344  {
345  // Swap iCl-cluster towers (of length "leng") with whatever was between it and the last subcluster (of length "ib-1-last") - to make it adjecent to the last subcluster
346  CopyVector(&vhit[last + 1 - leng], vt, leng);
347  CopyVector(&vhit[last + 1], &vhit[last + 1 - leng], ib - 1 - last);
348  CopyVector(vt, &vhit[ib - leng], leng);
349 
350  // Now the number of clusters is reduced by 1 and the length of the last one increased by iCl-cluster length "leng"
351  for (int i = iCl; i < nCl - 2; i++) LenCl[i] = LenCl[i + 1];
352  ib -= leng;
353  LenCl[nCl - 2] = LenCl[nCl - 1] + leng;
354  nCl--;
355  goto new_icl;
356  }
357  } // for( int ichc=last
358 
359  new_icl:
360  last = last - leng;
361  } // for( int iCl=LastCl
362 
363  } // if( nCl > 1
364 
365  } // if( (ia-vhit
366 
367  new_ich:
368  continue;
369  } // for( ich=1
370 
371  if (nCl > 0)
372  {
373  ib = 0;
374  for (int iCl = 0; iCl < nCl; iCl++)
375  {
376  leng = LenCl[iCl];
377  hl.clear();
378  for (ich = 0; ich < leng; ich++) hl.push_back(vhit[ib + ich]);
379  Clt.ReInitialize(hl);
380  ib += LenCl[iCl];
381  fClusters->push_back(Clt);
382  }
383  }
384  delete[] vhit;
385  delete[] vt;
386 
387  return nCl;
388 }

489 void BEmcRec::Gamma(int nh, EmcModule* phit0, float* pchi, float* pchi0,
490  //void BEmcRec::Gamma(int nh, EmcModule* phit, float* pchi, float* pchi0,
491  float* pe1, float* px1, float* py1, float* pe2,
492  float* px2, float* py2, int& ndf)
493 {
494  // Tests for 1 or 2 photon hypothesis by minimizing the chi2.
495  // If the energy of one of two showers is less then fgMinShowerEnergy
496  // they are merged
497  //
498  // Returns two shower parameters (pe1,px1,py1,pe2,px2,py2).
499  // If one shower hypothesis accepted -> *pe2=0
500  // *pchi - chi2 after splitting
501  // *pchi0 - chi2 before splitting
502  // ndf contains number of degrees of freedom (not equal to number of cluster
503  // modules for PbGl).
504 
505  float e1, x1, y1, e2, x2, y2;
506  float chi, chi0, chisq0, chisave;
507  int dof;
508  float x0, y0;
509  float stepx, stepy, parx, pary;
510  const float dxy = 0.06;
511  const float stepmin = 0.01;
512  const float zTG = 100;
513  const float xmcut = 0.0015; // (GeV), for overlapping showers separation
514 
515  *pe1 = 0;
516  *px1 = 0;
517  *py1 = 0;
518  *pe2 = 0;
519  *px2 = 0;
520  *py2 = 0;
521  if (nh <= 0) return;
522 
523  EmcModule* phit = new EmcModule[nh];
524  int ish = ShiftX(0, nh, phit0, phit);
525  if (ish < -fNx)
526  {
527  delete[] phit;
528  return;
529  }
530 
531  // int ish = 0;
532  float xx, yy, xy; // Not used anywhere
533  Momenta(nh, phit, &e1, &x1, &y1, &xx, &yy, &xy);
534  *pe1 = e1;
535 
537  // Until this is proved working and useful (and ClusterChisq() properly tuned) !!!!!
538  /*
539  *px1=x1-ish;
540  *py1=y1;
541  delete [] phit;
542  return;
543  */
545 
546  if (e1 <= 0)
547  {
548  delete[] phit;
549  return;
550  }
551  // if( e1 <= 0 ) return;
552 
553  SetProfileParameters(0, e1, x1, y1);
554 
555  chisave = *pchi;
556  chi = *pchi;
557  // ClusterChisq parameter list changed MV 28.01.00
558  chi0 = ClusterChisq(nh, phit, e1, x1, y1, ndf);
559 
560  chisq0 = chi0;
561  dof = ndf; // nh->ndf MV 28.01.00
562 
563  // ndf=0 means the cluster's chi2 cannot be found; in this case chi0=0.
564  if (dof < 1) dof = 1;
565  chi = chisq0 / dof;
566  x0 = x1;
567  y0 = y1;
568  for (;;)
569  {
570  double chir = ClusterChisq(nh, phit, e1, x0 + dxy, y0, ndf);
571  double chil = ClusterChisq(nh, phit, e1, x0 - dxy, y0, ndf);
572  double chiu = ClusterChisq(nh, phit, e1, x0, y0 + dxy, ndf);
573  double chid = ClusterChisq(nh, phit, e1, x0, y0 - dxy, ndf);
574 
575  if ((chi0 > chir) || (chi0 > chil))
576  {
577  stepx = dxy;
578  if (chir > chil) stepx = -stepx;
579  }
580  else
581  {
582  stepx = 0;
583  parx = chir + chil - 2 * chi0;
584  if (parx > 0) stepx = -dxy * (chir - chil) / 2 / parx;
585  }
586 
587  if ((chi0 > chiu) || (chi0 > chid))
588  {
589  stepy = dxy;
590  if (chiu > chid) stepy = -stepy;
591  }
592  else
593  {
594  stepy = 0;
595  pary = chiu + chid - 2 * chi0;
596  if (pary > 0) stepy = -dxy * (chiu - chid) / 2 / pary;
597  }
598  if ((EmcCluster::ABS(stepx) < stepmin) && (EmcCluster::ABS(stepy) < stepmin)) break;
599  double chi00 = ClusterChisq(nh, phit, e1, x0 + stepx, y0 + stepy, ndf);
600 
601  if (chi00 >= chi0) break;
602  chi0 = chi00;
603  x0 += stepx;
604  y0 += stepy;
605  }
606  if (chi0 < chisq0)
607  {
608  x1 = x0;
609  y1 = y0;
610  chi = chi0 / dof;
611  }
612 
613  *pchi0 = chi;
614  *pchi = chi;
615  *py1 = y1;
616  x1 -= float(ish);
617  while (x1 < -0.5) x1 += float(fNx);
618  while (x1 >= fNx - 0.5) x1 -= float(fNx);

1232  }
1233  grc = sqrt(grec * grec + grxc * grxc + gryc * gryc);

1235  if (iter > 0)
1236  {

1238  scal = EmcCluster::ABS(gr / grc - cosi);
1239  if (scal < 0.1) scal = 0.1;
1240  step /= scal;
1241  }

1243  eps0 = epsc;
1244  dx0 = dxc;
1245  dy0 = dyc;
1246  gre = grec;
1247  grx = grxc;
1248  gry = gryc;
1249  gr = grc;
1250  }
1251  epsc = eps0 - step * gre / gr;
1252  while (EmcCluster::ABS(epsc) >= epsmax)
1253  {
1254  step /= 2;
1255  epsc = eps0 - step * gre / gr;
1256  }
1257  dxc = dx0 - step * grx / gr;
1258  dyc = dy0 - step * gry / gr;
1259  if (step * gr < stpmin) break;
1260  }
1261  if ((*pchi) * nh - chisq2 < delch) return;
1262  dof = nh;
1263  if (dof < 1) dof = 1;
1264  *pchi = chisq2 / dof;
