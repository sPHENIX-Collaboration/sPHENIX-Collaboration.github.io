274 std::pair<G4LogicalVolume *, G4Transform3D>
275 PHG4SpacalDetector::Construct_AzimuthalSeg()
276 {
277  G4Tubs *sec_solid = new G4Tubs(G4String(GetName() + string("_sec")),
278  _geom->get_radius() * cm, _geom->get_max_radius() * cm,
279  _geom->get_length() * cm / 2.0, 0, twopi / _geom->get_azimuthal_n_sec());
280 
281  G4Material *cylinder_mat = G4Material::GetMaterial(_geom->get_absorber_mat());
282  assert(cylinder_mat);
283 
284  G4LogicalVolume *sec_logic = new G4LogicalVolume(sec_solid, cylinder_mat,
285  G4String(G4String(GetName() + string("_sec"))), 0, 0, nullptr);
286 
287  GetDisplayAction()->AddVolume(sec_logic, "AzimuthSegment");
288 
289  const double fiber_length = _geom->get_thickness() * cm - 2 * _geom->get_fiber_outer_r() * cm;
290  G4LogicalVolume *fiber_logic = Construct_Fiber(fiber_length, string(""));
291 
292  int fiber_count = 0;
293  // double z_step = _geom->get_fiber_distance() * cm * sqrt(3) / 2.;
294  double z_step = _geom->get_z_distance() * cm;
295  G4double z = _geom->get_zmin() * cm - _geom->get_zpos() * cm + z_step;
296  while (z < _geom->get_zmax() * cm - _geom->get_zpos() * cm - z_step)
297  {
298  const double rot = twopi / _geom->get_azimuthal_n_sec() * ((fiber_count % 2 == 0) ? 1. / 4. : 3. / 4.);
299 
300  G4Transform3D fiber_place(
301  G4RotateZ3D(rot) * G4TranslateZ3D(z) * G4TranslateX3D(_geom->get_half_radius() * cm) * G4RotateY3D(halfpi));
302 
303  stringstream name;
304  name << GetName() << "_fiber_" << fiber_count;
305 
306  G4PVPlacement *fiber_physi = new G4PVPlacement(fiber_place, fiber_logic,
307  G4String(name.str()), sec_logic, false, fiber_count,
308  OverlapCheck());
309  fiber_vol[fiber_physi] = fiber_count;
310  assert(gdml_config);
311  gdml_config->exclude_physical_vol(fiber_physi);
312 
313  z += z_step;
314  fiber_count++;
315  }
316  _geom->set_nscint(fiber_count);
317 
318  if (Verbosity() > 0)
319  {
320  cout << "PHG4SpacalDetector::Construct_AzimuthalSeg::" << GetName()
321  << " - constructed " << fiber_count << " fibers" << endl;
322  cout << "\t"
323  << "_geom->get_fiber_distance() = " << _geom->get_fiber_distance()
324  << endl;
325  cout << "\t"
326  << "fiber_length = " << fiber_length / cm << endl;
327  cout << "\t"
328  << "z_step = " << z_step << endl;
329  cout << "\t"
330  << "_geom->get_azimuthal_bin() = " << _geom->get_azimuthal_n_sec()
331  << endl;
332  cout << "\t"
333  << "_geom->get_azimuthal_distance() = "
334  << _geom->get_azimuthal_distance() << endl;
335  cout << "\t"
336  << "_geom->is_virualize_fiber() = " << _geom->is_virualize_fiber()
337  << endl;
338  }
339 
340  return make_pair(sec_logic, G4Transform3D::Identity);
341 }

344 PHG4SpacalDetector::Construct_Fiber(const G4double length, const string &id)
345 {
346  G4Tubs *fiber_solid = new G4Tubs(G4String(GetName() + string("_fiber") + id),
347  0, _geom->get_fiber_outer_r() * cm, length / 2.0, 0, twopi);
348 
349  G4Material *clading_mat = G4Material::GetMaterial(

351  assert(clading_mat);
352 
353  G4LogicalVolume *fiber_logic = new G4LogicalVolume(fiber_solid, clading_mat,
354  G4String(G4String(GetName() + string("_fiber") + id)), 0, 0,
355  nullptr);
356 
357  {
358  GetDisplayAction()->AddVolume(fiber_logic, "Fiber");
359  }
360 
361  G4Tubs *core_solid = new G4Tubs(
362  G4String(GetName() + string("_fiber_core") + id), 0,
363  _geom->get_fiber_core_diameter() * cm / 2, length / 2.0, 0, twopi);
364 
365  G4Material *core_mat = G4Material::GetMaterial(_geom->get_fiber_core_mat());
366  assert(core_mat);
367 
368  G4LogicalVolume *core_logic = new G4LogicalVolume(core_solid, core_mat,
369  G4String(G4String(GetName() + string("_fiber_core") + id)), 0, 0,
370  fiber_core_step_limits);
371 
372  {
373  GetDisplayAction()->AddVolume(core_logic, "FiberCore");
374  }
375 
376  const bool overlapcheck_fiber = OverlapCheck() and (Verbosity() >= 3);
377  G4PVPlacement *core_physi = new G4PVPlacement(0, G4ThreeVector(), core_logic,
378  G4String(G4String(GetName() + string("_fiber_core") + id)), fiber_logic,
379  false, 0, overlapcheck_fiber);
380  fiber_core_vol[core_physi] = 0;
381 
382  return fiber_logic;
383 }
