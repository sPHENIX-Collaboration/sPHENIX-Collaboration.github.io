117 
118  double tau = 410.1e-15; // D0 life time in seconds
119  double cc = GSL_CONST_CGSM_SPEED_OF_LIGHT; // speed of light cm/sec
120  double ctau = tau * cc * 1.0e+04; // ctau in micrometers
121 
122  // If not reusing existing vertex Randomly generate vertex position in z
123  if (!ReuseExistingVertex(topNode))
124  {
125  if (vtx_zmax != vtx_zmin)
126  {
127  vtx_z = (vtx_zmax - vtx_zmin) * gsl_rng_uniform_pos(RandomGenerator) + vtx_zmin;
128  }
129  else
130  {
131  vtx_z = vtx_zmin;
132  }
133  }
134  // taken randomly from a fitted pT distribution to Pythia Upsilons
135 
136  double pt = 0.0;
137  if (pt_max != pt_min)
138  {
139  pt = fpt->GetRandom();
140  }
141  else
142  {
143  pt = pt_min;
144  }
145 
146  // taken randomly from a fitted rapidity distribution to Pythia Upsilons
147 
148  double y = 0.0;
149  if (y_max != y_min)
150  {
151  y = frap->GetRandom();
152  }
153  else
154  {
155  y = y_min;
156  }
157  // 0 and 2*M_PI identical, so use gsl_rng_uniform which excludes 1.0
158  double phi = (2.0 * M_PI) * gsl_rng_uniform(RandomGenerator);
159 
160  double mnow = mass;
161 
162  // Get the pseudorapidity, eta, from the rapidity, mass and pt
163 
164  double mt = sqrt(pow(mnow, 2) + pow(pt, 2));
165  double eta = asinh(sinh(y) * mt / pt);
166 
167  // Put it in a TLorentzVector
168 
169  TLorentzVector vd0;
170  vd0.SetPtEtaPhiM(pt, eta, phi, mnow);
171

173  double gamma = vd0.Gamma();
174  double lifetime = gsl_ran_exponential(RandomGenerator, 410.1e-03) * 1.0e-12; // life time in seconds
175  double lifepath = lifetime * gamma * beta * cc; // path in cm
176  if (Verbosity() > 0)
177  {
178  cout << "D0 px,py,pz: " << vd0.Px() << " " << vd0.Py() << " " << vd0.Pz() << " " << beta << " " << gamma << endl;
179  cout << " ctau = " << ctau << " " << lifetime << " " << lifepath << " " << lifepath * 1.0e+04 << endl;
180  }
181 
182  vtx_x = vd0.Px() / vd0.P() * lifepath;
183  vtx_y = vd0.Py() / vd0.P() * lifepath;
184  vtx_z = vtx_z + vd0.Pz() / vd0.P() * lifepath;
185  t0 = lifetime;
186  int vtxindex = ineve->AddVtx(vtx_x, vtx_y, vtx_z, t0);
187  if (Verbosity() > 0)
188  {
189  cout << " XY vertex: " << sqrt(vtx_x * vtx_x + vtx_y * vtx_y) << " " << sqrt(vtx_x * vtx_x + vtx_y * vtx_y) * 1.0e+04 << endl;
190  }
191 
192  // Now decay it
193  // Get the decay energy and momentum in the frame of the D0 - this correctly handles decay particles of any mass.
194 
195  double E1 = (pow(mnow, 2) - pow(m2, 2) + pow(m1, 2)) / (2.0 * mnow);
196  double p1 = sqrt((pow(mnow, 2) - pow(m1 + m2, 2)) * (pow(mnow, 2) - pow(m1 - m2, 2))) / (2.0 * mnow);
197 
198  // In the frame of the D0, get a random theta and phi angle for particle 1
199  // Assume angular distribution in the frame of the decaying D0 that is uniform in phi and goes as sin(theta) in theta
200  // particle 2 has particle 1 momentum reflected through the origin
201 
202  double th1 = fsin->GetRandom();
203  double phi1 = 2.0 * M_PI * gsl_rng_uniform_pos(RandomGenerator);
204 
205  // Put particle 1 into a TLorentzVector
206 
207  double px1 = p1 * sin(th1) * cos(phi1);
208  double py1 = p1 * sin(th1) * sin(phi1);
209  double pz1 = p1 * cos(th1);
210  TLorentzVector v1;
211  v1.SetPxPyPzE(px1, py1, pz1, E1);
212 
213  // now boost the decay product v1 into the lab using a vector consisting of the beta values of the vector meson
214  // where p/E is v/c if we use GeV/c for p and GeV for E
215 
216  double betax = vd0.Px() / vd0.E();
217  double betay = vd0.Py() / vd0.E();
218  double betaz = vd0.Pz() / vd0.E();
219  v1.Boost(betax, betay, betaz);
220 
221  // The second decay product's lab vector is the difference between the original D0 and the boosted decay product 1
222 
223  TLorentzVector v2 = vd0 - v1;
224
