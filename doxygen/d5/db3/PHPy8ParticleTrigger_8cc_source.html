80  if (_doAbsEtaLowCut && abs(pythia->event[i].eta()) < _theEtaLow) continue;
81  if (_doAbsEtaHighCut && abs(pythia->event[i].eta()) > _theEtaHigh) continue;
82 
83  if (_doBothPtCut && (pythia->event[i].pT() < _thePtLow ||
84  pythia->event[i].pT() > _thePtHigh)) continue;
85  if (_doPtHighCut && pythia->event[i].pT() > _thePtHigh) continue;
86  if (_doPtLowCut && pythia->event[i].pT() < _thePtLow) continue;
87 
88  if (_doBothPCut && (pythia->event[i].pAbs() < _thePLow ||
89  pythia->event[i].pAbs() > _thePHigh)) continue;
90  if (_doPHighCut && pythia->event[i].pAbs() > _thePHigh) continue;

92 
93  if (_doBothPzCut && (pythia->event[i].pz() < _thePzLow ||
94  pythia->event[i].pz() > _thePzHigh)) continue;
95  if (_doPzHighCut && pythia->event[i].pz() > _thePzHigh) continue;
96  if (_doPzLowCut && pythia->event[i].pz() < _thePzLow) continue;
97 
98  if (Verbosity() > 5)
99  {
100  cout << "stable " << pythia->event[i].id()
101  << " pt: " << pythia->event[i].pT()
102  << " pz: " << pythia->event[i].pz()
103  << " p: " << pythia->event[i].pAbs()
104  << " eta: " << pythia->event[i].eta() << endl;
105  }
106 
107  // loop over all partents to this particle
108  bool passedParents = false;
109  for (int k = 0; k < int(_theParents.size()); k++)
110  {
111  // check Mothers
112  std::vector<int> moms = pythia->event[i].motherList();
113  for (int m = 0; m < int(moms.size()); m++)
114  {
115  if (abs(pythia->event[moms[m]].id()) == abs(_theParents[k]))
116  {
117  passedParents = true;
118  if (Verbosity() > 5) cout << "found parent!" << endl;
119  break;
120  }
121  } //moms for loop
122  if (passedParents) break;
123  } //parents for loop
124 
125  //If we made it here and it passes parents, success!
126  if (_theParents.size() == 0 || passedParents) return true;
127 
128  } //if _theParticles
129  } //_theParticles for loop
130 
131  } //pythia event for loop
132 
133  return false;
