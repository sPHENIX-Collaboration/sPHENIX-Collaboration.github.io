85 bool PHG4mRICHSteppingAction::UserSteppingAction( const G4Step* aStep, bool )
86 {
87  G4TouchableHandle touch = aStep->GetPreStepPoint()->GetTouchableHandle();
88  G4VPhysicalVolume* volume = touch->GetVolume();
89 
90  int isactive = detector_->IsInmRICH(volume);
91  if ( isactive > PHG4mRICHDetector::INACTIVE )
92  {
93  // collect energy and track length step by step
94  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
95  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
96 
97  const G4Track* aTrack = aStep->GetTrack();
98 
99  /* Check if particle is 'geantino' */
100  bool geantino = false;
101  if (aTrack->GetParticleDefinition()->GetPDGEncoding() == 0 && aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos)
102  {
103  geantino = true;
104  }
105 
106  // cout << "Name of volume: " << volume->GetName() << ", isactive = " << isactive << endl;
107  // G4VPhysicalVolume* v1 = touch->GetVolume(1);
108  // cout << "Name of mother volume: " << v1->GetName() << endl;
109  // G4VPhysicalVolume* v2 = touch->GetVolume(2);
110  // cout << "Name of grand mother volume: " << v2->GetName() << endl;
111 
112  // cout << "copyNum = " << touch->GetReplicaNumber() << ", Or = " << touch->GetReplicaNumber(1) << ", Or = " << touch->GetReplicaNumber(2) << ", Or = " << touch->GetReplicaNumber(3) << endl;
113  // int module_id=GetModuleID(touch->GetVolume(2)); // use mother volume to determine module_id
114  int module_id = touch->GetReplicaNumber(2)-1; // use copy number of mother volume to determine module_id
115  int PID=aTrack->GetDefinition()->GetPDGEncoding();
116  string PName = aTrack->GetDefinition()->GetParticleName();
117 
118  //-----------------------------------------------------------------------------------//
119  // if this block stops everything, just put all kinetic energy into edep
120  if (IsBlackHole)
121  {
122  edep = aTrack->GetKineticEnergy() / GeV;
123  G4Track* killtrack = const_cast<G4Track *> (aTrack);
124  killtrack->SetTrackStatus(fStopAndKill);
125  }
126 
127  /* Get Geant4 pre- and post-step points */
128  G4StepPoint * prePoint = aStep->GetPreStepPoint();
129  G4StepPoint * postPoint = aStep->GetPostStepPoint();
130 
131  switch (prePoint->GetStepStatus())
132  {
133  //-----------------
134  case fGeomBoundary:
135  //-----------------
136  case fUndefined:
137  if (! hit) hit = new PHG4Hitv1();
138  /* Set hit location (space point) */
139  hit->set_x( 0, prePoint->GetPosition().x() / cm);
140  hit->set_y( 0, prePoint->GetPosition().y() / cm );
141  hit->set_z( 0, prePoint->GetPosition().z() / cm );
142 
143  /* Set hit time */
144  hit->set_t( 0, prePoint->GetGlobalTime() / nanosecond );
145 
146  //set the track ID
147  hit->set_trkid(aTrack->GetTrackID());
148  savetrackid = aTrack->GetTrackID();
149 
150  /* set intial energy deposit */
151  hit->set_edep( 0 );
152  hit->set_eion( 0 );
153  if (isactive == PHG4mRICHDetector::SENSOR)
154  {
155  savehitcontainer = hits_;
156  if(PID == 0)
157  {
158  edep = aTrack->GetKineticEnergy() / GeV;
159  G4Track* killtrack = const_cast<G4Track *> (aTrack);
160  killtrack->SetTrackStatus(fStopAndKill);
161  hit->set_scint_id(module_id);
162  }
163  }
164  else
165  {
166  savehitcontainer = absorberhits_;
167  }
168 
169  // here we set what is common for scintillator and absorber hits
170  if ( G4VUserTrackInformation* p = aTrack->GetUserInformation() )
171  {
172  if ( PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p) )
173  {
174  hit->set_trkid(pp->GetUserTrackId());
175  hit->set_shower_id(pp->GetShower()->get_id());
176  saveshower = pp->GetShower();
177  }
178  }
179  break;
180  //-----------------
181  default:
182  break;
183  }
184 
185  // some sanity checks for inconsistencies
186  // check if this hit was created, if not print out last post step status
187  if (!hit || !isfinite(hit->get_x(0)))
188  {
189  cout << GetName() << ": hit was not created" << endl;
190  cout << "prestep status: " << prePoint->GetStepStatus()
191  << ", last post step status: " << savepoststepstatus << endl;
192  gSystem->Exit(1);
193  }
194  savepoststepstatus = postPoint->GetStepStatus();
195 
196  // check if track id matches the initial one when the hit was created
197  if (aTrack->GetTrackID() != savetrackid)
198  {
199  cout << GetName() << ": hits do not belong to the same track" << endl;
200  cout << "saved track: " << savetrackid
201  << ", current trackid: " << aTrack->GetTrackID()
202  << endl;
203  gSystem->Exit(1);
204  }
205 
206  //-----------------------------------------------------------------------------------//
207  /* Update exit values- will be overwritten with every step until
208  * we leave the volume or the particle ceases to exist */
209  hit->set_x( 1, postPoint->GetPosition().x() / cm );
210  hit->set_y( 1, postPoint->GetPosition().y() / cm );
211  hit->set_z( 1, postPoint->GetPosition().z() / cm );
212 
213  hit->set_t( 1, postPoint->GetGlobalTime() / nanosecond );
214 
215  /* sum up the energy to get total deposited */
216  hit->set_edep(hit->get_edep() + edep);
217  hit->set_eion(hit->get_eion() + eion);
218 
219  if (geantino)
220  {
221  hit->set_edep(-1); // only energy=0 g4hits get dropped, this way geantinos survive the g4hit compression
222  hit->set_eion(-1);
223  }
224  if (hit->get_edep() && PID == 0)
225  {
226  if ( G4VUserTrackInformation* p = aTrack->GetUserInformation() )
227  {
228  if ( PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p) )
229  {
230  pp->SetKeep(1); // we want to keep the track
231  }
232  }
233  }
234 
235  //-----------------------------------------------------------------------------------//
236  // if any of these conditions is true this is the last step in
237  // this volume and we need to save the hit
238  if (postPoint->GetStepStatus() == fGeomBoundary ||
239  postPoint->GetStepStatus() == fWorldBoundary ||
240  postPoint->GetStepStatus() == fAtRestDoItProc ||
241  aTrack->GetTrackStatus() == fStopAndKill)
242  {
243  // save only hits with energy deposit (or -1 for geantino)
244  if (hit->get_edep() && PID==0)
245  {
246  savehitcontainer->AddHit(module_id, hit);
247  if (saveshower)
248  {
249  saveshower->add_g4hit_id(hits_->GetID(), hit->get_hit_id());
250  }
251  // ownership has been transferred to container, set to null
252  // so we will create a new hit for the next track
253  hit = nullptr;
254  }
255  else
256  {
257  // if this hit has no energy deposit, just reset it for reuse
258  // this means we have to delete it in the dtor. If this was
259  // the last hit we processed the memory is still allocated
260  hit->Reset();
261  }
262  }
263  return true;
264  }
265  else
266  {
267  return false;
268  }

303 int PHG4mRICHSteppingAction::GetModuleID(G4VPhysicalVolume* volume)
304 {
305  // G4AssemblyVolumes naming convention:
306  // av_WWW_impr_XXX_YYY_ZZZ
307  // where:
308  // WWW - assembly volume instance number
309  // XXX - assembly volume imprint number
310  // YYY - the name of the placed logical volume
311  // ZZZ - the logical volume index inside the assembly volume
312  // e.g. av_1_impr_4_mRICH_module_pv_11
313  // 4 is the number of the mRICH sector volume
314  // mRICH_module: name of mRICH module
315  // 11: number of mRICH module logical volume
316  // use boost tokenizer to separate the _, then take value
317  // after "impr" for mRICH sector volume and after "pv" for mRICH module volume
318  // use boost lexical cast for string -> int conversion
319  int module_id = -1;
320  int sector_id = -1;
321  int mRICH_id = -1;
322 
323  boost::char_separator<char> sep("_");
324  boost::tokenizer<boost::char_separator<char>> tok(volume->GetName(), sep);
325  boost::tokenizer<boost::char_separator<char>>::const_iterator tokeniter;
326 
327  for (tokeniter = tok.begin(); tokeniter != tok.end(); ++tokeniter) {
328  if (*tokeniter == "impr") {
329  ++tokeniter;
330  if (tokeniter != tok.end()) {
331  sector_id = boost::lexical_cast<int>(*tokeniter);
332  }
333  else {
334  cout << PHWHERE << " Error parsing " << volume->GetName()
335  << " for mRICH sector id " << endl;
336  gSystem->Exit(1);
337  }
338  }
339  else if (*tokeniter == "pv")
340  {
341  ++tokeniter;
342  if (tokeniter != tok.end())
343  {
344  mRICH_id = boost::lexical_cast<int>(*tokeniter);
345  }
346  else
347  {
348  cout << PHWHERE << " Error parsing " << volume->GetName()
349  << " for mRICH id " << endl;
350  gSystem->Exit(1);
351  }
352  }
353  }
354 
355  module_id = (sector_id-1)*100+mRICH_id;
356 
357  // cout << "name of volume: " << volume->GetName() << ", sector_id = " << sector_id << ", mRICH_id = " << mRICH_id << ", module_id = " << module_id << endl;
358  // cout << endl;
359 
360  return module_id;
361 }
