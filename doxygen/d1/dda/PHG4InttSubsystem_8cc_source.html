105  dstNode->addNode(DetNode);
106  }
107  std::string nodename = (SuperDetector() != "NONE") ? boost::str(boost::format("G4HIT_%s") % SuperDetector()) : boost::str(boost::format("G4HIT_%s") % m_DetectorType);
108 
109  // create hit list
110  PHG4HitContainer *hitcontainer = findNode::getClass<PHG4HitContainer>(topNode, nodename.c_str());
111  if (!hitcontainer)

113 
114  if (absorberactive)
115  {
116  nodename = (SuperDetector() != "NONE") ? boost::str(boost::format("G4HIT_ABSORBER_%s") % SuperDetector()) : boost::str(boost::format("G4HIT_ABSORBER_%s") % m_DetectorType);
117 
118  hitcontainer = findNode::getClass<PHG4HitContainer>(topNode, nodename.c_str());
119  if (!hitcontainer)
120  {
121  DetNode->addNode(new PHIODataNode<PHObject>(hitcontainer = new PHG4HitContainer(nodename), nodename.c_str(), "PHObject"));
122  }
123  }
124 
125  // create stepping action
126  m_SteppingAction = new PHG4InttSteppingAction(m_Detector, GetParamsContainer(), layer_begin_end);
127  }
128  else
129  {
130  if (blackhole)
131  {
132  m_SteppingAction = new PHG4InttSteppingAction(m_Detector, GetParamsContainer(), layer_begin_end);
133  }
134  }
135 
136  return 0;
137 }
