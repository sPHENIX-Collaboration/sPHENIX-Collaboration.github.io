249 int Fun4AllDstInputManager::SyncIt(const SyncObject *mastersync)
250 {
251  if (!mastersync)
252  {
253  cout << PHWHERE << Name() << " No MasterSync object, cannot perform synchronization" << endl;
254  cout << "Most likely your first file does not contain a SyncObject and the file" << endl;
255  cout << "opened by the Fun4AllDstInputManager with Name " << Name() << " has one" << endl;
256  cout << "Change your macro and use the file opened by this input manager as first input" << endl;
257  cout << "and you will be okay. Fun4All will not process the current configuration" << endl
258  << endl;
259  return Fun4AllReturnCodes::SYNC_FAIL;
260  }
261  int iret = syncobject->Different(mastersync);
262  if (iret) // what to do if files are not in sync
263  {
264  if (mastersync->EventNumber() == -999999) // first file does not contain sync object
265  {
266  cout << PHWHERE << " Mastersync not filled, your first file does not contain a SyncObject" << endl;
267  cout << "This Event will not be processed further" << endl;
268  }
269  else // okay try to resync here
270  {
271  if (Verbosity() > 3)
272  {
273  cout << "Need to Resync, mastersync evt no: " << mastersync->EventNumber()
274  << ", this Event no: " << syncobject->EventNumber() << endl;
275  cout << "mastersync evt counter: " << mastersync->EventCounter()
276  << ", this Event counter: " << syncobject->EventCounter() << endl;
277  cout << "mastersync run number: " << mastersync->RunNumber()
278  << ", this run number: " << syncobject->RunNumber() << endl;
279  }
280  while (syncobject->RunNumber() < mastersync->RunNumber())
281 
282  {
283  events_skipped_during_sync++;
284  if (Verbosity() > 2)
285  {
286  cout << Name() << " Run Number: " << syncobject->RunNumber()
287  << ", master: " << mastersync->RunNumber()
288  << endl;
289  }
290  int iret = ReadNextEventSyncObject();
291  if (iret)
292  {
293  return iret;
294  }
295  }
296  int igood = 0;
297  if (syncobject->RunNumber() == mastersync->RunNumber())
298  {
299  igood = 1;
300  }
301  // only run up the Segment Number if run numbers are identical
302  while (syncobject->SegmentNumber() < mastersync->SegmentNumber() && igood)
303  {
304  events_skipped_during_sync++;
305  if (Verbosity() > 2)
306  {
307  cout << Name() << " Segment Number: " << syncobject->SegmentNumber()
308  << ", master: " << mastersync->SegmentNumber()
309  << endl;
310  }
311  int iret = ReadNextEventSyncObject();
312  if (iret)
313  {
314  return iret;
315  }
316  }
317  // only run up the Event Counter if run number and segment number are identical
318  if (syncobject->SegmentNumber() == mastersync->SegmentNumber() && syncobject->RunNumber() == mastersync->RunNumber())
319  {
320  igood = 1;
321  }
322  else
323  {
324  igood = 0;
325  }
326  while (syncobject->EventCounter() < mastersync->EventCounter() && igood)
327  {
328  events_skipped_during_sync++;
329  if (Verbosity() > 2)
330  {
331  cout << Name()
332  << ", EventCounter: " << syncobject->EventCounter()
333  << ", master: " << mastersync->EventCounter()
334  << endl;
335  }
336  int iret = ReadNextEventSyncObject();
337  if (iret)
338  {
339  return iret;
340  }
341  }
342  // Since up to here we only read the sync object we need to push
343  // the current event back inot the root file (subtract one from the
344  // local root file event counter) so we can read the full event
345  // if it syncs, if it does not sync we also read one event too many
346  // (otherwise we cannot determine that we are "too far")
347  // and also have to push this one back
348  PushBackEvents(1);
349  if (syncobject->RunNumber() > mastersync->RunNumber() || // check if run number too large
350  syncobject->EventCounter() > mastersync->EventCounter() || // check if event counter too large
351  syncobject->SegmentNumber() > mastersync->SegmentNumber()) // check segment number too large
352  {
353  // the event from first file which determines the mastersync
354  // and which we are trying to find on this file does not exist on this file
355  // so: return failure. This will cause the input managers to read
356  // the next event from the input files file
357  return Fun4AllReturnCodes::SYNC_FAIL;
358  }
359  // Here the event counter and segment number and run number do agree - we found the right match
360  // now read the full event (previously we only read the sync object)
361  PHCompositeNode *dummy;
362  dummy = IManager->read(dstNode);
363  if (!dummy)
364  {
365  cout << PHWHERE << " " << Name() << " Could not read full Event" << endl;
366  cout << "PLEASE NOTIFY PHENIX-OFF-L and post the macro you used" << endl;
367  fileclose();
368  return Fun4AllReturnCodes::SYNC_FAIL;
369  }
370  int iret = syncobject->Different(mastersync); // final check if they really agree
371  if (iret) // if not things are severely wrong
372  {
373  cout << PHWHERE << " MasterSync and SyncObject of " << Name() << " are different" << endl;
374  cout << "This Event will not be processed further, here is some debugging info:" << endl;
375  cout << "PLEASE NOTIFY PHENIX-OFF-L and post the macro you used" << endl;
376  cout << "MasterSync->identify:" << endl;
377  mastersync->identify();
378  cout << Name() << ": SyncObject->identify:" << endl;
379  syncobject->identify();
380  return Fun4AllReturnCodes::SYNC_FAIL;
381  }
382  else if (Verbosity() > 3)
383  {
384  cout << PHWHERE << " Resynchronization successfull for " << Name() << endl;
385  cout << "MasterSync->identify:" << endl;
386  mastersync->identify();
387  cout << Name() << ": SyncObject->identify:" << endl;
388  syncobject->identify();
389  }
390  }
391  }
392  // else
393  // {
394  // cout << "No Need to Resync" << endl;
395  // }
396  return Fun4AllReturnCodes::SYNC_OK;

399 int Fun4AllDstInputManager::ReadNextEventSyncObject()
400 {
401 readnextsync:
402  static int readfull = 0;
403  if (!IManager) // in case the old file was exhausted and there is no new file opened
404  {
405  return Fun4AllReturnCodes::SYNC_FAIL;
406  }
407  if (syncbranchname.empty())
408  {
409  readfull = 1; // we need to read a full events to set the root branches to phool nodes right for a new file
410  map<string, TBranch *>::const_iterator bIter;
411  for (bIter = IManager->GetBranchMap()->begin(); bIter != IManager->GetBranchMap()->end(); ++bIter)
412  {
413  if (Verbosity() > 2)
414  {
415  cout << Name() << ": branch: " << bIter->first << endl;
416  }
417  string::size_type pos = bIter->first.find("/Sync");
418  if (pos != string::npos) // found it
419  {
420  syncbranchname = bIter->first;
421  break;
422  }
423  }
424  if (syncbranchname.empty())
425  {
426  cout << PHWHERE << "Could not locate Sync Branch" << endl;
427  cout << "Please check for it in the following list of branch names and" << endl;
428  cout << "PLEASE NOTIFY PHENIX-OFF-L and post the macro you used" << endl;
429  for (bIter = IManager->GetBranchMap()->begin(); bIter != IManager->GetBranchMap()->end(); ++bIter)
430  {
431  cout << bIter->first << endl;
432  }
433  return Fun4AllReturnCodes::SYNC_FAIL;
434  }
435  }
436  size_t EventOnDst = 0;
437  int itest = 0;
438  if (!readfull)
439  {
440  // if all files are exhausted, the IManager is deleted and set to nullptr
441  // so check if IManager is valid before getting a new event
442  if (IManager)
443  {
444  EventOnDst = IManager->getEventNumber(); // this returns the next number of the event
445  itest = IManager->readSpecific(EventOnDst, syncbranchname.c_str());
446  }
447  else
448  {
449  if (Verbosity() > 2)
450  {
451  cout << Name() << ": File exhausted while resyncing" << endl;
452  }
453  return Fun4AllReturnCodes::SYNC_FAIL;
454  }
455  }
456  else
457  {
458  if (IManager->read(dstNode))
459  {
460  itest = 1;
461  }
462  else
463  {
464  itest = 0;
465  }
466  }
467  if (!itest)
468  {
469  if (Verbosity() > 2)
470  {
471  cout << Name() << ": File exhausted while resyncing" << endl;
472  }
473  fileclose();
474  if (OpenNextFile())
475  {
476  return Fun4AllReturnCodes::SYNC_FAIL;
477  }
478  syncbranchname.clear(); // clear the sync branch name, who knows - it might be different on new file
479  goto readnextsync;
480  }
481  if (!readfull)
482  {
483  EventOnDst++;
484  IManager->setEventNumber(EventOnDst); // update event number in phool io manager
485  }
486  else
487  {
488  readfull = 0;
489  }
490  return 0;
491 }
