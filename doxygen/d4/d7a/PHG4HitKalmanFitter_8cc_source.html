252 int PHG4TrackFastSim::PseudoPatternRecognition(
253  const PHG4Particle* particle,
254  std::vector<PHGenFit::Measurement*>& meas_out, TVector3& seed_pos,
255  TVector3& seed_mom, TMatrixDSym& seed_cov, const bool do_smearing) {
256 
257 
258 
259 
260  seed_pos.SetXYZ(0, 0, 0);
261  seed_mom.SetXYZ(0, 0, 10);
262  seed_cov.ResizeTo(6, 6);
263 
264  for (int i = 0; i < 3; i++) {
265  seed_cov[i][i] = _phi_resolution * _phi_resolution;
266  }
267 
268  for (int i = 3; i < 6; i++) {
269  seed_cov[i][i] = 10;
270  }
271 
272  if (particle) {
273  TVector3 True_mom(particle->get_px(), particle->get_py(),
274  particle->get_pz());
275 
276  seed_mom.SetXYZ(particle->get_px(), particle->get_py(),
277  particle->get_pz());
278  if(do_smearing)
279  {
280  const double momSmear = 3. / 180. * TMath::Pi(); // rad
281  const double momMagSmear = 0.1; // relative
282 
283  seed_mom.SetPhi(gRandom->Gaus(True_mom.Phi(), momSmear));
284  seed_mom.SetTheta(gRandom->Gaus(True_mom.Theta(), momSmear));
285  seed_mom.SetMag(
286  gRandom->Gaus(True_mom.Mag(),
287  momMagSmear * True_mom.Mag()));
288  }
289  }
290 
291 
292  meas_out.clear();
293 
294  for(int ilayer=0; ilayer<_N_DETECTOR_LAYER; ilayer++) {
295 
296  if(!_phg4hits[ilayer]) {
297  LogError("No _phg4hits[i] found!");
298  continue;
299  }
300 
301  for (PHG4HitContainer::ConstIterator itr =
302  _phg4hits[ilayer]->getHits().first;
303  itr != _phg4hits[ilayer]->getHits().second; ++itr) {
304  PHG4Hit * hit = itr->second;
305  if(!hit) {
306  LogDebug("No PHG4Hit Found!");
307  continue;
308  }
309  if (hit->get_trkid() == particle->get_track_id() || gRandom->Uniform(0,1) < _pat_rec_nosise_prob) {
310  PHGenFit::Measurement* meas = PHG4HitToMeasurementVerticalPlane(hit);
311  if(gRandom->Uniform(0,1) <= _pat_rec_hit_finding_eff)
312  meas_out.push_back(meas);
313  }
314  }
315 
316  } /*Loop detector layers*/
317 
318  return Fun4AllReturnCodes::EVENT_OK;
319 }

321 SvtxTrack* PHG4TrackFastSim::MakeSvtxTrack(
322  const PHGenFit::Track* phgf_track) {
323 
324  double chi2 = phgf_track->get_chi2();
325  double ndf = phgf_track->get_ndf();
326 
327  genfit::MeasuredStateOnPlane* gf_state = phgf_track->extrapolateToPlane(TVector3(0.,0.,0.), TVector3(0.,0.,1.));
328  TVector3 mom = gf_state->getMom();
329  TVector3 pos = gf_state->getPos();
330  TMatrixDSym cov = gf_state->get6DCov();
331 
332 // SvtxTrack_v1* out_track = new SvtxTrack_v1(*static_cast<const SvtxTrack_v1*> (svtx_track));
333  SvtxTrack_v1* out_track = new SvtxTrack_v1();
334 
341  double dca2d = gf_state->getState()[3];
342  out_track->set_dca2d(dca2d);
343  out_track->set_dca2d_error(gf_state->getCov()[3][3]);
344  double dca3d = sqrt(
345  dca2d*dca2d +
346  gf_state->getState()[4]*gf_state->getState()[4]);
347  out_track->set_dca(dca3d);
348 
349 
350  out_track->set_chisq(chi2);
351  out_track->set_ndf(ndf);
352  out_track->set_charge(phgf_track->get_charge());
353 
354  out_track->set_px(mom.Px());
355  out_track->set_py(mom.Py());
356  out_track->set_pz(mom.Pz());
357 
358  out_track->set_x(pos.X());
359  out_track->set_y(pos.Y());
360  out_track->set_z(pos.Z());
361 
362  for(int i=0;i<6;i++)
363  {
364  for(int j=i;j<6;j++)
365  {
366  out_track->set_error(i,j,cov[i][j]);
367  }
368  }
369 
370  return out_track;
371 }

375 PHGenFit::PlanarMeasurement* PHG4TrackFastSim::PHG4HitToMeasurementVerticalPlane(const PHG4Hit* g4hit) {
376 
377  PHGenFit::PlanarMeasurement* meas = NULL;
378 
379  TVector3 pos(
380  g4hit->get_avg_x(),
381  g4hit->get_avg_y(),
382  g4hit->get_avg_z());
383 
384 
385  TVector3 v(pos.X(), pos.Y(), 0);
386  v = 1/v.Mag() * v;
387 
388  TVector3 u = v.Cross(TVector3(0,0,1));
389  u = 1/u.Mag() * u;
390 
391  double u_smear = gRandom->Gaus(0, _phi_resolution);
392  double v_smear = gRandom->Gaus(0, _r_resolution);
393  pos.SetX(g4hit->get_avg_x() + u_smear*u.X() + v_smear*v.X());
394  pos.SetY(g4hit->get_avg_y() + u_smear*u.Y() + v_smear*v.Y());
395 
396 
397  meas = new PHGenFit::PlanarMeasurement(pos, u, v, _phi_resolution, _r_resolution);
398

402 // u.Print();

406 
407  return meas;
