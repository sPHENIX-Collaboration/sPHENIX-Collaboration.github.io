65  if (!ReuseExistingVertex(topNode))
66  {
67  vtx_z = (z_max - z_min) * gsl_rng_uniform_pos(RandomGenerator) + z_min;
68  }
69  int vtxindex = ineve->AddVtx(vtx_x, vtx_y, vtx_z, t0);

71  vector<PHG4Particle *>::iterator iter;
72  for (iter = particlelist.begin(); iter != particlelist.end(); ++iter)
73  {
74  PHG4Particle *particle = new PHG4Particlev2(*iter);
75  SetParticleId(particle, ineve);
76  double mom = (mom_max - mom_min) * gsl_rng_uniform_pos(RandomGenerator) + mom_min;
77  double eta = (eta_max - eta_min) * gsl_rng_uniform_pos(RandomGenerator) + eta_min;
78  double phi = (phi_max - phi_min) * gsl_rng_uniform_pos(RandomGenerator) + phi_min;
79  double pt = mom / cosh(eta);
80 
81  particle->set_e(mom);
82  particle->set_px(pt * cos(phi));
83  particle->set_py(pt * sin(phi));
84  particle->set_pz(pt * sinh(eta));
85  // update internal particle list with changed momenta
86  // needed for correct printout of particle kinematics
87  // in PHG4ParticleGenerator::Print()
88  (*iter)->set_e(particle->get_e());
89  (*iter)->set_px(particle->get_px());
90  (*iter)->set_py(particle->get_py());
91  (*iter)->set_pz(particle->get_pz());
92  ineve->AddParticle(vtxindex, particle);
93  }
94  if (Verbosity() > 0)
95  {
