43 int sHEPGen::Init(PHCompositeNode *topNode) {
44 
45  printlogo();
46 
47  /* electron and proton mass */
48  double mass_e = 5.109989e-4;
49  double mass_p = 9.382720e-1;
50 
51  /* 4-Vectors of colliding electron and proton in laboratory frame */
52  _p4_electron_lab = new HLorentzVector( 0.,
53  0.,
54  _p_electron_lab,
55  sqrt(_p_electron_lab*_p_electron_lab+mass_e*mass_e) );
56 
57  _p4_hadron_lab = new HLorentzVector( 0.,
58  0.,
59  _p_hadron_lab,
60  sqrt(_p_hadron_lab*_p_hadron_lab+mass_p*mass_p) );
61 
62  /* The current version of HEPGen supports only a "Fixed Target" mode, i.e.
63  the target (proton) is assumed at rest. Therefore, need to boost collision
64  from the laboratory frame to the proton-at-rest frame. */
65  _p4_hadron_lab_invert = new HLorentzVector( 0.,
66  0.,
67  -1 * _p_hadron_lab,
68  sqrt(_p_hadron_lab*_p_hadron_lab+mass_p*mass_p) );
69 
70  /* 4-Vectors of colliding electron and proton in proton-at-rest frame */
71  _p4_electron_prest = new HLorentzVector( *_p4_electron_lab );
72  _p4_electron_prest->boost(sqrt(_p4_hadron_lab_invert->getQuare()),*_p4_hadron_lab_invert);
73 
74  _p4_hadron_prest = new HLorentzVector( *_p4_hadron_lab );
75  _p4_hadron_prest->boost(sqrt(_p4_hadron_lab_invert->getQuare()),*_p4_hadron_lab_invert);
76 
77  if ( verbosity > 2 )
78  {
79  cout << "Electron and proton in laboratory frame:" << endl;
80  _p4_electron_lab->print();
81  _p4_hadron_lab->print();
82 
83  cout << "Electron and proton in proton-at-rest frame:" << endl;
84  _p4_electron_prest->print();
85  _p4_hadron_prest->print();
86  }
87 
88  /* flip sign of electron momentum z-component for HEPGen */
89  HVector3 flip_pz( _p4_electron_prest->getVector() );
90  flip_pz.setZ( flip_pz.Z() * -1 );
91  _p4_electron_prest->setVector( flip_pz );
92 
93  /* get instance of HepGenManager */
94  _hgenManager = HGenManager::getInstance();
95  _hgenManager->loadSettings( _datacardFile );
96  _hgenManager->setupGenerator();
97 
98  /* set beam parameters */
99  cout << "Colliding " << _p4_electron_lab->getVector().Z() << " GeV electron with " << _p4_hadron_lab->getVector().Z() << " GeV proton (laboratory frame)" << endl;
100  cout << "----ELEPT (proton-at-rest): " << _p4_electron_prest->getVector().Z() << " GeV" << endl;
101  _hgenManager->getParamManager()->getStruct()->ELEPT = _p4_electron_prest->getVector().Z();
102  _hgenManager->getParamManager()->getStruct()->PARL.at(2) = _p4_electron_prest->getVector().Z();
103 
104  /* set random seed */
105  unsigned int seed = PHRandomSeed();
106  _hgenManager->setSeed ( seed );
107 
108  /* enable detailed event record printput for debugging */
109  if ( verbosity > 2 )
110  _hgenManager->enableDebug();
111 
112  create_node_tree(topNode);
113 
114  _eventcount = 0;
115 
116  return Fun4AllReturnCodes::EVENT_OK;
117 }

148  evt->use_units(HepMC::Units::GEV, HepMC::Units::MM);
149

151  evt->set_event_number(_eventcount);
152 
153  /* Set the PDF information */
154  HepMC::PdfInfo pdfinfo;
155  pdfinfo.set_scalePDF( evt_mc->getQsq() );
156  pdfinfo.set_x2( evt_mc->getXbj() );
157  evt->set_pdf_info( pdfinfo );
158 
159  /* Set additional event parameters */
160  evt->set_event_scale( evt_mc->getQsq() );
161 
162  /* Event kinematics */
163  /* @TODO How can we store this information in HepMC record? */
164  //evt_mc->getNu();
165  //evt_mc->getY();
166  //evt_mc->getWsq();
167  //evt_mc->getXbj();
168  //evt_mc->getT();

176  0,

178  0 )
179  );

183  for ( unsigned p = 0; p < edata->listOfParticles.size(); p++ )
184  {
185  if (verbosity > 4)
186  {
187  cout << "______new particle_______" << endl;
188  cout << "Index: " << p+1 << endl;
189  cout << "PID: " << edata->listOfParticles.at(p)->getParticleType() << " -- " << (edata->listOfParticles.at(p) == &(edata->incBeamParticle)) << endl;
190  cout << "Particle aux flag: " << edata->listOfParticles.at(p)->getParticleAuxFlag() << endl;
191  cout << "Particle origin: " << edata->listOfParticles.at(p)->getParticleOrigin() << endl;
192  cout << "Particle daughter1: " << edata->listOfParticles.at(p)->getParticleDaughter1() << endl;
193  cout << "Particle daughter2: " << edata->listOfParticles.at(p)->getParticleDaughter2() << endl;
194  }
195

197 
198  /* flip z axis component of particle momentum: sPHENIX EIC detector coordinate system uses
199  electron flying in 'negative z' direction */
200  HVector3 flip_pz( v4_particle_p.getVector() );
201  flip_pz.setZ( flip_pz.Z() * -1 );
202  v4_particle_p.setVector( flip_pz );
203

205  HLorentzVector v4_particle_p_lab( v4_particle_p );
206

208 
209  if ( verbosity > 2 )
210  {
211  cout << "EVENT RECORD particle: " << edata->listOfParticles.at(p)->getParticleType()
212  << " (status: " << edata->listOfParticles.at(p)->getParticleAuxFlag() << ")" << endl;
213  cout << " --> PROTON REST frame: "; v4_particle_p.print();
214  cout << " --> LABORATORY frame: "; v4_particle_p_lab.print();
215  }
216

218  v4_particle_p_lab.getVector().Y(),
219  v4_particle_p_lab.getVector().Z(),
220  v4_particle_p_lab.getEnergy()),
221  edata->listOfParticles.at(p)->getParticleType() );

227  evt->add_vertex( hepmcvtx );
228 
229  /* pass HepMC to PHNode */

231  if (!success) {
232  cout << "sHEPGen::process_event - Failed to add event to HepMC record!" << endl;
233  return Fun4AllReturnCodes::ABORTRUN;
234  }
235 
236  /* print outs */

238 
239  ++_eventcount;
240 
241  return Fun4AllReturnCodes::EVENT_OK;
242 }
