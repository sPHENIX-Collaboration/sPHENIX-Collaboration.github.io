104 bool PHNodeIterator::cd(const string& pathString)
105 {
106  bool success = true;
107  if (pathString.empty())
108  {
109  while (currentNode->getParent())
110  {
111  currentNode = static_cast<PHCompositeNode*>(currentNode->getParent());
112  }
113  }
114  else
115  {
116  vector<string> splitpath;
117  boost::split(splitpath, pathString, boost::is_any_of(phooldefs::nodetreepathdelim));
118  bool pathFound;
119  PHNode* subNode;
120  int i = 0;
121  for (vector<string>::const_iterator iter = splitpath.begin(); iter != splitpath.end(); ++iter)
122  {
123  i++;
124  if (*iter == "..")
125  {
126  if (currentNode->getParent())
127  {
128  currentNode = static_cast<PHCompositeNode*>(currentNode->getParent());
129  }
130  else
131  {
132  success = false;
133  }
134  }
135  else
136  {
137  PHPointerListIterator<PHNode> subNodeIter(currentNode->subNodes);
138  pathFound = false;
139  while ((subNode = subNodeIter()))
140  {
141  if (subNode->getType() == "PHCompositeNode" && subNode->getName() == *iter)
142  {
143  currentNode = static_cast<PHCompositeNode*>(subNode);
144  pathFound = true;
145  }
146  }
147  if (!pathFound)
148  {
149  success = false;
150  }
151  }
152  }
153  }
