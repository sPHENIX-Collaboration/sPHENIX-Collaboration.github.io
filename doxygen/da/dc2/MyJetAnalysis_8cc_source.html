150 int MyJetAnalysis::process_event(PHCompositeNode* topNode)
151 {
152  if (Verbosity() >= MyJetAnalysis::VERBOSITY_SOME)
153  cout << "MyJetAnalysis::process_event() entered" << endl;
154 
155  m_jetEvalStack->next_event(topNode);
156  JetRecoEval* recoeval = m_jetEvalStack->get_reco_eval();
157  ++m_event;
158 
159  // interface to jets
160  JetMap* jets = findNode::getClass<JetMap>(topNode, m_recoJetName);
161  if (!jets)
162  {
163  cout
164  << "MyJetAnalysis::process_event - Error can not find DST JetMap node "
165  << m_recoJetName << endl;
166  exit(-1);
167  }
168 
169  // interface to tracks
170  SvtxTrackMap* trackmap = findNode::getClass<SvtxTrackMap>(topNode, "SvtxTrackMap");
171  if (!trackmap)
172  {
173  cout
174  << "MyJetAnalysis::process_event - Error can not find DST trackmap node SvtxTrackMap" << endl;
175  exit(-1);
176  }
177 
178  for (JetMap::Iter iter = jets->begin(); iter != jets->end(); ++iter)
179  {
180  Jet* jet = iter->second;
181  assert(jet);
182 
183  bool eta_cut = (jet->get_eta() >= m_etaRange.first) and (jet->get_eta() <= m_etaRange.second);
184  bool pt_cut = (jet->get_pt() >= m_ptRange.first) and (jet->get_pt() <= m_ptRange.second);
185  if ((not eta_cut) or (not pt_cut))
186  {
187  if (Verbosity() >= MyJetAnalysis::VERBOSITY_MORE)
188  {
189  cout << "MyJetAnalysis::process_event() - jet failed acceptance cut: ";
190  cout << "eta cut: " << eta_cut << ", ptcut: " << pt_cut << endl;
191  cout << "jet eta: " << jet->get_eta() << ", jet pt: " << jet->get_pt() << endl;
192  jet->identify();
193  }
194  continue;
195  }
196 
197  // fill histograms
198  assert(m_hInclusiveE);
199  m_hInclusiveE->Fill(jet->get_e());
200  assert(m_hInclusiveEta);
201  m_hInclusiveEta->Fill(jet->get_eta());
202  assert(m_hInclusivePhi);
203  m_hInclusivePhi->Fill(jet->get_phi());
204 
205  // fill trees - jet spectrum
206  Jet* truthjet = recoeval->max_truth_jet_by_energy(jet);
207 
208  m_id = jet->get_id();
209  m_nComponent = jet->size_comp();
210  m_eta = jet->get_eta();
211  m_phi = jet->get_phi();
212  m_e = jet->get_e();
213  m_pt = jet->get_pt();
214 
215  m_truthID = NAN;
216  m_truthNComponent = NAN;
217  m_truthEta = NAN;
218  m_truthPhi = NAN;
219  m_truthE = NAN;
220  m_truthPt = NAN;
221 
222  if (truthjet)
223  {
224  m_truthID = truthjet->get_id();
225  m_truthNComponent = truthjet->size_comp();
226  m_truthEta = truthjet->get_eta();
227  m_truthPhi = truthjet->get_phi();
228  m_truthE = truthjet->get_e();
229  m_truthPt = truthjet->get_pt();
230  }
231 
232  // fill trees - jet track matching
233  m_nMatchedTrack = 0;
234 
235  for (SvtxTrackMap::Iter iter = trackmap->begin();
236  iter != trackmap->end();
237  ++iter)
238  {
239  SvtxTrack* track = iter->second;
240 
241  TVector3 v(track->get_px(), track->get_py(), track->get_pz());
242  const double dEta = v.Eta() - m_eta;
243  const double dPhi = v.Phi() - m_phi;
244  const double dR = sqrt(dEta * dEta + dPhi * dPhi);
245 
246  if (dR < m_trackJetMatchingRadius)
247  {
248  //matched track to jet
249 
250  assert(m_nMatchedTrack < kMaxMatchedTrack);
251 
252  m_trackdR[m_nMatchedTrack] = dR;
253  m_trackpT[m_nMatchedTrack] = v.Perp();
254 
255  ++m_nMatchedTrack;
256  }
257 
258  if (m_nMatchedTrack >= kMaxMatchedTrack)
259  {
260  cout << "MyJetAnalysis::process_event() - reached max track that matching a jet. Quit iterating tracks" << endl;
261  break;
262  }
263 
264  } // for (SvtxTrackMap::Iter iter = trackmap->begin();
265 
266  m_T->Fill();
267  } // for (JetMap::Iter iter = jets->begin(); iter != jets->end(); ++iter)
268 
269  return Fun4AllReturnCodes::EVENT_OK;
270 }
