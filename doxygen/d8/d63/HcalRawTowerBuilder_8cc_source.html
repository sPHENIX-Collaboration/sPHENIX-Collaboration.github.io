228 int HcalRawTowerBuilder::process_event(PHCompositeNode *topNode)
229 {
230  if (Verbosity() > 3)
231  {
232  std::cout << PHWHERE << "Process event entered" << std::endl;
233  }
234 
235  // get cells
236  std::string cellnodename = "G4CELL_" + m_Detector;
237  PHG4CellContainer *slats = findNode::getClass<PHG4CellContainer>(topNode, cellnodename);
238  if (!slats)
239  {
240  std::cerr << PHWHERE << " " << cellnodename
241  << " Node missing, doing nothing." << std::endl;
242  return Fun4AllReturnCodes::ABORTEVENT;
243  }
244 
245  // loop over all slats in an event
246  PHG4CellContainer::ConstIterator cell_iter;
247  PHG4CellContainer::ConstRange cell_range = slats->getCells();
248  for (cell_iter = cell_range.first; cell_iter != cell_range.second;
249  ++cell_iter)
250  {
251  PHG4Cell *cell = cell_iter->second;
252 
253  if (Verbosity() > 2)
254  {
255  std::cout << PHWHERE << " print out the cell:" << std::endl;
256  cell->identify();
257  }
258  short twrrow = get_tower_row(PHG4CellDefs::ScintillatorSlatBinning::get_row(cell->get_cellid()));
259  // add the energy to the corresponding tower
260  // towers are addressed column/row to make the mapping more intuitive
261  RawTower *tower = m_Towers->getTower(PHG4CellDefs::ScintillatorSlatBinning::get_column(cell->get_cellid()), twrrow);
262  if (!tower)
263  {
264  tower = new RawTowerv1();
265  tower->set_energy(0);
266  m_Towers->AddTower(PHG4CellDefs::ScintillatorSlatBinning::get_column(cell->get_cellid()), twrrow, tower);
267  }
268  double cell_weight = 0;
269  if (m_TowerEnergySrc == kEnergyDeposition)
270  {
271  cell_weight = cell->get_edep();
272  }
273  else if (m_TowerEnergySrc == kLightYield)
274  {
275  cell_weight = cell->get_light_yield();
276  }
277  else if (m_TowerEnergySrc == kIonizationEnergy)
278  {
279  cell_weight = cell->get_eion();
280  }
281  else
282  {
283  cout << Name() << ": unknown tower energy source "
284  << m_TowerEnergySrc << endl;
285  gSystem->Exit(1);
286  }
287 
288  tower->add_ecell(cell->get_cellid(), cell_weight);
289 
290  PHG4Cell::ShowerEdepConstRange range = cell->get_g4showers();
291  for (PHG4Cell::ShowerEdepConstIterator shower_iter = range.first;
292  shower_iter != range.second;
293  ++shower_iter)
294  {
295  tower->add_eshower(shower_iter->first, shower_iter->second);
296  }
297  tower->set_energy(tower->get_energy() + cell_weight);
298  }
299  double towerE = 0;
300  if (m_ChkEnergyConservationFlag)
301  {
302  double cellE = slats->getTotalEdep();
303  towerE = m_Towers->getTotalEdep();
304  if (fabs(cellE - towerE) / cellE > 1e-5)
305  {
306  cout << "towerE: " << towerE << ", cellE: " << cellE << ", delta: "
307  << cellE - towerE << endl;
308  }
309  }
310  if (Verbosity())
311  {
312  towerE = m_Towers->getTotalEdep();
313  }
314 
315  m_Towers->compress(m_Emin);
316  if (Verbosity())
317  {
318  cout << "Energy lost by dropping towers with less than " << m_Emin
319  << " energy, lost energy: " << towerE - m_Towers->getTotalEdep()
320  << endl;
321  m_Towers->identify();
322  RawTowerContainer::ConstRange begin_end = m_Towers->getTowers();
323  RawTowerContainer::ConstIterator iter;
324  for (iter = begin_end.first; iter != begin_end.second; ++iter)
325  {
326  iter->second->identify();
327  }
328  }
329 
330  return Fun4AllReturnCodes::EVENT_OK;
331 }

333 void HcalRawTowerBuilder::CreateNodes(PHCompositeNode *topNode)

353  PHNodeIterator dstiter(dstNode);

355  "PHCompositeNode", m_Detector));
356  if (!DetNode)

358  DetNode = new PHCompositeNode(m_Detector);
359  dstNode->addNode(DetNode);
360  }
361 
362  // Create the tower nodes on the tree
363  if (m_SimTowerNodePrefix.empty())
364  {

366  m_TowerNodeName = "TOWER_" + m_Detector;
