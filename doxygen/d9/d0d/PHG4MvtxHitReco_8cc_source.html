470  }
471  if (Verbosity() > 5)
472  {
473  cout << " pixnum " << pixnum << " xbin " << ix << " zbin " << iz
474  << " pixel_area fraction of circle " << pixarea_frac << " accumulated pixel energy " << pixenergy[ix - xbin_min][iz - zbin_min]
475  << endl;
476  }
477  }
478  }
479  } // end loop over segments
480 
481  // now we have the energy deposited in each pixel, summed over all tracklet segments. We make a vector of all pixels with non-zero energy deposited
482  for (int ix = xbin_min; ix <= xbin_max; ix++)
483  {
484  for (int iz = zbin_min; iz <= zbin_max; iz++)
485  {
486  if (pixenergy[ix - xbin_min][iz - zbin_min] > 0.0)
487  {
488  int pixnum = layergeom->get_pixel_number_from_xbin_zbin(ix, iz);
489  vpixel.push_back(pixnum);
490  vxbin.push_back(ix);
491  vzbin.push_back(iz);
492  pair<double, double> tmppair = make_pair(pixenergy[ix - xbin_min][iz - zbin_min], pixeion[ix - xbin_min][iz - zbin_min]);
493  venergy.push_back(tmppair);
494  if (Verbosity() > 1)
495  cout << " Added pixel number " << pixnum << " xbin " << ix << " zbin " << iz << " to vectors with energy " << pixenergy[ix - xbin_min][iz - zbin_min] << endl;
496  }
497  }
498  }
499

501  // End of charge sharing implementation
502  //===================================
503 
504  // loop over all fired cells for this g4hit and add them to the TrkrHitSet
505  for (unsigned int i1 = 0; i1 < vpixel.size(); i1++) // loop over all fired cells
506  {
507  // This is the new storage object version
508  //====================================
509 
510  // We need to create the TrkrHitSet if not already made - each TrkrHitSet should correspond to a chip for the Mvtx
511  TrkrDefs::hitsetkey hitsetkey = MvtxDefs::genHitSetKey(*layer, stave_number, chip_number);
512  // Use existing hitset or add new one if needed
513  TrkrHitSetContainer::Iterator hitsetit = trkrhitsetcontainer->findOrAddHitSet(hitsetkey);
514 
515  // generate the key for this hit
516  TrkrDefs::hitkey hitkey = MvtxDefs::genHitKey(vzbin[i1], vxbin[i1]);
517  // See if this hit already exists
518  TrkrHit *hit = nullptr;
519  hit = hitsetit->second->getHit(hitkey);
520  if (!hit)
521  {
522  // Otherwise, create a new one
523  hit = new MvtxHit();
524  hitsetit->second->addHitSpecificKey(hitkey, hit);
525  }
526 
527  // Either way, add the energy to it
528  hit->addEnergy(venergy[i1].first);
529 
530  // now we update the TrkrHitTruthAssoc map - the map contains <hitsetkey, std::pair <hitkey, g4hitkey> >
531  // There is only one TrkrHit per pixel, but there may be multiple g4hits
532  // How do we know how much energy from PHG4Hit went into TrkrHit? We don't, have to sort it out in evaluator to save memory
533 
534  // How do we check if this association already exists?
535  //cout << "PHG4MvtxHitReco: adding association entry for hitkey " << hitkey << " and g4hitkey " << hiter->first << endl;
536  hittruthassoc->addAssoc(hitsetkey, hitkey, hiter->first);
537 
538  } // end loop over hit cells
539  } // end loop over g4hits for this layer
540 
541  } // end loop over layers
542 
543  // print the list of entries in the association table
544  if (Verbosity() > 2)
545  {
546  cout << "From PHG4MvtxHitReco: " << endl;
547  hittruthassoc->identify();
548  }
