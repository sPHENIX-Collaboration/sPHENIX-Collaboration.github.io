242 int PHG4SimpleEventGenerator::process_event(PHCompositeNode *topNode) {
243 
244  if (Verbosity() > 0) {
245  cout << "====================== PHG4SimpleEventGenerator::process_event() =====================" << endl;
246  cout <<"PHG4SimpleEventGenerator::process_event - reuse_existing_vertex = "<<reuse_existing_vertex<<endl;
247  }
248 
249  // vtx_x, vtx_y and vtx_z are doubles from the base class
250  // common methods modify those, please no private copies
251  // at some point we might rely on them being up to date
252  if (!ReuseExistingVertex(topNode))
253  {
254  // generate a new vertex point
255  vtx_x = smearvtx(_vertex_x,_vertex_width_x,_vertex_func_x);
256  vtx_y = smearvtx(_vertex_y,_vertex_width_y,_vertex_func_y);
257  vtx_z = smearvtx(_vertex_z,_vertex_width_z,_vertex_func_z);
258  }
259 
260  vtx_x += _vertex_offset_x;
261  vtx_y += _vertex_offset_y;
262  vtx_z += _vertex_offset_z;
263 
264  if (Verbosity() > 0) {
265  cout <<"PHG4SimpleEventGenerator::process_event - vertex center"<<reuse_existing_vertex
266  << vtx_x<<", "<< vtx_y<<", "<< vtx_z<<" cm"
267  <<endl;
268  }
269 
270  int vtxindex = -1;
271  int trackid = -1;
272  for (unsigned int i=0; i<_particle_names.size(); ++i) {
273 
274  string pdgname = _particle_names[i].first;
275  int pdgcode = get_pdgcode(pdgname);
276  unsigned int nparticles = _particle_names[i].second;
277 
278  for (unsigned int j=0; j<nparticles; ++j) {
279 
280  if ((_vertex_size_width > 0.0)||(_vertex_size_mean != 0.0)) {
281 
282  double r = smearvtx(_vertex_size_mean,_vertex_size_width,_vertex_size_func_r);
283 
284  double x = 0.0;
285  double y = 0.0;
286  double z = 0.0;
287  gsl_ran_dir_3d(RandomGenerator,&x,&y,&z);
288  x *= r;
289  y *= r;
290  z *= r;
291 
292  vtxindex = _ineve->AddVtx(vtx_x+x,vtx_y+y,vtx_z+z,_t0);
293  } else if ((i==0)&&(j==0)) {
294  vtxindex = _ineve->AddVtx(vtx_x,vtx_y,vtx_z,_t0);
295  }
296 
297  ++trackid;
298 
299  double eta = (_eta_max-_eta_min) * gsl_rng_uniform_pos(RandomGenerator) + _eta_min;
300  double phi = (_phi_max-_phi_min) * gsl_rng_uniform_pos(RandomGenerator) + _phi_min;
301 
302  double pt;
303  if (!std::isnan(_p_min) && !std::isnan(_p_max) && !std::isnan(_p_gaus_width)) {
304  pt = ((_p_max-_p_min) * gsl_rng_uniform_pos(RandomGenerator) + _p_min + gsl_ran_gaussian(RandomGenerator, _p_gaus_width)) / cosh(eta);
305  } else if (!std::isnan(_pt_min) && !std::isnan(_pt_max) && !std::isnan(_pt_gaus_width)) {
306  pt = (_pt_max-_pt_min) * gsl_rng_uniform_pos(RandomGenerator) + _pt_min + gsl_ran_gaussian(RandomGenerator, _pt_gaus_width);
307  } else {
308  cout << PHWHERE << "Error: neither a p range or pt range was specified" << endl;
309  exit(-1);
310  }
311 
312  double px = pt*cos(phi);
313  double py = pt*sin(phi);
314  double pz = pt*sinh(eta);
315  double m = get_mass(pdgcode);
316  double e = sqrt(px*px+py*py+pz*pz+m*m);
317 
318  PHG4Particle *particle = new PHG4Particlev2();
319  particle->set_track_id(trackid);
320  particle->set_vtx_id(vtxindex);
321  particle->set_parent_id(0);
322  particle->set_name(pdgname);
323  particle->set_pid(pdgcode);
324  particle->set_px(px);
325  particle->set_py(py);
326  particle->set_pz(pz);
327  particle->set_e(e);
328 
329  _ineve->AddParticle(vtxindex, particle);
330  if (embedflag != 0) _ineve->AddEmbeddedParticle(particle,embedflag);
331  }
332  }
333 
334  if (Verbosity() > 0) {
335  _ineve->identify();
336  cout << "======================================================================================" << endl;
337  }
338 
339  return Fun4AllReturnCodes::EVENT_OK;
340 }

342 double
343 PHG4SimpleEventGenerator::smearvtx(const double position, const double width, FUNCTION dist) const
344 {
345  double res = position;
346  if (dist == Uniform)
347  {
348  res = (position-width) + 2*gsl_rng_uniform_pos(RandomGenerator)*width;
349  }
350  else if (dist == Gaus)
351  {
352  res = position + gsl_ran_gaussian(RandomGenerator,width);
353  }
354  return res;
