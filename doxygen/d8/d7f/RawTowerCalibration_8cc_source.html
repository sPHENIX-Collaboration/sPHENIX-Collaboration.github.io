144 
145  const double raw_energy = raw_tower->get_energy();

180 void RawTowerCalibration::CreateNodes(PHCompositeNode *topNode)

184  "PHCompositeNode", "RUN"));

187  std::cerr << Name() << "::" << detector << "::" << __PRETTY_FUNCTION__
188  << "Run Node missing, doing nothing." << std::endl;

190  "Failed to find Run node in RawTowerCalibration::CreateNodes");
191  }
192

195  TowerGeomNodeName.c_str());

199  << " " << TowerGeomNodeName << " Node missing, doing bail out!"

202  "Failed to find " + TowerGeomNodeName + " node in RawTowerCalibration::CreateNodes");
203  }
204

209 
210  PHCompositeNode *dstNode = dynamic_cast<PHCompositeNode *>(iter.findFirst(
211  "PHCompositeNode", "DST"));
212  if (!dstNode)

215  << "DST Node missing, doing nothing." << std::endl;
