1809 void PHHoughSeeding::convertHelixCovarianceToEuclideanCovariance(float B,
1810  float phi, float d, float kappa, float z0, float dzdl,
1811  Eigen::Matrix<float, 5, 5> const& input,
1812  Eigen::Matrix<float, 6, 6>& output)
1813 {
1814  Eigen::Matrix<float, 6, 5> J = Eigen::Matrix<float, 6, 5>::Zero(6, 5);
1815 
1816  // phi,d,nu,z0,dzdl
1817  // -->
1818  // x,y,z,px,py,pz
1819 
1820  float nu = sqrt(kappa);
1821  float dk_dnu = 2 * nu;
1822 
1823  float cosphi = cos(phi);
1824  float sinphi = sin(phi);
1825 
1826  J(0, 0) = -sinphi * d;
1827  J(0, 1) = cosphi;
1828  J(1, 0) = d * cosphi;
1829  J(1, 1) = sinphi;
1830  J(2, 3) = 1;
1831 
1832  float pt = 0.003 * B / kappa;
1833  float dpt_dk = -0.003 * B / (kappa * kappa);
1834 
1835  J(3, 0) = -cosphi * pt;
1836  J(3, 2) = -sinphi * dpt_dk * dk_dnu;
1837  J(4, 0) = -sinphi * pt;
1838  J(4, 2) = cosphi * dpt_dk * dk_dnu;
1839 
1840  float alpha = 1. / (1. - dzdl * dzdl);
1841  float alpha_half = pow(alpha, 0.5);
1842  float alpha_3_half = alpha * alpha_half;
1843 
1844  J(5, 2) = dpt_dk * dzdl * alpha_half * dk_dnu;
1845  J(5, 4) = pt * (alpha_half + dzdl * dzdl * alpha_3_half) * dk_dnu;
1846 
1847  output = J * input * (J.transpose());
1848 }
