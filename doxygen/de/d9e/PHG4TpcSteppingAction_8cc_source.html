76 bool PHG4TpcSteppingAction::UserSteppingAction(const G4Step* aStep, bool)
77 {
78  G4TouchableHandle touch = aStep->GetPreStepPoint()->GetTouchableHandle();
79  G4TouchableHandle touchpost = aStep->GetPostStepPoint()->GetTouchableHandle();
80  // get volume of the current step
81  G4VPhysicalVolume* volume = touch->GetVolume();
82 
83  // detector_->IsInTpc(volume)
84  // returns
85  // 0 is outside of Tpc
86  // 1 is inside tpc gas
87  // <0 is in tpc support structure (cage, endcaps,...)
88 
89  int whichactive = detector_->IsInTpc(volume);
90  if (!whichactive)
91  {
92  return false;
93  }
94  unsigned int layer_id = 99; // no layer number for the hit, use a non-existent one for now, replace it later
95  // collect energy and track length step by step
96  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
97  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
98  const G4Track* aTrack = aStep->GetTrack();
99 
100  // if this block stops everything, just put all kinetic energy into edep
101  if (IsBlackHole)
102  {
103  edep = aTrack->GetKineticEnergy() / GeV;
104  G4Track* killtrack = const_cast<G4Track*>(aTrack);
105  killtrack->SetTrackStatus(fStopAndKill);
106  }
107 
108  bool geantino = false;
109 
110  // the check for the pdg code speeds things up, I do not want to make
111  // an expensive string compare for every track when we know
112  // geantino or chargedgeantino has pid=0
113  if (aTrack->GetParticleDefinition()->GetPDGEncoding() == 0 &&
114  aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos)
115  {
116  geantino = true;
117  }
118  G4StepPoint* prePoint = aStep->GetPreStepPoint();
119  G4StepPoint* postPoint = aStep->GetPostStepPoint();
120  int prepointstatus = prePoint->GetStepStatus();
121 
122  // cout << "track id " << aTrack->GetTrackID() << endl;
123  // cout << "time prepoint: " << prePoint->GetGlobalTime() << endl;
124  // cout << "time postpoint: " << postPoint->GetGlobalTime() << endl;
125 
126  if ((use_g4_steps > 0 && whichactive > 0) ||
127  prepointstatus == fGeomBoundary ||
128  prepointstatus == fUndefined ||
129  (prepointstatus == fPostStepDoItProc && savepoststepstatus == fGeomBoundary))
130  {
131  // this is for debugging weird occurances we have occasionally
132  if (prepointstatus == fPostStepDoItProc && savepoststepstatus == fGeomBoundary)
133  {
134  cout << GetName() << ": New Hit for " << endl;
135  cout << "prestep status: " << PHG4StepStatusDecode::GetStepStatus(prePoint->GetStepStatus())
136  << ", poststep status: " << PHG4StepStatusDecode::GetStepStatus(postPoint->GetStepStatus())
137  << ", last pre step status: " << PHG4StepStatusDecode::GetStepStatus(saveprestepstatus)
138  << ", last post step status: " << PHG4StepStatusDecode::GetStepStatus(savepoststepstatus) << endl;
139  cout << "last track: " << savetrackid
140  << ", current trackid: " << aTrack->GetTrackID() << endl;
141  cout << "phys pre vol: " << volume->GetName()
142  << " post vol : " << touchpost->GetVolume()->GetName() << endl;
143  cout << " previous phys pre vol: " << savevolpre->GetName()
144  << " previous phys post vol: " << savevolpost->GetName() << endl;
145  }
146  // if previous hit was saved, hit pointer was set to nullptr
147  // and we have to make a new one
148  if (!hit)
149  {
150  hit = new PHG4Hitv1();
151  }
152  hit->set_layer(layer_id);
153  //here we set the entrance values in cm
154  hit->set_x(0, prePoint->GetPosition().x() / cm);
155  hit->set_y(0, prePoint->GetPosition().y() / cm);
156  hit->set_z(0, prePoint->GetPosition().z() / cm);
157  // time in ns
158  hit->set_t(0, prePoint->GetGlobalTime() / nanosecond);
159  //set and save the track ID
160  hit->set_trkid(aTrack->GetTrackID());
161  savetrackid = aTrack->GetTrackID();
162  //set the initial energy deposit
163  hit->set_edep(0);
164  if (whichactive > 0) // return of IsInTpcDetector, > 0 hit in tpc gas volume, < 0 hit in support structures
165  {
166  hit->set_eion(0);
167  // Now save the container we want to add this hit to
168  savehitcontainer = hits_;
169  }
170  else
171  {
172  savehitcontainer = absorberhits_;
173  }
174  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
175  {
176  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
177  {
178  hit->set_trkid(pp->GetUserTrackId());
179  hit->set_shower_id(pp->GetShower()->get_id());
180  saveshower = pp->GetShower();
181  }
182  }
183  // some sanity checks for inconsistencies
184  // check if this hit was created, if not print out last post step status
185  if (!hit || !isfinite(hit->get_x(0)))
186  {
187  cout << GetName() << ": hit was not created" << endl;
188  cout << "prestep status: " << PHG4StepStatusDecode::GetStepStatus(prePoint->GetStepStatus())
189  << ", poststep status: " << PHG4StepStatusDecode::GetStepStatus(postPoint->GetStepStatus())
190  << ", last pre step status: " << PHG4StepStatusDecode::GetStepStatus(saveprestepstatus)
191  << ", last post step status: " << PHG4StepStatusDecode::GetStepStatus(savepoststepstatus) << endl;
192  cout << "last track: " << savetrackid
193  << ", current trackid: " << aTrack->GetTrackID() << endl;
194  cout << "phys pre vol: " << volume->GetName()
195  << " post vol : " << touchpost->GetVolume()->GetName() << endl;
196  cout << " previous phys pre vol: " << savevolpre->GetName()
197  << " previous phys post vol: " << savevolpost->GetName() << endl;
198  exit(1);
199  }
200  // check if track id matches the initial one when the hit was created
201  if (aTrack->GetTrackID() != savetrackid)
202  {
203  cout << GetName() << ": hits do not belong to the same track" << endl;
204  cout << "saved track: " << savetrackid
205  << ", current trackid: " << aTrack->GetTrackID()
206  << ", prestep status: " << prePoint->GetStepStatus()
207  << ", previous post step status: " << savepoststepstatus
208  << endl;
209 
210  exit(1);
211  }
212  saveprestepstatus = prePoint->GetStepStatus();
213  savepoststepstatus = postPoint->GetStepStatus();
214  savevolpre = volume;
215  savevolpost = touchpost->GetVolume();
216  // here we just update the exit values, it will be overwritten
217  // for every step until we leave the volume or the particle
218  // ceases to exist
219  hit->set_x(1, postPoint->GetPosition().x() / cm);
220  hit->set_y(1, postPoint->GetPosition().y() / cm);
221  hit->set_z(1, postPoint->GetPosition().z() / cm);
222 
223  hit->set_t(1, postPoint->GetGlobalTime() / nanosecond);
224 
225  //sum up the energy to get total deposited
226  hit->set_edep(hit->get_edep() + edep);
227  if (whichactive > 0)
228  {
229  hit->set_eion(hit->get_eion() + eion);
230  }
231  if (geantino)
232  {
233  hit->set_edep(-1); // only energy=0 g4hits get dropped, this way geantinos survive the g4hit compression
234  if (whichactive > 0)
235  {
236  hit->set_eion(-1);
237  }
238  }
239  if (edep > 0)
240  {
241  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
242  {
243  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
244  {
245  pp->SetKeep(1); // we want to keep the track
246  }
247  }
248  }
249 
250  // if any of these conditions is true this is the last step in
251  // this volume and we need to save the hit
252  // postPoint->GetStepStatus() == fGeomBoundary: track leaves this volume
253  // postPoint->GetStepStatus() == fWorldBoundary: track leaves this world
254  // (happens when your detector goes outside world volume)
255  // postPoint->GetStepStatus() == fAtRestDoItProc: track stops (typically
256  // aTrack->GetTrackStatus() == fStopAndKill is also set)
257  // aTrack->GetTrackStatus() == fStopAndKill: track ends
258  if ((use_g4_steps > 0 && whichactive > 0) ||
259  postPoint->GetStepStatus() == fGeomBoundary ||
260  postPoint->GetStepStatus() == fWorldBoundary ||
261  postPoint->GetStepStatus() == fAtRestDoItProc ||
262  aTrack->GetTrackStatus() == fStopAndKill)
263  {
264  // save only hits with energy deposit (or -1 for geantino)
265  if (hit->get_edep())
266  {
267  savehitcontainer->AddHit(layer_id, hit);
268  if (saveshower)
269  {
270  saveshower->add_g4hit_id(savehitcontainer->GetID(), hit->get_hit_id());
271  }
272 
273  double rin = sqrt(hit->get_x(0) * hit->get_x(0) + hit->get_y(0) * hit->get_y(0));
274  double rout = sqrt(hit->get_x(1) * hit->get_x(1) + hit->get_y(1) * hit->get_y(1));
275  if (Verbosity() > 10)
276  if ((rin > 69.0 && rin < 70.125) || (rout > 69.0 && rout < 70.125))
277  {
278  cout << "Added Tpc g4hit with rin, rout = " << rin << " " << rout
279  << " g4hitid " << hit->get_hit_id() << endl;
280  cout << " xin " << hit->get_x(0)
281  << " yin " << hit->get_y(0)
282  << " zin " << hit->get_z(0)
283  << " rin " << rin
284  << endl;
285  cout << " xout " << hit->get_x(1)
286  << " yout " << hit->get_y(1)
287  << " zout " << hit->get_z(1)
288  << " rout " << rout
289  << endl;
290  cout << " xav " << (hit->get_x(1) + hit->get_x(0)) / 2.0
291  << " yav " << (hit->get_y(1) + hit->get_y(0)) / 2.0
292  << " zav " << (hit->get_z(1) + hit->get_z(0)) / 2.0
293  << " rav " << (rout + rin) / 2.0
294  << endl;
295  }
296 
297  // ownership has been transferred to container, set to null
298  // so we will create a new hit for the next track
299  hit = nullptr;
300  }
301  else
302  {
303  // if this hit has no energy deposit, just reset it for reuse
304  // this means we have to delete it in the dtor. If this was
305  // the last hit we processed the memory is still allocated
306  hit->Reset();
307  }
308  }
309  // return true to indicate the hit was used
310  return true;
311  }
312  else
313  {
314  return false;
315  }
