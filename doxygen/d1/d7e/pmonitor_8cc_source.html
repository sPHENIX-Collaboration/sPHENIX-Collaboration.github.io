599 void update_process (void * ptr)
600 {
601  int highest_subpad = 0;
602 
603  // our argument
604  threadargument *ta = ( threadargument *) ptr;
605 
606  TVirtualPad *myPad = ta->thePad;
607  unsigned int my_refreshinterval = ta->refreshinterval;
608 
609  // and done with it.
610  delete ta;
611 
612  time_t last_time = time(0) + my_refreshinterval + 1; //force an update on start
613 
614  TVirtualPad * s = 0;
615  int i;
616 
617  for ( i = 0; i < 128; i++)
618  {
619  if ( s == myPad->cd(i)) break;
620  s = myPad->cd(i);
621  }
622  highest_subpad = i-1;
623 
624 
625  while ( 1)
626  {
627  if ( stop_update)
628  {
629  if ( pad_to_stop == 0)
630  {
631  break;
632  }
633  else
634  {
635  if (pad_to_stop == myPad) break;
636  }
637  }
638 
639  time_t x = time(0);
640  if ( x - last_time > my_refreshinterval)
641  {
642  last_time = x;
643 
644  for ( i = 0; i <= highest_subpad; i++)
645  {
646  s = myPad->cd(i);
647  s->Modified();
648  s->Update();
649  }
650  sleep(1);
651  }
652  }
