151  if (iy < 0 || iy >= NBINY)

154  return false;
155  }
156 
157  ix += BINX0;
158  iy += BINY0;
159 
160  RawTowerDefs::keytype key = RawTowerDefs::encode_towerid(towergeom->get_calorimeter_id(), ix, iy);
161  RawTowerGeom *geom0 = towergeom->get_tower_geometry(key);
162 
163  // Next tower in x
164  key = RawTowerDefs::encode_towerid(towergeom->get_calorimeter_id(), ix + 1, iy);
165  RawTowerGeom *geomx = towergeom->get_tower_geometry(key);
166  if (geomx == nullptr)
167  {
168  key = RawTowerDefs::encode_towerid(towergeom->get_calorimeter_id(), ix - 1, iy);
169  geomx = towergeom->get_tower_geometry(key);
170  if (geomx == nullptr)
171  {
172  printf("RawClusterBuilderTemplateEEMC::Cell2Abs: error in geometry extraction: %d %d\n", ix, iy);
173  return false;
174  }
175  }
176 
177  // Next tower in y
178  key = RawTowerDefs::encode_towerid(towergeom->get_calorimeter_id(), ix, iy + 1);
179  RawTowerGeom *geomy = towergeom->get_tower_geometry(key);
180  if (geomy == nullptr)
181  {
182  key = RawTowerDefs::encode_towerid(towergeom->get_calorimeter_id(), ix, iy - 1);
183  geomy = towergeom->get_tower_geometry(key);
184  if (geomy == nullptr)
185  {

187  return false;
188  }
189  }
190 
191  // float dx = geom->get_size_x();
192  // float dy = geom->get_size_y();
193  float dx = fabs(geom0->get_center_x() - geomx->get_center_x());
194  float dy = fabs(geom0->get_center_y() - geomy->get_center_y());
195  xA = geom0->get_center_x() + (xC - ix + BINX0) * dx;
196  yA = geom0->get_center_y() + (yC - iy + BINY0) * dy;
