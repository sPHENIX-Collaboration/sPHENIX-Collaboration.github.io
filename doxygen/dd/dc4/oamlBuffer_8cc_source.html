216 int oamlBuffer::readn (int fd, char *ptr, int nbytes)
217 {
218  int nleft, nread;
219  nleft = nbytes;
220  while ( nleft>0 )
221  {
222  nread = read (fd, ptr, nleft);
223  if ( nread < 0 )
224  return nread;
225  else if (nread == 0)
226  break;
227  nleft -= nread;
228  ptr += nread;
229  }
230  return (nbytes-nleft);
