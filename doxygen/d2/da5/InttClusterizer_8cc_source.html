43 bool InttClusterizer::ladder_are_adjacent( const std::pair<TrkrDefs::hitkey, TrkrHit*> &lhs, const std::pair<TrkrDefs::hitkey, TrkrHit*> &rhs, const int layer)
44 {
45  if (get_z_clustering(layer))

260  // Get Nodes
261  //----------
262 
263  // get the geometry node
264  PHG4CylinderGeomContainer* geom_container = findNode::getClass<PHG4CylinderGeomContainer>(topNode, "CYLINDERGEOM_INTT");
265  if (!geom_container) return;

268  // Clustering
269  //-----------
270 
271  // loop over the InttHitSet objects
272  TrkrHitSetContainer::ConstRange hitsetrange =
273  m_hits->getHitSets(TrkrDefs::TrkrId::inttId);
274  for (TrkrHitSetContainer::ConstIterator hitsetitr = hitsetrange.first;
275  hitsetitr != hitsetrange.second;
276  ++hitsetitr)
277  {
278  // Each hitset contains only hits that are clusterizable - i.e. belong to a single sensor
279  TrkrHitSet *hitset = hitsetitr->second;
280 
281  if(Verbosity() > 1) cout << "InttClusterizer found hitsetkey " << hitsetitr->first << endl;
282  if (Verbosity() > 2)
283  hitset->identify();
284 
285  // we have a single hitset, get the info that identifies the sensor
286  int layer = TrkrDefs::getLayer(hitsetitr->first);
287  int ladder_z_index = InttDefs::getLadderZId(hitsetitr->first);
288  int ladder_phi_index = InttDefs::getLadderPhiId(hitsetitr->first);
289 
290  // we will need the geometry object for this layer to get the global position
291  CylinderGeomIntt* geom = dynamic_cast<CylinderGeomIntt*>(geom_container->GetLayerGeom(layer));
292  float thickness = geom->get_thickness();
293  float pitch = geom->get_strip_y_spacing();
294  float length = geom->get_strip_z_spacing();
295 
296  // fill a vector of hits to make things easier - gets every hit in the hitset
297  std::vector <std::pair< TrkrDefs::hitkey, TrkrHit*> > hitvec;
298  TrkrHitSet::ConstRange hitrangei = hitset->getHits();
299  for (TrkrHitSet::ConstIterator hitr = hitrangei.first;
300  hitr != hitrangei.second;
301  ++hitr)
302  {
303  hitvec.push_back(make_pair(hitr->first, hitr->second));
304  }
305  if (Verbosity() > 2)
306  cout << "hitvec.size(): " << hitvec.size() << endl;

308  typedef adjacency_list<vecS, vecS, undirectedS> Graph;
309  Graph G;
310 
311  // Find adjacent strips
312  for (unsigned int i = 0; i < hitvec.size(); i++)
313  {
314  for (unsigned int j = i + 1; j < hitvec.size(); j++)
315  {
316  if (ladder_are_adjacent(hitvec[i], hitvec[j], layer))
317  {
318  add_edge(i, j, G);
319  }
320  }
321 
322  add_edge(i, i, G);
323  }
324 
325  // Find the connections between the vertices of the graph (vertices are the rawhits,
326  // connections are made when they are adjacent to one another)
327  vector<int> component(num_vertices(G));
328 
329  // this is the actual clustering, performed by boost
330  connected_components(G, &component[0]);
331 
332  // Loop over the components(hit cells) compiling a list of the
333  // unique connected groups (ie. clusters).
334  set<int> cluster_ids; // unique components
335 
336  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*> > clusters;
337  for (unsigned int i = 0; i < component.size(); i++)
338  {
339  cluster_ids.insert(component[i]); // one entry per unique cluster id
340  clusters.insert(make_pair(component[i], hitvec[i])); // multiple entries per unique cluster id
341  }
342 
343  // loop over the cluster ID's and make the clusters from the connected hits
344  for (set<int>::iterator clusiter = cluster_ids.begin(); clusiter != cluster_ids.end(); ++clusiter)
345  {
346  int clusid = *clusiter;
347  //cout << " intt clustering: add cluster number " << clusid << endl;
348  // get all hits for this cluster ID only
349  pair<multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator,
350  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator> clusrange = clusters.equal_range(clusid);
351  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator mapiter = clusrange.first;
352 
353  // make the cluster directly in the node tree
354  TrkrDefs::cluskey ckey = InttDefs::genClusKey(hitset->getHitSetKey(), clusid);
355  TrkrClusterv1 *clus = static_cast<TrkrClusterv1 *>((m_clusterlist->findOrAddCluster(ckey))->second);
356 
357  if (Verbosity() > 2)
358  cout << "Filling cluster with key " << ckey << endl;
359 
360  // determine the size of the cluster in phi and z, useful for track fitting the cluster
361  set<int> phibins;
362  set<int> zbins;
363 
364  // tilt refers to a rotation around the radial vector from the origin, and this is zero for the INTT ladders
365  float tilt = 0; //geom->get_strip_tilt();
366 
367  // determine the cluster position...
368  double xsum = 0.0;
369  double ysum = 0.0;
370  double zsum = 0.0;
371  double clus_adc = 0.0;
372  unsigned nhits = 0;
373 
374  for (mapiter = clusrange.first; mapiter != clusrange.second; ++mapiter)
375  {
376  // mapiter->second.first is the hit key
377  //cout << " adding hitkey " << mapiter->second.first << endl;
378  int col = InttDefs::getCol( (mapiter->second).first);
379  int row = InttDefs::getRow( (mapiter->second).first);
380  zbins.insert(col);
381  phibins.insert(row);
382 
383  // mapiter->second.second is the hit
384  double hit_adc = (mapiter->second).second->getAdc();
385 
386  // now get the positions from the geometry
387 
388  double hit_location[3] = {0.0, 0.0, 0.0};
389  geom->find_strip_center(ladder_z_index,
390  ladder_phi_index,
391  col,
392  row,
393  hit_location);
394 
395  if (_make_e_weights[layer])
396  {
397  xsum += hit_location[0] * hit_adc;
398  ysum += hit_location[1] * hit_adc;
399  zsum += hit_location[2] * hit_adc;
400  }
401  else
402  {
403  xsum += hit_location[0];
404  ysum += hit_location[1];
405  zsum += hit_location[2];
406  }
407 
408  clus_adc += hit_adc;
409  ++nhits;
410 
411  // add this cluster-hit association to the association map of (clusterkey,hitkey)
412  m_clusterhitassoc->addAssoc(ckey, mapiter->second.first);
413 
414  if (Verbosity() > 2) cout << " nhits = " << nhits << endl;
415  if (Verbosity() > 2)
416  {
417  cout << " From geometry object: hit x " << hit_location[0] << " hit y " << hit_location[1] << " hit z " << hit_location[2] << endl;
418  cout << " nhits " << nhits << " clusx = " << xsum / nhits << " clusy " << ysum / nhits << " clusz " << zsum / nhits << endl;
419 
420  }
421  }
422 
423  float phisize = phibins.size() * pitch;
424  float zsize = zbins.size() * length;
425 
426  double clusx = NAN;
427  double clusy = NAN;
428  double clusz = NAN;
429 
430 
431  if (_make_e_weights[layer])
432  {
433  clusx = xsum / clus_adc;
434  clusy = ysum / clus_adc;
435  clusz = zsum / clus_adc;
436  }
437  else
438  {
439  clusx = xsum / nhits;
440  clusy = ysum / nhits;
441  clusz = zsum / nhits;
442  }
443 
444  double ladder_location[3] = {0.0, 0.0, 0.0};
445  geom->find_segment_center(ladder_z_index,
446  ladder_phi_index,
447  ladder_location);
448  double ladderphi = atan2(ladder_location[1], ladder_location[0]);
449  ladderphi += geom->get_strip_phi_tilt();
450

452  clus->setAdc(clus_adc);
453  clus->setPosition(0, clusx);
454  clus->setPosition(1, clusy);

472

475  ERR[0][1] = 0.0;
476  ERR[0][2] = 0.0;
477  ERR[1][0] = 0.0;
478  ERR[1][1] = pow(phisize * invsqrt12 * corr_factor, 2);
479  ERR[1][2] = 0.0;
480  ERR[2][0] = 0.0;
481  ERR[2][1] = 0.0;
482  ERR[2][2] = pow(zsize * invsqrt12 * corr_factor, 2);

486  ROT[0][1] = -1.0 * sin(ladderphi);
487  ROT[0][2] = 0.0;
488  ROT[1][0] = sin(ladderphi);
489  ROT[1][1] = cos(ladderphi);
490  ROT[1][2] = 0.0;
491  ROT[2][0] = 0.0;
492  ROT[2][1] = 0.0;
493  ROT[2][2] = 1.0;

498  TILT[0][2] = 0.0;
499  TILT[1][0] = 0.0;

501  TILT[1][2] = -1.0 * sin(tilt);
502  TILT[2][0] = 0.0;
503  TILT[2][1] = sin(tilt);
504  TILT[2][2] = cos(tilt);
505 
506  TMatrixF R(3, 3);
507  R = ROT * TILT;
508  R = ROT;
509 
510  TMatrixF R_T(3, 3);
511  R_T.Transpose(R);
512 
513  TMatrixF COVAR_DIM(3, 3);
514  COVAR_DIM = R * DIM * R_T;
515 
516  clus->setSize(0, 0, COVAR_DIM[0][0]);
517  clus->setSize(0, 1, COVAR_DIM[0][1]);
518  clus->setSize(0, 2, COVAR_DIM[0][2]);
