156 void CaloCalibration::CreateNodeTree(PHCompositeNode *topNode)
157 {
158  PHNodeIterator iter(topNode);
159 
160  PHCompositeNode *dstNode =
161  dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
162  if (!dstNode)
163  {
164  cout << Name() << "::" << detector << "::" << __PRETTY_FUNCTION__
165  << "DST Node missing, doing nothing." << endl;
166  throw runtime_error(
167  "Failed to find DST node in RawTowerCalibration::CreateNodes");
168  }
169 
170  RawTowerNodeName = "TOWER_" + _raw_tower_node_prefix + "_" + detector;
171  _raw_towers =
172  findNode::getClass<RawTowerContainer>(dstNode, RawTowerNodeName.c_str());
173  if (!_raw_towers)
174  {
175  cout << Name() << "::" << detector << "::" << __PRETTY_FUNCTION__ << " "
176  << RawTowerNodeName << " Node missing, doing bail out!" << endl;
177  throw runtime_error("Failed to find " + RawTowerNodeName +
178  " node in RawTowerCalibration::CreateNodes");
179  }
180 
181  // Create the tower nodes on the tree
182  PHNodeIterator dstiter(dstNode);
183  PHCompositeNode *DetNode = dynamic_cast<PHCompositeNode *>(
184  dstiter.findFirst("PHCompositeNode", detector));
185  if (!DetNode)
186  {
187  DetNode = new PHCompositeNode(detector);
188  dstNode->addNode(DetNode);
189  }
190 
191  // Be careful as a previous calibrator may have been registered for this
192  // detector
193  CaliTowerNodeName = "TOWER_" + _calib_tower_node_prefix + "_" + detector;
194  _calib_towers =
195  findNode::getClass<RawTowerContainer>(DetNode, CaliTowerNodeName.c_str());
196  if (!_calib_towers)
197  {
198  _calib_towers = new RawTowerContainer(_raw_towers->getCalorimeterID());
199  PHIODataNode<PHObject> *towerNode = new PHIODataNode<PHObject>(
200  _calib_towers, CaliTowerNodeName.c_str(), "PHObject");
201  DetNode->addNode(towerNode);
202  }
203 
204  // update the parameters on the node tree
205  PHCompositeNode *parNode =
206  dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "RUN"));
207  assert(parNode);
