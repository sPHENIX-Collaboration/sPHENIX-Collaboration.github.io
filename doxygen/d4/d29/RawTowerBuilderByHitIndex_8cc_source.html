94 int RawTowerBuilderByHitIndex::process_event(PHCompositeNode *topNode)
95 {
96  // get hits
97  string NodeNameHits = "G4HIT_" + m_Detector;
98  PHG4HitContainer *g4hit = findNode::getClass<PHG4HitContainer>(topNode, NodeNameHits);
99  if (!g4hit)
100  {
101  cout << "Could not locate g4 hit node " << NodeNameHits << endl;
102  exit(1);
103  }
104 
105  // loop over all hits in the event
106  PHG4HitContainer::ConstIterator hiter;
107  PHG4HitContainer::ConstRange hit_begin_end = g4hit->getHits();
108 
109  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; hiter++)
110  {
111  PHG4Hit *g4hit_i = hiter->second;
112 
113  // Don't include hits with zero energy
114  if (g4hit_i->get_edep() <= 0) continue;
115 
116  /* encode CaloTowerID from j, k index of tower / hit and calorimeter ID */
117  RawTowerDefs::keytype calotowerid = RawTowerDefs::encode_towerid(m_CaloId,
118  g4hit_i->get_index_j(),
119  g4hit_i->get_index_k());
120 
121  /* add the energy to the corresponding tower */
122  RawTowerv1 *tower = dynamic_cast<RawTowerv1 *>(m_Towers->getTower(calotowerid));
123  if (!tower)
124  {
125  tower = new RawTowerv1(calotowerid);
126  tower->set_energy(0);
127  m_Towers->AddTower(tower->get_id(), tower);
128  }
129 
130  tower->add_ecell((g4hit_i->get_index_j() << 16) + g4hit_i->get_index_k(), g4hit_i->get_light_yield());
131  tower->set_energy(tower->get_energy() + g4hit_i->get_light_yield());
132  tower->add_eshower(g4hit_i->get_shower_id(), g4hit_i->get_edep());
133  }
134 
135  float towerE = 0.;
136 
137  if (Verbosity())
138  {
139  towerE = m_Towers->getTotalEdep();
140  }
141 
142  m_Towers->compress(m_Emin);
143  if (Verbosity())
144  {
145  cout << "Energy lost by dropping towers with less than " << m_Emin
146  << " energy, lost energy: " << towerE - m_Towers->getTotalEdep() << endl;
147  m_Towers->identify();
148  RawTowerContainer::ConstRange begin_end = m_Towers->getTowers();
149  RawTowerContainer::ConstIterator iter;
150  for (iter = begin_end.first; iter != begin_end.second; ++iter)
151  {
152  iter->second->identify();
153  }
154  }
155 
156  return Fun4AllReturnCodes::EVENT_OK;
157 }

170 void RawTowerBuilderByHitIndex::CreateNodes(PHCompositeNode *topNode)
171 {
172  PHNodeIterator iter(topNode);
173  PHCompositeNode *runNode = static_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "RUN"));
174  if (!runNode)
175  {
176  std::cerr << PHWHERE << "Run Node missing, doing nothing." << std::endl;
177  throw std::runtime_error("Failed to find Run node in RawTowerBuilderByHitIndex::CreateNodes");
178  }
179 
180  PHCompositeNode *dstNode = static_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
181  if (!dstNode)
182  {
183  std::cerr << PHWHERE << "DST Node missing, doing nothing." << std::endl;
184  throw std::runtime_error("Failed to find DST node in RawTowerBuilderByHitIndex::CreateNodes");
185  }
186 
187  // Create the tower geometry node on the tree

192  runNode->addNode(geomNode);
193 
194  // Find detector node (or create new one if not found)
195  PHNodeIterator dstiter(dstNode);

201  dstNode->addNode(DetNode);

223 bool RawTowerBuilderByHitIndex::ReadGeometryFromTable()
224 {
225  /* Stream to read table from file */
226  ifstream istream_mapping;
227 
228  /* Open the datafile, if it won't open return an error */
229  if (!istream_mapping.is_open())
230  {
231  istream_mapping.open(m_MappingTowerFile);
232  if (!istream_mapping)
233  {
234  cerr << "CaloTowerGeomManager::ReadGeometryFromTable - ERROR Failed to open mapping file " << m_MappingTowerFile << endl;
235  exit(1);
236  }
237  }
238 
239  string line_mapping;
240 
241  while (getline(istream_mapping, line_mapping))
242  {
243  /* Skip lines starting with / including a '#' */
244  if (line_mapping.find("#") != string::npos)
245  {
246  if (Verbosity() > 0)
247  {
248  cout << "RawTowerBuilderByHitIndex: SKIPPING line in mapping file: " << line_mapping << endl;
249  }
250  continue;
251  }
252 
253  istringstream iss(line_mapping);
254 
255  /* If line starts with keyword Tower, add to tower positions */
256  if (line_mapping.find("Tower ") != string::npos)
257  {
258  unsigned idx_j, idx_k, idx_l;
259  double pos_x, pos_y, pos_z;
260  double size_x, size_y, size_z;
261  double rot_x, rot_y, rot_z;
262  double type;
263  string dummys;
264 
265  /* read string- break if error */
266  if (!(iss >> dummys >> type >> idx_j >> idx_k >> idx_l >> pos_x >> pos_y >> pos_z >> size_x >> size_y >> size_z >> rot_x >> rot_y >> rot_z))
267  {
268  cerr << "ERROR in RawTowerBuilderByHitIndex: Failed to read line in mapping file " << m_MappingTowerFile << endl;
269  exit(1);
270  }
271 
272  /* Construct unique Tower ID */
273  unsigned int temp_id = RawTowerDefs::encode_towerid(m_CaloId, idx_j, idx_k);
274 
275  /* Create tower geometry object */
276  RawTowerGeom *temp_geo = new RawTowerGeomv3(temp_id);
277  temp_geo->set_center_x(pos_x);
278  temp_geo->set_center_y(pos_y);
279  temp_geo->set_center_z(pos_z);
280  temp_geo->set_size_x(size_x);
281  temp_geo->set_size_y(size_y);
282  temp_geo->set_size_z(size_z);
283  temp_geo->set_tower_type((int) type);
284 
285  /* Insert this tower into position map */
286  m_Geoms->add_tower_geometry(temp_geo);
287  }
288 
289  /* If line does not start with keyword Tower, read as parameter */
290  else
291  {
292  /* If this line is not a comment and not a tower, save parameter as string / value. */
293  string parname;
294  double parval;
295 
296  /* read string- break if error */
297  if (!(iss >> parname >> parval))
298  {
299  cerr << "ERROR in RawTowerBuilderByHitIndex: Failed to read line in mapping file " << m_MappingTowerFile << endl;
300  exit(1);
301  }
302 
303  m_GlobalParameterMap.insert(make_pair(parname, parval));
304  }
305  }
306 
307  /* Update member variables for global parameters based on parsed parameter file */
308  std::map<string, double>::iterator parit;
309 
310  parit = m_GlobalParameterMap.find("Gx0");
311  if (parit != m_GlobalParameterMap.end())
312  m_GlobalPlaceInX = parit->second;
313 
314  parit = m_GlobalParameterMap.find("Gy0");
315  if (parit != m_GlobalParameterMap.end())
316  m_GlobalPlaceInY = parit->second;
317 
318  parit = m_GlobalParameterMap.find("Gz0");
319  if (parit != m_GlobalParameterMap.end())
320  m_GlobalPlaceInZ = parit->second;
321 
322  parit = m_GlobalParameterMap.find("Grot_x");
323  if (parit != m_GlobalParameterMap.end())
324  m_RotInX = parit->second;
325 
326  parit = m_GlobalParameterMap.find("Grot_y");
327  if (parit != m_GlobalParameterMap.end())
328  m_RotInY = parit->second;
329 
330  parit = m_GlobalParameterMap.find("Grot_z");
331  if (parit != m_GlobalParameterMap.end())
332  m_RotInZ = parit->second;
333 
334  /* Correct tower geometries for global calorimter translation / rotation
335  * after reading parameters from file */
336  RawTowerGeomContainer::ConstRange all_towers = m_Geoms->get_tower_geometries();
337 
338  for (RawTowerGeomContainer::ConstIterator it = all_towers.first;
339  it != all_towers.second; ++it)
340  {
341  double x_temp = it->second->get_center_x();
342  double y_temp = it->second->get_center_y();
343  double z_temp = it->second->get_center_z();
344 
345  /* Rotation */
346  TRotation rot;
347  rot.RotateX(m_RotInX);
348  rot.RotateY(m_RotInY);
349  rot.RotateZ(m_RotInZ);
350 
351  TVector3 v_temp_r(x_temp, y_temp, z_temp);
352  v_temp_r.Transform(rot);
353 
354  /* Translation */
355  double x_temp_rt = v_temp_r.X() + m_GlobalPlaceInX;
356  double y_temp_rt = v_temp_r.Y() + m_GlobalPlaceInY;
357  double z_temp_rt = v_temp_r.Z() + m_GlobalPlaceInZ;
358 
359  /* Update tower geometry object */
360  it->second->set_center_x(x_temp_rt);
361  it->second->set_center_y(y_temp_rt);
362  it->second->set_center_z(z_temp_rt);
363 
364  if (Verbosity() > 2)
365  {
366  cout << "* Local tower x y z : " << x_temp << " " << y_temp << " " << z_temp << endl;
367  cout << "* Globl tower x y z : " << x_temp_rt << " " << y_temp_rt << " " << z_temp_rt << endl;
368  }
369  }
370 
371  return true;
