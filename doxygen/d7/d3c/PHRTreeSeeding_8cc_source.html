417 double PHRTreeSeeding::chisq(const double *xx)
418 {
419  double chi2 = 0.;
420  for (vector<TrkrCluster *>::iterator i = clusterpoints.begin(); i != clusterpoints.end(); ++i)
421  {
422  TrkrCluster *tc = *i;
423  double qa = tc->getError(0, 0);
424  double qb = tc->getError(0, 1);
425  double qc = tc->getError(1, 0);
426  double qd = tc->getError(1, 1);
427  double qz = tc->getZ() - clusterpoints.at(0)->getZ();
428  double predictedx = xx[0] - sin(xx[2]) / xx[4] + cos(xx[4] / tan(xx[3]) * qz + xx[2] - M_PI / 2) / xx[4];
429  double predictedy = xx[1] + cos(xx[2]) / xx[4] + sin(xx[4] / tan(xx[3]) * qz + xx[2] - M_PI / 2) / xx[4];
430  double qx = tc->getX() - predictedx;
431  double qy = tc->getY() - predictedy;
432 
433  chi2 += (qd * qx * qx - (qb + qc) * qx * qy + qa * qy * qy) / (qa * qd - qb * qc);
434  }
435 
436  return chi2;
437 }

520  {
521  double StartPhi = StartCluster->first.get<0>();
522  double StartEta = StartCluster->first.get<1>();
523 
524  vector<pointKey> SecondLayerClusters;
525  QueryTree(_rtree,
526  StartPhi - phisr,
527  StartEta - etasr,
528  (double) _start_layer - 1.5,
529  StartPhi + phisr,
530  StartEta + etasr,
531  (double) _start_layer - 0.5,
532  SecondLayerClusters);
533  cout << " entries in second layer: " << SecondLayerClusters.size() << endl;
534 
535  for (vector<pointKey>::iterator SecondCluster = SecondLayerClusters.begin(); SecondCluster != SecondLayerClusters.end(); ++SecondCluster)
536  {
537  double currentphi = SecondCluster->first.get<0>();
538  double currenteta = SecondCluster->first.get<1>();
539  int lastgoodlayer = _start_layer - 1;
540 
541  int failures = 0;
542 
543  double dphidr = phidiff(StartPhi, currentphi) / (_radii_all[_start_layer] - _radii_all[_start_layer - 1]);
544  double detadr = (StartEta - currenteta) / (_radii_all[_start_layer] - _radii_all[_start_layer - 1]);
545 
546  double ther = (_radii_all[_start_layer] + _radii_all[_start_layer - 1]) / 2;
547 
548  vector<double> curvatureestimates;
549  curvatureestimates.push_back(copysign(2 / sqrt(ther * ther + 1 / dphidr / dphidr), dphidr));
550  vector<double> phi_zigzag;
551  phi_zigzag.push_back(dphidr);
552  vector<double> z_zigzag;
553  z_zigzag.push_back(detadr);
554 
555  vector<TrkrDefs::cluskey> cluskeys;
556  cluskeys.push_back(StartCluster->second);
557  cluskeys.push_back(SecondCluster->second);
558  cout << " phi 1: " << StartPhi
559  << " phi 2: " << currentphi
560  << " dphi: " << dphidr
561  << " eta 1: " << StartEta
562  << " eta 2: " << currenteta
563  << " deta: " << detadr
564  << endl;
565 
566  for (unsigned int newlayer = _start_layer - 2; newlayer >= (_start_layer - 7); newlayer--)
567  {
568  vector<pointKey> newlayer_clusters;
569  cout << " window - "
570  << " phimin " << currentphi - dphidr * (_radii_all[lastgoodlayer] - _radii_all[newlayer]) - phist
571  << " phimax " << currentphi - dphidr * (_radii_all[lastgoodlayer] - _radii_all[newlayer]) + phist
572  << " etamin " << currenteta - etast
573  << " etamax " << currenteta + etast
574  << endl;
575  QueryTree(_rtree,
576  currentphi - dphidr * (_radii_all[lastgoodlayer] - _radii_all[newlayer]) - phist,
577  currenteta - etast,
578  newlayer - 0.5,
579  currentphi - dphidr * (_radii_all[lastgoodlayer] - _radii_all[newlayer]) + phist,
580  currenteta + etast,
581  newlayer + 0.5,
582  newlayer_clusters);
583 
584  if (newlayer_clusters.empty())
585  {
586  failures += 1;
587  if (failures > 2) break; //0.7 2 0.9 3
588  }
589  else
590  {
591  double xinrecord = 100.0;
592  pointKey *xinkey = &*newlayer_clusters.begin();
593  for (std::vector<pointKey>::iterator it = newlayer_clusters.begin(); it != newlayer_clusters.end(); ++it)
594  {
595  double dist = abs(phidiff(it->first.get<0>(), currentphi - dphidr * (_radii_all[lastgoodlayer] - _radii_all[newlayer]))) + abs(it->first.get<1>() - currenteta);
596 
597  cout << " nuphi: " << it->first.get<0>()
598  << " nueta: " << it->first.get<1>()
599  << " dist: " << dist
600  << " lay: " << newlayer
601  << " dl: " << lastgoodlayer - newlayer
602  << " r: " << _radii_all[newlayer]
603  << " dr: " << _radii_all[lastgoodlayer] - _radii_all[newlayer]
604  << endl;
605  if (dist < xinrecord)
606  {
607  *xinkey = *it;
608  xinrecord = dist;
609  }
610  }
611 
612  dphidr = phidiff(currentphi, xinkey->first.get<0>()) / (_radii_all[lastgoodlayer] - _radii_all[newlayer]);
613  detadr = (currenteta - xinkey->first.get<1>()) / (_radii_all[lastgoodlayer] - _radii_all[newlayer]);
614  ther = (_radii_all[lastgoodlayer] - _radii_all[newlayer]) / 2;
615 
616  curvatureestimates.push_back(copysign(2 / sqrt(ther * ther + 1 / dphidr / dphidr), dphidr));
617 
618  phi_zigzag.push_back(dphidr);
619  z_zigzag.push_back(detadr);
620 
621  cluskeys.push_back(xinkey->second);
622 
623  currentphi = xinkey->first.get<0>();
624  currenteta = (currenteta + xinkey->first.get<1>()) / 2;
625 
626  lastgoodlayer = newlayer;
627  }
628  }
629  if (failures > 2) continue; //0.7 2 0.9 3
630 
631  double phi_sum = std::accumulate(phi_zigzag.begin(), phi_zigzag.end(), 0.0);
632  double phi_mean = phi_sum / phi_zigzag.size();
633 
634  std::vector<double> phi_diff(phi_zigzag.size());
635  std::transform(phi_zigzag.begin(), phi_zigzag.end(), phi_diff.begin(),
636  std::bind2nd(std::minus<double>(), phi_mean));
637  double phi_sq_sum = std::inner_product(phi_diff.begin(), phi_diff.end(), phi_diff.begin(), 0.0);
638  double phi_stdev = std::sqrt(phi_sq_sum / (phi_zigzag.size() - 1));
639 
640  double z_sum = std::accumulate(z_zigzag.begin(), z_zigzag.end(), 0.0);
641  double z_mean = z_sum / z_zigzag.size();
642 
643  std::vector<double> z_diff(z_zigzag.size());
644  std::transform(z_zigzag.begin(), z_zigzag.end(), z_diff.begin(),
645  std::bind2nd(std::minus<double>(), z_mean));
646  double z_sq_sum = std::inner_product(z_diff.begin(), z_diff.end(), z_diff.begin(), 0.0);
647  double z_stdev = std::sqrt(z_sq_sum / (z_zigzag.size() - 1));
648 
649  double curv_sum = std::accumulate(curvatureestimates.begin(), curvatureestimates.end(), 0.0);
650  double curv_mean = curv_sum / curvatureestimates.size();
651 
652  std::vector<double> curv_diff(curvatureestimates.size());
653  std::transform(curvatureestimates.begin(), curvatureestimates.end(), curv_diff.begin(),
654  std::bind2nd(std::minus<double>(), curv_mean));
655  double curv_sq_sum = std::inner_product(curv_diff.begin(), curv_diff.end(), curv_diff.begin(), 0.0);
656  double curv_stdev = std::sqrt(curv_sq_sum / (curvatureestimates.size() - 1));
657 
658  const double BQ = 0.01 * 1.4 * 0.299792458;
659  double pt = BQ / abs(curv_mean);
660  double pterror = BQ * curv_stdev / (curv_mean * curv_mean);
661 
662  // pt:z:dz:phi:dphi:c:dc
663  NT->Fill(pt, pterror, z_mean, z_stdev, phi_mean, phi_stdev, curv_mean, curv_stdev, cluskeys.size());
664  SvtxTrack_v1 track;
665  track.set_id(numberofseeds);
666 
667  for (unsigned int j = 0; j < cluskeys.size(); j++)
668  {
669  track.insert_cluster_key(cluskeys.at(j));
670  }
671  track.set_ndf(2 * cluskeys.size() - 5);
672  short int helicity = 1;
673  if (StartPhi * curv_mean < 0) helicity -= 2;
674  track.set_charge(-helicity);
675 
676  TrkrCluster *cl = _cluster_map->findCluster(StartCluster->second);
677  track.set_x(_vertex->get_x()); //track.set_x(cl->getX());
678  track.set_y(_vertex->get_y()); //track.set_y(cl->getY());
679  track.set_z(_vertex->get_z()); //track.set_z(cl->getZ());
680  track.set_px(pt * cos(StartPhi));
681  track.set_py(pt * sin(StartPhi));
682  track.set_pz(pt / tan(2 * atan(exp(-StartEta))));
683  track.set_error(0, 0, cl->getError(0, 0));
684  track.set_error(0, 1, cl->getError(0, 1));
685  track.set_error(0, 2, cl->getError(0, 2));
686  track.set_error(1, 1, cl->getError(1, 1));
687  track.set_error(1, 2, cl->getError(1, 2));
688  track.set_error(2, 2, cl->getError(2, 2));
689  track.set_error(3, 3, pterror * pterror * cos(StartPhi) * cos(StartPhi));
690  track.set_error(4, 4, pterror * pterror * sin(StartPhi) * sin(StartPhi));
691  track.set_error(5, 5, pterror * pterror / tan(2 * atan(exp(-StartEta))) / tan(2 * atan(exp(-StartEta))));
692  _track_map->insert(&track);
693 
694  numberofseeds++;
695  }
696  }
697  }
698  t_seed->stop();
699  cout << "number of seeds " << numberofseeds << endl;
700  cout << "seeding time: " << t_seed->get_accumulated_time() / 1000 << " s" << endl;
701

707  NT->Write();
708  fpara.Close();
709

711 }
