98 void PHG4MvtxDigitizer::CalculateMvtxLadderCellADCScale(PHCompositeNode *topNode)
99 {
100  // defaults to 8-bit ADC, short-axis MIP placed at 1/4 dynamic range
101 
102  PHG4CylinderGeomContainer *geom_container = findNode::getClass<PHG4CylinderGeomContainer>(topNode, "CYLINDERGEOM_MVTX");
103 
104  if (!geom_container) return;
105 
106  if (Verbosity())
107  cout << "Found CYLINDERGEOM_MVTX node" << endl;
108 
109  PHG4CylinderGeomContainer::ConstRange layerrange = geom_container->get_begin_end();
110  for (PHG4CylinderGeomContainer::ConstIterator layeriter = layerrange.first;
111  layeriter != layerrange.second;
112  ++layeriter)
113  {
114  int layer = layeriter->second->get_layer();
115  float thickness = (layeriter->second)->get_pixel_thickness();
116  float pitch = (layeriter->second)->get_pixel_x();
117  float length = (layeriter->second)->get_pixel_z();
118 
119  float minpath = pitch;
120  if (length < minpath) minpath = length;
121  if (thickness < minpath) minpath = thickness;
122  float mip_e = 0.003876 * minpath;
123 
124  if (Verbosity())
125  cout << "mip_e = " << mip_e << endl;
126 
127  if (_max_adc.find(layer) == _max_adc.end())
128  {
129  _max_adc[layer] = 255;
130  _energy_scale[layer] = mip_e / 64;
131  }
132  }
133 
134  return;
135 }

137 void PHG4MvtxDigitizer::DigitizeMvtxLadderCells(PHCompositeNode *topNode)
138 {
139  //----------
140  // Get Nodes
141  //----------
142 
143  // new containers
144  //=============
145  // Get the TrkrHitSetContainer node
146  TrkrHitSetContainer *trkrhitsetcontainer = findNode::getClass<TrkrHitSetContainer>(topNode, "TRKR_HITSET");
147  if (!trkrhitsetcontainer)
148  {
149  cout << "Could not locate TRKR_HITSET node, quit! " << endl;

152 
153  // Digitization
154 
155  // We want all hitsets for the Mvtx
156  TrkrHitSetContainer::ConstRange hitset_range = trkrhitsetcontainer->getHitSets(TrkrDefs::TrkrId::mvtxId);
157  for (TrkrHitSetContainer::ConstIterator hitset_iter = hitset_range.first;
158  hitset_iter != hitset_range.second;
159  ++hitset_iter)
160  {
161  // we have an itrator to one TrkrHitSet for the mvtx from the trkrHitSetContainer
162  // get the hitset key so we can find the layer
163  TrkrDefs::hitsetkey hitsetkey = hitset_iter->first;
164  int layer = TrkrDefs::getLayer(hitsetkey);
165  if (Verbosity() > 1) cout << "PHG4MvtxDigitizer: found hitset with key: " << hitsetkey << " in layer " << layer << endl;
166 
167  // get all of the hits from this hitset
168  TrkrHitSet *hitset = hitset_iter->second;
169  TrkrHitSet::ConstRange hit_range = hitset->getHits();
170  for (TrkrHitSet::ConstIterator hit_iter = hit_range.first;
171  hit_iter != hit_range.second;
172  ++hit_iter)
173  {
174  TrkrHit *hit = hit_iter->second;
175 
176  // Convert the signal value to an ADC value and write that to the hit
177  unsigned int adc = hit->getEnergy() / _energy_scale[layer];
178  if (adc > _max_adc[layer]) adc = _max_adc[layer];
179  hit->setAdc(adc);
180 
181  if (Verbosity() > 0) cout << " PHG4MvtxDigitizer: found hit with key: " << hit_iter->first << " and signal " << hit->getEnergy() << " and adc " << adc << endl;
182  }
183  }
184 
185  // end new containers
186  //===============
187 
188  return;
