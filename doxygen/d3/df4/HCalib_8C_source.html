216 void HCalib::GetNodes(PHCompositeNode *topNode, const string &det)
217 {
218  static int ncalls = 0;

224  ncalls++;

234  }
235 
236  //Proto slats

239  proto_slats = findNode::getClass<PHG4ScintillatorSlatContainer>(topNode, cellnodename);
