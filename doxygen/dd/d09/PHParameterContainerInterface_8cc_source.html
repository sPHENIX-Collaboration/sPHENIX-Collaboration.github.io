214  std::pair< PHParameters::dIter, PHParameters::dIter> double_begin_end = iter->second->get_all_double_params();
215  for (PHParameters::dIter diter = double_begin_end.first; diter != double_begin_end.second; ++diter)
216  {
217  params->set_double_param(diter->first,diter->second);
218  }
219

221  for (PHParameters::iIter iiter = int_begin_end.first; iiter != int_begin_end.second; ++iiter)
