273  // cout << "number of layers: " << g4hit->num_layers() << endl;
274  // cout << "number of hits: " << g4hit->size() << endl;
275  // for (layer = layer_begin_end.first; layer != layer_begin_end.second; layer++)
276  // {
277  // cout << "layer number: " << *layer << endl;
278  // }
279

281  {
282  // only handle layers/detector ids which have parameters set

284  {
285  continue;

287  PHG4HitContainer::ConstIterator hiter;
288  PHG4HitContainer::ConstRange hit_begin_end = g4hit->getHits(*layer);

290  int nxbins = n_x_z_bins[*layer].first;

293

295  {
296  cellptarray.resize(nbins, 0);
297  }
298 
299  // ------- eta/x binning ------------------------------------------------------------------------
300  if (binning[*layer] == PHG4CellDefs::etaphibinning)
301  {
302  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; hiter++)
303  {
304  // checking ADC timing integration window cut
305  if (hiter->second->get_t(0) > tmin_max[*layer].second) continue;
306  if (hiter->second->get_t(1) < tmin_max[*layer].first) continue;
307 
308  pair<double, double> etax[2];
309  double xbin[2];
310  double etabin[2];
311  for (int i = 0; i < 2; i++)
312  {
313  etax[i] = PHG4Utils::get_etaphi(hiter->second->get_x(i), hiter->second->get_y(i), hiter->second->get_z(i));
314  etabin[i] = geo->get_etabin(etax[i].first);
315  xbin[i] = geo->get_xbin(etax[i].second);
316  }
317 
318  // check bin range
319  if (xbin[0] < 0 || xbin[0] >= nxbins || xbin[1] < 0 || xbin[1] >= nxbins)
320  {
321  continue;
322  }
323  if (etabin[0] < 0 || etabin[0] >= nzbins || etabin[1] < 0 || etabin[1] >= nzbins)
324  {
325  continue;
326  }
327 
328  if (etabin[0] < 0)
329  {
330  if (Verbosity() > 0)
331  {
332  hiter->second->identify();
333  }
334  continue;
335  }
336 
337  sum_energy_g4hit += hiter->second->get_edep();
338  int intxbin = xbin[0];
339  int intetabin = etabin[0];
340  int intxbinout = xbin[1];
341  int intetabinout = etabin[1];
342 
343  // Determine all fired cells
344 
345  double ax = (etax[0]).second; // x
346  double ay = (etax[0]).first; // eta
347  double bx = (etax[1]).second;
348  double by = (etax[1]).first;
349  if (intxbin > intxbinout)
350  {
351  int tmp = intxbin;
352  intxbin = intxbinout;
353  intxbinout = tmp;
354  }
355 
356  if (intetabin > intetabinout)
357  {
358  int tmp = intetabin;
359  intetabin = intetabinout;
360  intetabinout = tmp;
361  }
362 
363  double trklen = sqrt((ax - bx) * (ax - bx) + (ay - by) * (ay - by));
364  // if entry and exit hit are the same (seems to happen rarely), trklen = 0
365  // which leads to a 0/0 and an NaN in edep later on
366  // this code does for particles in the same cell a trklen/trklen (vdedx[ii]/trklen)
367  // so setting this to any non zero number will do just fine
368  // I just pick -1 here to flag those strange hits in case I want t oanalyze them
369  // later on
370  if (trklen == 0)
371  {
372  trklen = -1.;
373  }
374  vector<int> vx;
375  vector<int> veta;
376  vector<double> vdedx;
377 
378  if (intxbin == intxbinout && intetabin == intetabinout) // single cell fired
379  {
380  if (Verbosity() > 0) cout << "SINGLE CELL FIRED: " << intxbin << " " << intetabin << endl;
381  vx.push_back(intxbin);
382  veta.push_back(intetabin);
383  vdedx.push_back(trklen);
384  }
385  else
386  {
387  for (int ibp = intxbin; ibp <= intxbinout; ibp++)
388  {
389  for (int ibz = intetabin; ibz <= intetabinout; ibz++)
390  {
391  double cx = geo->get_xcenter(ibp) - geo->get_xstep() / 2.;
392  double dx = geo->get_xcenter(ibp) + geo->get_xstep() / 2.;
393  double cy = geo->get_etacenter(ibz) - geo->get_etastep() / 2.;
394  double dy = geo->get_etacenter(ibz) + geo->get_etastep() / 2.;
395  //cout << "##### line: " << ax << " " << ay << " " << bx << " " << by << endl;
396  //cout << "####### cell: " << cx << " " << cy << " " << dx << " " << dy << endl;
397  pair<bool, double> intersect = PHG4Utils::line_and_rectangle_intersect(ax, ay, bx, by, cx, cy, dx, dy);
398  if (intersect.first)
399  {
400  if (Verbosity() > 0) cout << "CELL FIRED: " << ibp << " " << ibz << " " << intersect.second << endl;
401  vx.push_back(ibp);
402  veta.push_back(ibz);
403  vdedx.push_back(intersect.second);
404  }
405  }
406  }
407  }

409

411  for (unsigned int ii = 0; ii < vx.size(); ii++)
412  {
413  tmpsum += vdedx[ii];

421  int ixbin = vx[i1];

479  }

520 int PHG4BlockCellReco::CheckEnergy(PHCompositeNode *topNode)
521 {
522  PHG4CellContainer *cells = findNode::getClass<PHG4CellContainer>(topNode, cellnodename);
523  double sum_energy_cells = 0.;
524  double sum_energy_stored_hits = 0.;
525  double sum_energy_stored_showers = 0.;
526 
527  PHG4CellContainer::ConstRange cell_begin_end = cells->getCells();
528  PHG4CellContainer::ConstIterator citer;
529  for (citer = cell_begin_end.first; citer != cell_begin_end.second; ++citer)
530  {
531  sum_energy_cells += citer->second->get_edep();
532  PHG4Cell::EdepConstRange cellrange = citer->second->get_g4hits();
533  for (PHG4Cell::EdepConstIterator iter = cellrange.first; iter != cellrange.second; ++iter)
534  {
535  sum_energy_stored_hits += iter->second;
536  }
537  PHG4Cell::ShowerEdepConstRange shwrrange = citer->second->get_g4showers();
538  for (PHG4Cell::ShowerEdepConstIterator iter = shwrrange.first; iter != shwrrange.second; ++iter)
539  {
540  sum_energy_stored_showers += iter->second;
541  }
542  }
543 
544  // the fractional eloss for particles traversing eta bins leads to minute rounding errors
545  if (sum_energy_stored_hits > 0)
546  {
547  if (fabs(sum_energy_cells - sum_energy_stored_hits) / sum_energy_cells > 1e-6)
548  {
549  cout << "energy mismatch between cell energy " << sum_energy_cells
550  << " and stored hit energies " << sum_energy_stored_hits
551  << endl;
552  }
553  }
554  if (sum_energy_stored_showers > 0)
555  {
556  if (fabs(sum_energy_cells - sum_energy_stored_showers) / sum_energy_cells > 1e-6)
557  {
558  cout << "energy mismatch between cell energy " << sum_energy_cells
559  << " and stored shower energies " << sum_energy_stored_showers
560  << endl;
561  }
562  }
563 
564  if (fabs(sum_energy_cells - sum_energy_g4hit) / sum_energy_g4hit > 1e-6)
565  {
566  cout << "energy mismatch between cells: " << sum_energy_cells
567  << " and hits: " << sum_energy_g4hit
568  << " diff sum(cells) - sum(hits): " << sum_energy_cells - sum_energy_g4hit
569  << endl;
