597 void SvtxClusterEval::fill_cluster_layer_map()
598 {
599  // loop over all the clusters
600  TrkrClusterContainer::ConstRange all_clusters = _clustermap->getClusters();
601  for (TrkrClusterContainer::ConstIterator iter = all_clusters.first; iter != all_clusters.second; ++iter)
602  {
603  TrkrDefs::cluskey cluster_key = iter->first;
604  unsigned int ilayer = TrkrDefs::getLayer(cluster_key);
605  TrkrCluster *cluster = iter->second;
606  float clus_phi = fast_approx_atan2(cluster->getY(), cluster->getX());
607 
608  multimap<unsigned int, innerMap>::iterator it = _clusters_per_layer.find(ilayer);
609  if (it == _clusters_per_layer.end())
610  {
611  it = _clusters_per_layer.insert(make_pair(ilayer, innerMap()));
612  }
613  it->second.insert(make_pair(clus_phi, cluster_key));
614 
615  //address wrapping along +/-PI by filling larger area of the map
616  if (clus_phi - (-M_PI) < _clusters_searching_window) it->second.insert(make_pair(clus_phi + 2 * M_PI, cluster_key));
617  if (M_PI - clus_phi < _clusters_searching_window) it->second.insert(make_pair(clus_phi - 2 * M_PI, cluster_key));
618  }
619  return;

637 float SvtxClusterEval::fast_approx_atan2(float y, float x)
638 {
639  if (x != 0.0f)
640  {
641  if (fabsf(x) > fabsf(y))
642  {
643  const float z = y / x;
644  if (x > 0.0)
645  {
646  // atan2(y,x) = atan(y/x) if x > 0
647  return fast_approx_atan2(z);
648  }
649  else if (y >= 0.0)
650  {
651  // atan2(y,x) = atan(y/x) + PI if x < 0, y >= 0
652  return fast_approx_atan2(z) + M_PI;
653  }
654  else
655  {
656  // atan2(y,x) = atan(y/x) - PI if x < 0, y < 0
657  return fast_approx_atan2(z) - M_PI;
658  }
659  }
660  else // Use property atan(y/x) = PI/2 - atan(x/y) if |y/x| > 1.
661  {
662  const float z = x / y;
663  if (y > 0.0)
664  {
665  // atan2(y,x) = PI/2 - atan(x/y) if |y/x| > 1, y > 0
666  return -fast_approx_atan2(z) + M_PI_2;
667  }
668  else
669  {
670  // atan2(y,x) = -PI/2 - atan(x/y) if |y/x| > 1, y < 0
671  return -fast_approx_atan2(z) - M_PI_2;
672  }
673  }
674  }
675  else
676  {
677  if (y > 0.0f) // x = 0, y > 0
678  {
679  return M_PI_2;
680  }
681  else if (y < 0.0f) // x = 0, y < 0
682  {
683  return -M_PI_2;
684  }
685  }
686  return 0.0f; // x,y = 0. Could return NaN instead.
687 }

689 float SvtxClusterEval::fast_approx_atan2(float z)
690 {
691  // Polynomial approximating arctangenet on the range -1,1.
692  // Max error < 0.005 (or 0.29 degrees)
693  const float n1 = 0.97239411f;
694  const float n2 = -0.19194795f;
695  return (n1 + n2 * z * z) * z;
696 }
