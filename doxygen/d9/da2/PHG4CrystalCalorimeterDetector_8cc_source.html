66 int PHG4CrystalCalorimeterDetector::IsInCrystalCalorimeter(G4VPhysicalVolume* volume) const
67 {
68  if (volume->GetName().find(_towerlogicnameprefix) != string::npos)
69  {
70  if (volume->GetName().find("crystal") != string::npos)
71  {
72  return 1;
73  }
74  /* only record energy in actual absorber- drop energy lost in air gaps inside envelope */
75  else if (volume->GetName().find("absorber") != string::npos)
76  {
77  return -1;
78  }
79  else if (volume->GetName().find("envelope") != string::npos)
80  {
81  return 0;
82  }
83  }
84 
85  return 0;

143 PHG4CrystalCalorimeterDetector::ConstructTower()
144 {
145  if (Verbosity() > 0)
146  {
147  cout << "PHG4CrystalCalorimeterDetector: Build logical volume for single tower..." << endl;
148  }
149 
150  G4double carbon_thickness = 0.009 * cm;
151  G4double airgap_crystal_carbon = 0.012 * cm;
152 
153  /* dimesnions of full tower */
154  G4double tower_dx = _crystal_dx + 2 * (carbon_thickness + airgap_crystal_carbon);
155  G4double tower_dy = _crystal_dy + 2 * (carbon_thickness + airgap_crystal_carbon);
156  G4double tower_dz = _crystal_dz;
157 
158  /* create logical volume for single tower */
159  G4Material* material_air = G4Material::GetMaterial("G4_AIR");
160 
161  G4VSolid* single_tower_solid = new G4Box(G4String("single_tower_solid"),
162  tower_dx / 2.0,
163  tower_dy / 2.0,
164  tower_dz / 2.0);
165 
166  G4LogicalVolume* single_tower_logic = new G4LogicalVolume(single_tower_solid,
167  material_air,
168  "single_tower_logic",
169  0, 0, 0);
170 
171  /* create geometry volume for crystal inside single_tower */
172  G4VSolid* solid_crystal = new G4Box(G4String("single_crystal_solid"),
173  _crystal_dx / 2.0,
174  _crystal_dy / 2.0,
175  _crystal_dz / 2.0);
176 
177  /* create geometry volume for frame (carbon fober shell) inside single_tower */
178  G4VSolid* Carbon_hunk_solid = new G4Box(G4String("Carbon_hunk_solid"),
179  tower_dx / 2.0,
180  tower_dy / 2.0,
181  ((tower_dz / 2.0) - 1 * mm));
182 
183  G4double lead_dx = tower_dx - 2.0 * carbon_thickness;
184  G4double lead_dy = tower_dy - 2.0 * carbon_thickness;
185 
186  G4VSolid* lead_solid = new G4Box(G4String("lead_solid"),
187  lead_dx / 2.0,
188  lead_dy / 2.0,
189  tower_dz / 2.0);
190 
191  G4SubtractionSolid* Carbon_shell_solid = new G4SubtractionSolid(G4String("Carbon_Shell_solid"),
192  Carbon_hunk_solid,
193  lead_solid,
194  0,
195  G4ThreeVector(0.00 * mm, 0.00 * mm, 0.00 * mm));
196 
197  /* create logical volumes for crystal inside single_tower */
198  G4Material* material_crystal = G4Material::GetMaterial(_materialCrystal.c_str());
199 
200  G4LogicalVolume* logic_crystal = new G4LogicalVolume(solid_crystal,
201  material_crystal,
202  "single_crystal_logic",
203  0, 0, 0);
204 
205  GetDisplayAction()->AddVolume(logic_crystal, "Crystal");
206 
207  /* create logical volumes for structural frame */
208  //Carbon Fiber
209  G4double a = 12.01 * g / mole;
210  G4Element* elC = new G4Element("Carbon", "C", 6., a);
211  G4double density_carbon_fiber = 0.144 * g / cm3;
212  G4Material* CarbonFiber = new G4Material("CarbonFiber", density_carbon_fiber, 1);
213  CarbonFiber->AddElement(elC, 1);
214 
215  G4Material* material_shell = CarbonFiber;
216 
217  G4LogicalVolume* logic_shell = new G4LogicalVolume(Carbon_shell_solid,
218  material_shell,
219  "single_absorber_logic",
220  0, 0, 0);
221 
222  GetDisplayAction()->AddVolume(logic_shell, "CarbonShell");
223 
224  /* Place structural frame in logical tower volume */
225  ostringstream name_shell;
226  name_shell.str("");
227  name_shell << _towerlogicnameprefix << "_single_absorber" << endl;
228 
229  new G4PVPlacement(0, G4ThreeVector(0, 0, 0),
230  logic_shell,
231  name_shell.str().c_str(),
232  single_tower_logic,
233  0, 0, OverlapCheck());
234 
235  /* Place crystal in logical tower volume */
236  ostringstream name_crystal;
237  name_crystal.str("");
238  name_crystal << _towerlogicnameprefix << "_single_crystal" << endl;
239 
240  new G4PVPlacement(0, G4ThreeVector(0, 0, 0),
241  logic_crystal,
242  name_crystal.str().c_str(),
243  single_tower_logic,
244  0, 0, OverlapCheck());
245 
246  if (Verbosity() > 0)
247  {
248  cout << "PHG4CrystalCalorimeterDetector: Building logical volume for single tower done." << endl;
249  }

277 int PHG4CrystalCalorimeterDetector::ParseParametersFromTable()
278 {
279  /* Open the datafile, if it won't open return an error */
280  ifstream istream_mapping;
281  if (!istream_mapping.is_open())
282  {
283  istream_mapping.open(_mapping_tower_file.c_str());
284  if (!istream_mapping)
285  {
286  cerr << "ERROR in PHG4CrystalCalorimeterDetector: Failed to open mapping file " << _mapping_tower_file << endl;
287  exit(1);
288  }
289  }
290 
291  /* loop over lines in file */
292  string line_mapping;
293  while (getline(istream_mapping, line_mapping))
294  {
295  /* Skip lines starting with / including a '#' */
296  if (line_mapping.find("#") != string::npos)
297  {
298  if (Verbosity() > 0)
299  {
300  cout << "PHG4CrystalCalorimeterDetector: SKIPPING line in mapping file: " << line_mapping << endl;
301  }
302  continue;
303  }
304 
305  istringstream iss(line_mapping);
306 
307  /* If line starts with keyword Tower, add to tower positions */
308  if (line_mapping.find("Tower ") != string::npos)
309  {
310  unsigned idx_j, idx_k, idx_l;
311  G4double pos_x, pos_y, pos_z;
312  G4double size_x, size_y, size_z;
313  G4double rot_x, rot_y, rot_z;
314  G4double dummy;
315  string dummys;
316 
317  /* read string- break if error */
318  if (!(iss >> dummys >> dummy >> idx_j >> idx_k >> idx_l >> pos_x >> pos_y >> pos_z >> size_x >> size_y >> size_z >> rot_x >> rot_y >> rot_z))
319  {
320  cerr << "ERROR in PHG4CrystalCalorimeterDetector: Failed to read line in mapping file " << _mapping_tower_file << endl;
321  exit(1);
322  }
323 
324  /* Construct unique name for tower */
325  /* Mapping file uses cm, this class uses mm for length */
326  ostringstream towername;
327  towername.str("");
328  towername << _towerlogicnameprefix << "_j_" << idx_j << "_k_" << idx_k;
329 
330  /* Add Geant4 units */
331  pos_x = pos_x * cm;
332  pos_y = pos_y * cm;
333  pos_z = pos_z * cm;
334 
335  /* insert tower into tower map */
336  towerposition tower_new;
337  tower_new.x = pos_x;
338  tower_new.y = pos_y;
339  tower_new.z = pos_z;
340  _map_tower.insert(make_pair(towername.str(), tower_new));
341  }
342  else
343  {
344  /* If this line is not a comment and not a tower, save parameter as string / value. */
345  string parname;
346  G4double parval;
347 
348  /* read string- break if error */
349  if (!(iss >> parname >> parval))
350  {
351  cerr << "ERROR in PHG4CrystalCalorimeterDetector: Failed to read line in mapping file " << _mapping_tower_file << endl;
352  exit(1);
353  }
354 
355  _map_global_parameter.insert(make_pair(parname, parval));
356  }
357  }
358 
359  /* Update member variables for global parameters based on parsed parameter file */
360  std::map<string, G4double>::iterator parit;
361 
362  parit = _map_global_parameter.find("Gcrystal_dx");
363  if (parit != _map_global_parameter.end())
364  _crystal_dx = parit->second * cm;
365 
366  parit = _map_global_parameter.find("Gcrystal_dy");
367  if (parit != _map_global_parameter.end())
368  _crystal_dy = parit->second * cm;
369 
370  parit = _map_global_parameter.find("Gcrystal_dz");
371  if (parit != _map_global_parameter.end())
372  _crystal_dz = parit->second * cm;
373 
374  parit = _map_global_parameter.find("Gr1_inner");
375  if (parit != _map_global_parameter.end())
376  _rMin1 = parit->second * cm;
377 
378  parit = _map_global_parameter.find("Gr1_outer");
379  if (parit != _map_global_parameter.end())
380  _rMax1 = parit->second * cm;
381 
382  parit = _map_global_parameter.find("Gr2_inner");
383  if (parit != _map_global_parameter.end())
384  _rMin2 = parit->second * cm;
385 
386  parit = _map_global_parameter.find("Gr2_outer");
387  if (parit != _map_global_parameter.end())
388  _rMax2 = parit->second * cm;
389 
390  parit = _map_global_parameter.find("Gdz");
391  if (parit != _map_global_parameter.end())
392  _dZ = parit->second * cm;
393 
394  parit = _map_global_parameter.find("Gx0");
395  if (parit != _map_global_parameter.end())
396  _place_in_x = parit->second * cm;
397 
398  parit = _map_global_parameter.find("Gy0");
399  if (parit != _map_global_parameter.end())
400  _place_in_y = parit->second * cm;
401 
402  parit = _map_global_parameter.find("Gz0");
403  if (parit != _map_global_parameter.end())
404  _place_in_z = parit->second * cm;
405 
406  parit = _map_global_parameter.find("Grot_x");
407  if (parit != _map_global_parameter.end())
408  _rot_in_x = parit->second;
409
