78 inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
79 {
80  // convert to row/col w/o over/underflow check
81  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
82  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
83  iRow = int(xRow / PitchRow);
84  iCol = int(zCol / PitchCol);
85  if (xRow < 0)
86  iRow -= 1;
87  if (zCol < 0)
88  iCol -= 1;
89 }

93 {
94  // convert to row/col
95  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt left edge of Active matrix
96  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
97  if (xRow < 0 || xRow >= ActiveMatrixSizeRows || zCol < 0 || zCol >= ActiveMatrixSizeCols) {
98  iRow = iCol = -1;
99  return false;
100  }
101  iRow = int(xRow / PitchRow);
102  iCol = int(zCol / PitchCol);
103  return true;
