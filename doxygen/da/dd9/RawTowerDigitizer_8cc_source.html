98 int RawTowerDigitizer::process_event(PHCompositeNode *topNode)
99 {
100  if (Verbosity())
101  {
102  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
103  << "Process event entered. "
104  << "Digitalization method: ";
105 
106  if (m_DigiAlgorithm == kNo_digitization)
107  {
108  cout << "directly pass the energy of sim tower to digitalized tower";
109  }
110  else if (m_DigiAlgorithm == kSimple_photon_digitization)
111  {
112  cout << "simple digitization with photon statistics, ADC conversion and pedstal";
113  }
114  cout << endl;
115  }
116  // loop over all possible towers, even empty ones. The digitization can add towers containing
117  // pedestals
118  RawTowerGeomContainer::ConstRange all_towers = m_RawTowerGeom->get_tower_geometries();
119 
120  double deadChanEnergy = 0;
121 
122  for (RawTowerGeomContainer::ConstIterator it = all_towers.first;
123  it != all_towers.second; ++it)
124  {
125  const RawTowerDefs::keytype key = it->second->get_id();
126  if (m_TowerType >= 0)
127  {
128  // Skip towers that don't match the type we are supposed to digitize
129  if (m_TowerType != it->second->get_tower_type())
130  {
131  continue;
132  }
133  }
134 
135  RawTower *sim_tower = m_SimTowers->getTower(key);
136  if (m_DeadMap)
137  {
138  if (m_DeadMap->isDeadTower(key))
139  {
140  if (sim_tower) deadChanEnergy += sim_tower->get_energy();
141 
142  sim_tower = nullptr;
143 
144  if (Verbosity() >= VERBOSITY_MORE)
145  {
146  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
147  << " apply dead tower " << key << endl;
148  }
149  }
150  }
151 
152  RawTower *digi_tower = nullptr;
153 
154  if (m_DigiAlgorithm == kNo_digitization)
155  {
156  // for no digitization just copy existing towers
157  if (sim_tower)
158  {
159  digi_tower = new RawTowerv1(*sim_tower);
160  }
161  }
162  else if (m_DigiAlgorithm == kSimple_photon_digitization)
163  {
164  // for photon digitization towers can be created if sim_tower is null pointer
165  digi_tower = simple_photon_digitization(sim_tower);
166  }
167  else if (m_DigiAlgorithm == kSiPM_photon_digitization)
168  {
169  // for photon digitization towers can be created if sim_tower is null pointer
170  digi_tower = sipm_photon_digitization(sim_tower);
171  }
172  else
173  {
174  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
175  << " invalid digitization algorithm #" << m_DigiAlgorithm
176  << endl;
177 
178  return Fun4AllReturnCodes::ABORTRUN;
179  }
180 
181  if (digi_tower)
182  {
183  m_RawTowers->AddTower(key, digi_tower);
184 
185  if (Verbosity() >= VERBOSITY_MORE)
186  {
187  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
188  << " output tower:"
189  << endl;
190  digi_tower->identify();
191  }
192  }
193  }
194 
195  if (Verbosity())
196  {
197  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
198  << "input sum energy = " << m_SimTowers->getTotalEdep() << " GeV"
199  << ", dead channel masked energy = " << deadChanEnergy << " GeV"
200  << ", output sum digitalized value = " << m_RawTowers->getTotalEdep() << " ADC"
201  << endl;
202  }
203  return Fun4AllReturnCodes::EVENT_OK;
204 }

207 RawTowerDigitizer::simple_photon_digitization(RawTower *sim_tower)
208 {
209  RawTower *digi_tower = nullptr;
210 
211  double energy = 0;
212  if (sim_tower)
213  {
214  energy = sim_tower->get_energy();
215  }
216  const double photon_count_mean = energy * m_PhotonElecYieldVisibleGeV;
217  const int photon_count = gsl_ran_poisson(m_RandomGenerator, photon_count_mean);
218  const int signal_ADC = floor(photon_count / m_PhotonElecADC);
219 
220  const double pedstal = m_PedstalCentralADC + ((m_PedstalWidthADC > 0) ? gsl_ran_gaussian(m_RandomGenerator, m_PedstalWidthADC) : 0);
221  const int sum_ADC = signal_ADC + (int) pedstal;
222 
223  if (sum_ADC > m_ZeroSuppressionADC)
224  {
225  // create new digitalizaed tower
226  if (sim_tower)
227  {
228  digi_tower = new RawTowerv1(*sim_tower);
229  }
230  else
231  {
232  digi_tower = new RawTowerv1();
233  }
234  digi_tower->set_energy((double) sum_ADC);
235  }
236 
237  if (Verbosity() >= 2)
238  {
239  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
240  << endl;
241 
242  cout << "input: ";
243  if (sim_tower)
244  {
245  sim_tower->identify();
246  }
247  else
248  {
249  cout << "None" << endl;
250  }
251  cout << "output based on "

336 void RawTowerDigitizer::CreateNodes(PHCompositeNode *topNode)
337 {
338  PHNodeIterator iter(topNode);
339  PHCompositeNode *runNode = static_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "RUN"));
340  if (!runNode)
341  {
342  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
343  << "Run Node missing, doing nothing." << endl;
344  throw std::runtime_error(
345  "Failed to find Run node in RawTowerDigitizer::CreateNodes");
346  }
347 
348  string TowerGeomNodeName = "TOWERGEOM_" + m_Detector;
349  m_RawTowerGeom = findNode::getClass<RawTowerGeomContainer>(topNode, TowerGeomNodeName);
350  if (!m_RawTowerGeom)
351  {
352  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
353  << " " << TowerGeomNodeName << " Node missing, doing bail out!"
354  << endl;
355  throw std::runtime_error("Failed to find " + TowerGeomNodeName + " node in RawTowerDigitizer::CreateNodes");
356  }
357 
358  if (Verbosity() >= 1)
359  {
360  m_RawTowerGeom->identify();
361  }
362 
363  const string deadMapName = "DEADMAP_" + m_Detector;
364  m_DeadMap = findNode::getClass<RawTowerDeadMap>(topNode, deadMapName);
365  if (m_DeadMap)
366  {
367  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
368  << " use dead map: ";
369  m_DeadMap->identify();
370  }
371 
372  PHCompositeNode *dstNode = dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
373  if (!dstNode)
374  {
375  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
376  << "DST Node missing, doing nothing." << endl;
377  throw std::runtime_error("Failed to find DST node in RawTowerDigitizer::CreateNodes");
378  }
379 
380  string SimTowerNodeName = "TOWER_" + m_SimTowerNodePrefix + "_" + m_Detector;
381  m_SimTowers = findNode::getClass<RawTowerContainer>(dstNode, SimTowerNodeName);
382  if (!m_SimTowers)
383  {
384  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
385  << " " << SimTowerNodeName << " Node missing, doing bail out!"
386  << endl;
387  throw std::runtime_error("Failed to find " + SimTowerNodeName + " node in RawTowerDigitizer::CreateNodes");
388  }
389 
390  // Create the tower nodes on the tree
391  PHNodeIterator dstiter(dstNode);
392  PHCompositeNode *DetNode = dynamic_cast<PHCompositeNode *>(dstiter.findFirst("PHCompositeNode", m_Detector));
393  if (!DetNode)
394  {
395  DetNode = new PHCompositeNode(m_Detector);
396  dstNode->addNode(DetNode);
397  }
398 
399  // Be careful as a previous digitizer may have been registered for this detector
400  string RawTowerNodeName = "TOWER_" + m_RawTowerNodePrefix + "_" + m_Detector;
401  m_RawTowers = findNode::getClass<RawTowerContainer>(DetNode, RawTowerNodeName);
402  if (!m_RawTowers)
403  {
404  m_RawTowers = new RawTowerContainer(m_SimTowers->getCalorimeterID());
405  PHIODataNode<PHObject> *towerNode = new PHIODataNode<PHObject>(m_RawTowers,
406  RawTowerNodeName, "PHObject");
407  DetNode->addNode(towerNode);
408  }
409 
410  return;
