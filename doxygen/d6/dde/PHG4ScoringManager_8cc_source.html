128 int PHG4ScoringManager::process_event(PHCompositeNode *topNode)
129 {
130  Fun4AllHistoManager *hm = getHistoManager();
131  assert(hm);
132 
133  TH1D *h_norm = dynamic_cast<TH1D *>(hm->getHisto("hNormalization"));
134  assert(h_norm);
135 
136  h_norm->Fill("Event count", 1);
137 
138  PHHepMCGenEventMap *geneventmap = findNode::getClass<PHHepMCGenEventMap>(topNode, "PHHepMCGenEventMap");
139  if (!geneventmap)
140  {
141  static bool once = true;
142  if (once)
143  {
144  once = false;
145  cout << "PHG4ScoringManager::process_event - - missing node PHHepMCGenEventMap. Skipping HepMC stat." << std::endl;
146  }
147  }
148  else
149  {
150  h_norm->Fill("Collision count", geneventmap->size());
151 
152  TH1D *hVertexZ = dynamic_cast<TH1D *>(hm->getHisto("hVertexZ"));
153  assert(hVertexZ);
154 
155  for (const auto genevntpair : geneventmap->get_map())
156  {
157  const PHHepMCGenEvent *genevnt = genevntpair.second;
158  assert(genevnt);
159 
160  hVertexZ->Fill(genevnt->get_collision_vertex().z());
161  }
162  }
163 
164  TH1D *hNChEta = dynamic_cast<TH1D *>(hm->getHisto("hNChEta"));
165  assert(hNChEta);
166 
167  PHG4InEvent *ineve = findNode::getClass<PHG4InEvent>(topNode, "PHG4INEVENT");
168  if (!ineve)
169  {
170  cout << "PHG4ScoringManager::process_event - Error - "
171  << "unable to find DST node "
172  << "PHG4INEVENT" << endl;
173  }
174  else
175  {
176  const auto primary_range = ineve->GetParticles();
177  for (auto particle_iter = primary_range.first;
178  particle_iter != primary_range.second;
179  ++particle_iter)
180  {
181  const PHG4Particle *p = particle_iter->second;
182  assert(p);
183  TParticlePDG *pdg_p = TDatabasePDG::Instance()->GetParticle(p->get_pid());
184  assert(pdg_p);
185  if (fabs(pdg_p->Charge()) > 0)
186  {
187  TVector3 pvec(p->get_px(), p->get_py(), p->get_pz());
188  if (pvec.Perp2() > 0)
189  {
190  assert(hNChEta);
191  hNChEta->Fill(pvec.PseudoRapidity());
192  }
193  }
194  } // if (_load_all_particle) else
195  }
196 
197  return Fun4AllReturnCodes::EVENT_OK;
198 }
