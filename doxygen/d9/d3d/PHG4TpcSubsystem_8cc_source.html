49 int PHG4TpcSubsystem::InitRunSubsystem(PHCompositeNode *topNode)
50 {
51  PHNodeIterator iter(topNode);
52  PHCompositeNode *dstNode = dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
53 
54  // create display settings before detector (detector adds its volumes to it)
55  m_DisplayAction = new PHG4TpcDisplayAction(Name());
56  // create detector
57  detector_ = new PHG4TpcDetector(this, topNode, GetParams(), Name());
58  detector_->SuperDetector(SuperDetector());
59  detector_->OverlapCheck(CheckOverlap());
60  set<string> nodes;
61  if (GetParams()->get_int_param("active"))
62  {
63  PHNodeIterator dstIter(dstNode);
64  PHCompositeNode *DetNode = dynamic_cast<PHCompositeNode *>(dstIter.findFirst("PHCompositeNode", SuperDetector()));
65  if (!DetNode)
66  {
67  DetNode = new PHCompositeNode(SuperDetector());
68  dstNode->addNode(DetNode);
69  }
70 
71  ostringstream nodename;
72  if (SuperDetector() != "NONE")
73  {
74  nodename << "G4HIT_" << SuperDetector();
75  }
76  else
77  {
78  nodename << "G4HIT_" << Name();
79  }
80  nodes.insert(nodename.str());
81  if (GetParams()->get_int_param("absorberactive"))
82  {
83  nodename.str("");
84  if (SuperDetector() != "NONE")
85  {
86  nodename << "G4HIT_ABSORBER_" << SuperDetector();
87  }
88  else
89  {
90  nodename << "G4HIT_ABSORBER_" << Name();
91  }
92  nodes.insert(nodename.str());
93  }
94  BOOST_FOREACH (string node, nodes)
95  {
96  PHG4HitContainer *g4_hits = findNode::getClass<PHG4HitContainer>(topNode, node.c_str());
97  if (!g4_hits)
98  {
99  g4_hits = new PHG4HitContainer(node);
100  DetNode->addNode(new PHIODataNode<PHObject>(g4_hits, node.c_str(), "PHObject"));
101  }
102  }
103 
104  // create stepping action
105  steppingAction_ = new PHG4TpcSteppingAction(detector_, GetParams());
106  }
107  else
108  {
109  // if this is a black hole it does not have to be active
110  if (GetParams()->get_int_param("blackhole"))
111  {
112  steppingAction_ = new PHG4TpcSteppingAction(detector_, GetParams());
113  }
114  }
115  return 0;
116 }

200  // Thomas K Hemmick <Thomas.Hemmick@stonybrook.edu>
201  // The total thickness along Zed would be 5.6 millimeters (+/- 2.8 mm around Zed=0).
202  // The outer surfaces would have 0.005 inches (125 um) FR4 coated with a negligible thickness of Al. (revised to Au as below)
203  // The interior would be some stiffener of either honeycomb or rohacell. The range of radiation lengths for this material are:

206  // Close cell honeycomb: 635 cm (0.064 g/cm^3 density)

208  set_default_string_param("window_core_material", "ROHACELL_FOAM_51");
209  set_default_double_param("window_thickness", 0.56); // overall thickness
210  //I just checked with PC manufacturers and we can get 8.9 micron thick copper in reasonably large sheets.
211  // At normal incidence, 8.9 microns is 0.06% of a radiation length.
212  set_default_string_param("window_surface1_material", "G4_Cu");
213  set_default_double_param("window_surface1_thickness", 8.9e-4); // 8.9 um outter shell thickness be default
214  // The FR4 should be either 5 or 10 mils thick. 10 mils is 254 microns and 5 mils is 0.127 microns. I think either of these is mechanically fine...
