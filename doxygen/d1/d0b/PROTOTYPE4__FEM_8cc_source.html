268 double PROTOTYPE4_FEM::SignalShape_PowerLawExp(double *x, double *par)
269 {
270  double pedestal = par[4];
271  // + ((x[0] - 1.5 * par[1]) > 0) * par[5]; // quick
272  // fix on exting tails on the signal function
273  if (x[0] < par[1])
274  return pedestal;
275  // double signal =
276  // (-1)*par[0]*pow((x[0]-par[1]),par[2])*exp(-(x[0]-par[1])*par[3]);
277  double signal =
278  par[0] * pow((x[0] - par[1]), par[2]) * exp(-(x[0] - par[1]) * par[3]);
279  return pedestal + signal;

499 double PROTOTYPE4_FEM::SignalShape_PowerLawDoubleExp(double *x, double *par)
500 {
501  double pedestal = par[4];
502  // + ((x[0] - 1.5 * par[1]) > 0) * par[5]; // quick
503  // fix on exting tails on the signal function
504  if (x[0] < par[1])
505  return pedestal;
506  // double signal =
507  // (-1)*par[0]*pow((x[0]-par[1]),par[2])*exp(-(x[0]-par[1])*par[3]);
508  // peak / pow(fits.GetParameter(2) / fits.GetParameter(3),
509  // fits.GetParameter(2)) * exp(fits.GetParameter(2)) = fits.GetParameter(0);
510  // // exact peak height is (p0*Power(p2/p3,p2))/Power(E,p2)
511  // fits.GetParameter(2) / peak_shift = fits.GetParameter(3); // signal peak
512  // time
513 
514  double signal = //
515  par[0] //
516  * pow((x[0] - par[1]), par[2]) //
517  * (((1. - par[5]) / pow(par[3], par[2]) * exp(par[2])) *
518  exp(-(x[0] - par[1]) * (par[2] / par[3])) //
519  + (par[5] / pow(par[6], par[2]) * exp(par[2])) *
520  exp(-(x[0] - par[1]) * (par[2] / par[6])) //
521  );
522  return pedestal + signal;
