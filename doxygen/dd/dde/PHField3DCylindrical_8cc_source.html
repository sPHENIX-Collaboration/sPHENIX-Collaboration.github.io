218 void PHField3DCylindrical::GetFieldValue(const double point[4], double *Bfield) const
219 {
220  if (Verbosity() > 2)
221  cout << "\nPHField3DCylindrical::GetFieldValue" << endl;
222  double x = point[0];
223  double y = point[1];
224  double z = point[2];
225  double r = sqrt(x * x + y * y);
226  double phi;
227  if (x == 0)
228  {
229  phi = atan2(y, 0.00000000001);
230  }
231  else
232  {
233  phi = atan2(y, x);
234  }
235  if (phi < 0) phi += 2 * M_PI; // normalize phi to be over the range [0,2*pi]
236 
237  // Check that the point is within the defined z region (check r in a second)
238  if ((z >= minz_) && (z <= maxz_))
239  {
240  double BFieldCyl[3];
241  double cylpoint[4] = {z, r, phi, 0};
242 
243  // take <z,r,phi> location and return a vector of <Bz, Br, Bphi>
244  GetFieldCyl(cylpoint, BFieldCyl);
245 
246  // X direction of B-field ( Bx = Br*cos(phi) - Bphi*sin(phi)
247  Bfield[0] = cos(phi) * BFieldCyl[1] - sin(phi) * BFieldCyl[2]; // unit vector transformations
248 
249  // Y direction of B-field ( By = Br*sin(phi) + Bphi*cos(phi)
250  Bfield[1] = sin(phi) * BFieldCyl[1] + cos(phi) * BFieldCyl[2];
251 
252  // Z direction of B-field
253  Bfield[2] = BFieldCyl[0];
254  }
255  else
256  { // x,y,z is outside of z range of the field map
257 
258  Bfield[0] = 0.0;
259  Bfield[1] = 0.0;
260  Bfield[2] = 0.0;
261  if (Verbosity() > 2)
262  cout << "!!!!!!!!!! Field point not in defined region (outside of z bounds)" << endl;
263  }
264 
265  if (Verbosity() > 2)
266  {
267  cout << "END PHField3DCylindrical::GetFieldValue\n"
268  << " ---> {Bx, By, Bz} : "
269  << "< " << Bfield[0] << ", " << Bfield[1] << ", " << Bfield[2] << " >" << endl;
270  }
271 
272  return;
273 }

430 // a binary search algorithm that puts the location that "key" would be, into index...
431 // it returns true if key was found, and false if not.
432 bool PHField3DCylindrical::bin_search(const vector<float> &vec, unsigned start, unsigned end, const float &key, unsigned &index) const
433 {
434  // Termination condition: start index greater than end index
435  if (start > end)
436  {
437  index = start;
438  return false;
439  }
440 
441  // Find the middle element of the vector and use that for splitting
442  // the array into two pieces.
443  unsigned middle = start + ((end - start) / 2);
444 
445  if (vec[middle] == key)
446  {
447  index = middle;
448  return true;
449  }
450  else if (vec[middle] > key)
451  {
452  return bin_search(vec, start, middle - 1, key, index);
453  }
454  return bin_search(vec, middle + 1, end, key, index);
455 }
