70  gSystem->Load("libg4detectors.so");

73  const double z_ring1 = (2025 + 2050) / 2. / 10.;
74  const double z_ring2 = (2150 + 2175) / 2. / 10.;
75  const double dz = 25. / 10.;

83  PHG4CylinderSubsystem *cyl;
84 
85  for (int i = 0; i < 4; i++)
86  {
87  double innerradius = innerradius_e;
88  if ( z_rings[i] > 0 )
89  innerradius = innerradius_h;
90 
91  cyl = new PHG4CylinderSubsystem("HCALIN_SPT_N1", i);
92  cyl->set_double_param("place_z",z_rings[i]);
93  cyl->SuperDetector("HCALIN_SPT");
94  cyl->set_double_param("radius",innerradius);
95  cyl->set_int_param("lengthviarapidity",0);
96  cyl->set_double_param("length",dz);
97  cyl->set_string_param("material","SS310");
98  cyl->set_double_param("thickness",maxradius - innerradius);
99  if (absorberactive)
