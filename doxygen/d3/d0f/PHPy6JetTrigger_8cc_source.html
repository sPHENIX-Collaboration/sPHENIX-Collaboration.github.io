38 bool PHPy6JetTrigger::Apply(const HepMC::GenEvent *evt)
39 {
40  // Loop over all particles in the event
41  int idx = 0;
42  std::vector<fastjet::PseudoJet> pseudojets;
43  for (HepMC::GenEvent::particle_const_iterator p = evt->particles_begin(); p != evt->particles_end(); ++p)
44  {
45  idx++;
46  if (((*p)->status() != 1) != 0) continue;
47 
48  // remove some particles (muons, taus, neutrinos)...
49  // 12 == nu_e
50  // 13 == muons
51  // 14 == nu_mu
52  // 15 == taus
53  // 16 == nu_tau
54  if ((abs(((*p)->pdg_id())) >= 12) && (abs(((*p)->pdg_id())) <= 16)) continue;
55 
56  // acceptance... _etamin,_etamax
57  if (((*p)->momentum().px() == 0.0) && ((*p)->momentum().py() == 0.0)) continue; // avoid pt=0
58  if ((((*p)->momentum().pseudoRapidity()) < m_theEtaLow) ||
59  (((*p)->momentum().pseudoRapidity()) > m_theEtaHigh)) continue;
60 
61  fastjet::PseudoJet pseudojet((*p)->momentum().px(),
62  (*p)->momentum().py(),
63  (*p)->momentum().pz(),
64  (*p)->momentum().e());
65  pseudojet.set_user_index(idx);
66  pseudojets.push_back(pseudojet);
67  }
68 
69  // Call FastJet
70 
71  fastjet::JetDefinition *jetdef = new fastjet::JetDefinition(fastjet::antikt_algorithm, m_R, fastjet::E_scheme, fastjet::Best);
72  fastjet::ClusterSequence jetFinder(pseudojets, *jetdef);
73  std::vector<fastjet::PseudoJet> fastjets = jetFinder.inclusive_jets();
74  delete jetdef;
75 
76  bool jetFound = false;
77  double max_pt = -1;
78  for (unsigned int ijet = 0; ijet < fastjets.size(); ++ijet)
79  {
80  const double pt = sqrt(pow(fastjets[ijet].px(), 2) + pow(fastjets[ijet].py(), 2));
81 
82  if (pt > max_pt) max_pt = pt;
83 
84  vector<fastjet::PseudoJet> constituents = fastjets[ijet].constituents();
85  const int nconst = constituents.size();
86 
87  if (pt > m_minPt && nconst >= m_nconst)
88  {
89  jetFound = true;
90  break;
91  }
92  }
93 
94  if (_verbosity > 2)
95  {
96  cout << "PHPy6JetTrigger::Apply - max_pt = " << max_pt << ", and jetFound = " << jetFound << endl;
97  }
98 
99  return jetFound;
100 }
