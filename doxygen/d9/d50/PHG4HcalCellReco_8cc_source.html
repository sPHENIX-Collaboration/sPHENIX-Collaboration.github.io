119 PHG4HcalCellReco::process_event(PHCompositeNode *topNode)
120 {
121  PHG4HitContainer *g4hit = findNode::getClass<PHG4HitContainer>(topNode, hitnodename.c_str());
122  if (!g4hit)
123  {
124  cout << "Could not locate g4 hit node " << hitnodename << endl;

131  exit(1);
132  }

134  PHG4HitContainer::ConstIterator hiter;

136  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; ++hiter)
137  {
138  if (hiter->second->get_t(0) > tmax) continue;
139  if (hiter->second->get_t(1) < tmin) continue;
140  short icolumn = hiter->second->get_scint_id();
141  int introw = (hiter->second->get_hit_id() >> PHG4HitDefs::hit_idbits);
142  if ( introw >= ROWDIM || introw < 0)
143  {
144  cout << "row " << introw
145  << " exceed array size: " << ROWDIM
146  << " adjust ROWDIM and recompile" << endl;
147  exit(1);
148  }
149  // after checking for size of introw so we do not run into
150  // overflow issues, put this into the short we want later
151  short irow = introw;
152  if (icolumn >= COLUMNDIM || icolumn < 0)
153  {
154  cout << "column: " << icolumn
155  << " exceed array size: " << COLUMNDIM
156  << " adjust COLUMNDIM and recompile" << endl;
157  exit(1);
158  }
159 
160 
161  if (!slatarray[irow][icolumn])
162  {
163  // hcal has no layers so far, I do not want to make an expensive
164  // call to the g4hits to find that out use 0 as layer number
165  PHG4CellDefs::keytype key = PHG4CellDefs::ScintillatorSlatBinning::genkey(0,icolumn,irow);
166  slatarray[irow][icolumn] = new PHG4Cellv1(key);
167  }
168  slatarray[irow][icolumn]->add_edep(hiter->second->get_edep());
169  slatarray[irow][icolumn]->add_eion(hiter->second->get_eion());
170  slatarray[irow][icolumn]->add_light_yield(hiter->second->get_light_yield());

172  slatarray[irow][icolumn]->add_shower_edep(hiter->second->get_shower_id(), hiter->second->get_edep());
173  } // end loop over g4hits
174  int nslathits = 0;
175  for (int irow = 0; irow<ROWDIM; irow++)
176  {
177  for (int icolumn = 0; icolumn<COLUMNDIM; icolumn++)
178  {
179  if (slatarray[irow][icolumn])
180  {
181  slats->AddCell(slatarray[irow][icolumn]);
182  slatarray[irow][icolumn] = nullptr;
183  nslathits++;
184  }
185  }
186  }
187  if (Verbosity() > 0)
188  {
189  cout << Name() << ": found " << nslathits << " slats with energy deposition" << endl;
190  }
191 
192  if (chkenergyconservation)
193  {
194  CheckEnergy(topNode);
195  }
196  return Fun4AllReturnCodes::EVENT_OK;
197 }

206 int
207 PHG4HcalCellReco::CheckEnergy(PHCompositeNode *topNode)
208 {
209  PHG4HitContainer *g4hit = findNode::getClass<PHG4HitContainer>(topNode, hitnodename.c_str());
210  PHG4CellContainer *slats = findNode::getClass<PHG4CellContainer>(topNode, cellnodename);
211  double sum_energy_g4hit = 0.;
212  double sum_energy_cells = 0.;
213  PHG4HitContainer::ConstRange hit_begin_end = g4hit->getHits();
214  PHG4HitContainer::ConstIterator hiter;
215  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; ++hiter)
216  {
217  sum_energy_g4hit += hiter->second->get_edep();
218  }
219  PHG4CellContainer::ConstRange cell_begin_end = slats->getCells();
220  PHG4CellContainer::ConstIterator citer;
221  for (citer = cell_begin_end.first; citer != cell_begin_end.second; ++citer)
222  {
223  sum_energy_cells += citer->second->get_edep();
224 
225  }
226  // the fractional eloss for particles traversing eta bins leads to minute rounding errors
227  if (fabs(sum_energy_cells - sum_energy_g4hit) / sum_energy_g4hit > 1e-6)
228  {
229  cout << "hint: timing cuts tmin/tmax will do this to you" << endl;
230  cout << "energy mismatch between cells: " << sum_energy_cells
231  << " and hits: " << sum_energy_g4hit
232  << " diff sum(cells) - sum(hits): " << sum_energy_cells - sum_energy_g4hit
233  << endl;
234  return -1;
235  }
236  else
237  {
238  if (Verbosity() > 0)
239  {
240  cout << Name() << ": total energy for this event: " << sum_energy_g4hit << " GeV" << endl;
241  }
242  }
243  return 0;
244 }
