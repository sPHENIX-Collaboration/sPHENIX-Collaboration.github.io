159 {
160  // attempt to retrieve a valid event from inputs
161  while (true)

218  }

275 int Fun4AllHepMCInputManager::PushBackEvents(const int i)
276 {
277  // PushBackEvents is supposedly pushing events back on the stack which works
278  // easily with root trees (just grab a different entry) but hard in these HepMC ASCII files.
279  // A special case is when the synchronization fails and we need to only push back a single
280  // event. In this case we save the evt pointer as save_evt which is used in the run method
281  // instead of getting the next event.
282  if (i > 0)
283  {
284  if (i == 1 && evt) // check on evt pointer makes sure it is not done from the cmd line
285  {
286  if (Verbosity() > 3)
287  {
288  cout << Name() << ": pushing back evt no " << evt->event_number() << endl;
289  }
290  save_evt = evt;
291  return 0;
292  }
293  cout << PHWHERE << Name()
294  << " Fun4AllHepMCInputManager cannot pop back events into file"
295  << endl;
296  return -1;
297  }
298  if (!IsOpen())
299  {
300  cout << PHWHERE << Name()
301  << " no file opened yet" << endl;
302  return -1;
303  }
304  // Skipping events is implemented as
305  // pushing a negative number of events on the stack, so in order to implement
306  // the skipping of events we read -i events.
307  int nevents = -i; // negative number of events to push back -> skip num events
308  int errorflag = 0;
309  while (nevents > 0 && !errorflag)
310  {
311  evt = ascii_in->read_next_event();
312  if (!evt)
313  {
314  cout << "Error after skipping " << i - nevents << endl;
315  cout << "error type: " << ascii_in->error_type()
316  << ", rdstate: " << ascii_in->rdstate() << endl;
317  errorflag = -1;
318  fileclose();
319  }
320  else
321  {
322  if (Verbosity() > 3)
323  {
324  cout << "Skipping evt no: " << evt->event_number() << endl;
325  }
326  }
327  delete evt;
328  nevents--;
329  }
330  return errorflag;
331 }

333 HepMC::GenEvent *
334 Fun4AllHepMCInputManager::ConvertFromOscar()
335 {
336  delete evt;
337  evt = nullptr;
338  if (theOscarFile.eof()) // if the file is exhausted bail out during this next read
339  {
340  return evt;
341  }
342 
343  //use PHENIX unit
344  evt = new HepMC::GenEvent(HepMC::Units::GEV, HepMC::Units::CM);
345 
346  if (Verbosity() > 1) cout << "Reading Oscar Event " << events_total << endl;
347  //Grab New Event From Oscar
348  string theLine;
349  vector<vector<double> > theEventVec;
350  vector<HepMC::FourVector> theVtxVec;
351  while (getline(theOscarFile, theLine))
352  {
353  if (theLine.find("#") == 0) continue;
354  vector<double> theInfo; //format: N,pid,px,py,pz,E,mass,xvtx,yvtx,zvtx,?
355  double number = NAN;
356  for (istringstream numbers_iss(theLine); numbers_iss >> number;)
357  {
358  theInfo.push_back(number);
359  }
360 
361  if (theInfo.size() == 2 && theInfo[0] == 0 && theInfo[1] == 0)
362  {
363  break;
364  }
365  else if (theInfo.size() == 2 && theInfo[0] == 0 && theInfo[1] > 0)
366  {
367  continue;
368  }
369  else
370  {
371  theEventVec.push_back(theInfo);
372  HepMC::FourVector vert(theInfo[8] * toMM, theInfo[9] * toMM, theInfo[10] * toMM, theInfo[11]);
373  theVtxVec.push_back(vert);
374  }
375 
376  } //while(getline)
377 
378  //Set Event Number
379  evt->set_event_number(events_total);
380 
381  //Loop Over One Event, Fill HepMC
382  for (unsigned int i = 0; i < theEventVec.size(); i++)
383  {
384  //int N = (int)theEventVec[i][0];
385  int pid = (int) theEventVec[i][1];
386  double px = theEventVec[i][3];
387  double py = theEventVec[i][4];
388  double pz = theEventVec[i][5];
389  double E = theEventVec[i][6];
390  double m = theEventVec[i][7];
391  int status = 1; //oscar only writes final state particles
392 
393  HepMC::GenVertex *v = new HepMC::GenVertex(theVtxVec[i]);
394  evt->add_vertex(v);
395 
396  HepMC::GenParticle *p = new HepMC::GenParticle(HepMC::FourVector(px, py, pz, E), pid, status);
397  p->setGeneratedMass(m);
398  p->suggest_barcode(i + 1);
399  v->add_particle_out(p);
400  }
401  if (Verbosity() > 3)
402  {
403  evt->print();
404  }
405  return evt;
406 }
