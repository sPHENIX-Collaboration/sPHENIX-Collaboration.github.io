150 void PHG4InttDigitizer::CalculateLadderCellADCScale(PHCompositeNode *topNode)
151 {
152  // FPHX 3-bit ADC, thresholds are set in "set_fphx_adc_scale".
153 
154  //PHG4CellContainer *cells = findNode::getClass<PHG4CellContainer>(topNode, "G4CELL_INTT");
155  PHG4CylinderGeomContainer *geom_container = findNode::getClass<PHG4CylinderGeomContainer>(topNode, "CYLINDERGEOM_INTT");
156 
157  //if (!geom_container || !cells) return;
158  if (!geom_container) return;
159 
160  PHG4CylinderGeomContainer::ConstRange layerrange = geom_container->get_begin_end();
161  for (PHG4CylinderGeomContainer::ConstIterator layeriter = layerrange.first;
162  layeriter != layerrange.second;
163  ++layeriter)
164  {
165  int layer = layeriter->second->get_layer();
166  if (_max_fphx_adc.find(layer) == _max_fphx_adc.end())
167  {
168  cout << "Error: _max_fphx_adc is not available." << endl;
169  gSystem->Exit(1);
170  }
171  float thickness = (layeriter->second)->get_thickness(); // cm
172  float mip_e = 0.003876 * thickness; // GeV
173  _energy_scale.insert(std::make_pair(layer, mip_e));
174  }
175 
176  return;
177 }

179 void PHG4InttDigitizer::DigitizeLadderCells(PHCompositeNode *topNode)
180 {
181  //---------------------------
182  // Get common Nodes
183  //---------------------------
184  const InttDeadMap *deadmap = findNode::getClass<InttDeadMap>(topNode, "DEADMAP_INTT");
185  if (Verbosity() >= VERBOSITY_MORE)
186  {
187  if (deadmap)
188  {
189  cout << "PHG4InttDigitizer::DigitizeLadderCells - Use deadmap ";
190  deadmap->identify();
191  }
192  else
193  {
194  cout << "PHG4InttDigitizer::DigitizeLadderCells - Can not find deadmap, all channels enabled " << endl;
195  }
196  }
197 
198  // Get the TrkrHitSetContainer node
199  TrkrHitSetContainer *trkrhitsetcontainer = findNode::getClass<TrkrHitSetContainer>(topNode, "TRKR_HITSET");
200  if(!trkrhitsetcontainer)
201  {
202  cout << "Could not locate TRKR_HITSET node, quit! " << endl;
203  exit(1);
204  }
205 
206  //-------------
207  // Digitization
208  //-------------
209 
210  // We want all hitsets for the Intt
211  TrkrHitSetContainer::ConstRange hitset_range = trkrhitsetcontainer->getHitSets(TrkrDefs::TrkrId::inttId);
212  for (TrkrHitSetContainer::ConstIterator hitset_iter = hitset_range.first;
213  hitset_iter != hitset_range.second;
214  ++hitset_iter)
215  {
216  // we have an itrator to one TrkrHitSet for the intt from the trkrHitSetContainer
217  // get the hitset key so we can find the layer
218  TrkrDefs::hitsetkey hitsetkey = hitset_iter->first;
219  const int layer = TrkrDefs::getLayer(hitsetkey);
220  const int ladder_phi = InttDefs::getLadderPhiId(hitsetkey);
221  const int ladder_z = InttDefs::getLadderZId(hitsetkey);
222 
223  if(Verbosity() > 1)
224  cout << "PHG4InttDigitizer: found hitset with key: " << hitsetkey << " in layer " << layer << endl;
225 
226  // get all of the hits from this hitset
227  TrkrHitSet *hitset = hitset_iter->second;
228  TrkrHitSet::ConstRange hit_range = hitset->getHits();
229  for(TrkrHitSet::ConstIterator hit_iter = hit_range.first;
230  hit_iter != hit_range.second;
231  ++hit_iter)
232  {
233  ++m_nCells;
234 
235  TrkrHit *hit = (InttHit*) hit_iter->second;
236  TrkrDefs::hitkey hitkey = hit_iter->first;
237  int strip_col = InttDefs::getCol(hitkey); // strip z index
238  int strip_row = InttDefs::getRow(hitkey); // strip phi index
239 
240  // Apply deadmap here if desired
241  if (deadmap)
242  {
243  if (deadmap->isDeadChannelIntt(
244  layer,
245  ladder_phi,
246  ladder_z,
247  strip_col,
248  strip_row
249  ))
250  {
251  ++m_nDeadCells;
252  if (Verbosity() >= VERBOSITY_MORE)
253  {
254  cout << "PHG4InttDigitizer::DigitizeLadderCells - dead strip at layer " << layer << ": ";
255  hit->identify();
256  }
257  continue;
258  }
259  } // if (deadmap)
260 
261  if (_energy_scale.count(layer) > 1)
262  assert(!"Error: _energy_scale has two or more keys.");
263 
264  const float mip_e = _energy_scale[layer];
265 
266  std::vector<std::pair<double, double> > vadcrange = _max_fphx_adc[layer];
267 
268  int adc = -1;
269  for (unsigned int irange = 0; irange < vadcrange.size(); ++irange)
270  if (hit->getEnergy() >= vadcrange[irange].first * (double) mip_e && hit->getEnergy() < vadcrange[irange].second * (double) mip_e)
271  adc = (int) irange;
272 
273  if(adc == -1)
274  // how do we specify underflow or overflow?
275  adc = 0;
276 
277  hit->setAdc(adc);
278 
279  if(Verbosity() > 2)
280  cout << "PHG4InttDigitizer: found hit with layer " << layer << " ladder_z " << ladder_z << " ladder_phi " << ladder_phi
281  << " strip_col " << strip_col << " strip_row " << strip_row << " adc " << adc << endl;
282 
283  } // end loop over hits in this hitset
284  } // end loop over hitsets
285 
286  return;
287 }
