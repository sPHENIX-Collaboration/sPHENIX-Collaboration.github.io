150 int PHRaveVertexing::process_event(PHCompositeNode* topNode)
151 {
152  _event++;
153 
154  if (Verbosity() > 1)
155  std::cout << PHWHERE << "Events processed: " << _event << std::endl;
156 
157  GetNodes(topNode);
158 
160  GenFitTrackMap gf_track_map;
161  vector<genfit::Track*> gf_tracks;
162  if (Verbosity() > 1) _t_translate->restart();
163  for (SvtxTrackMap::Iter iter = _trackmap->begin(); iter != _trackmap->end();
164  ++iter)
165  {
166  SvtxTrack* svtx_track = iter->second;
167  if (!svtx_track)
168  continue;
169 
170  if (!(svtx_track->get_ndf() >= _vertex_min_ndf))
171  continue;
172 
173  //auto genfit_track = shared_ptr<genfit::Track> (TranslateSvtxToGenFitTrack(svtx_track));
174  auto genfit_track = TranslateSvtxToGenFitTrack(svtx_track);
175  if (!genfit_track)
176  continue;
177  gf_track_map.insert({genfit_track, iter->first});
178  gf_tracks.push_back(const_cast<genfit::Track*>(genfit_track));
179  }
180  if (Verbosity() > 1) _t_translate->stop();
181 
182  if (Verbosity() > 1) _t_rave->restart();
183  vector<genfit::GFRaveVertex*> rave_vertices;
184  if (gf_tracks.size() >= 2)
185  {
186  try
187  {
188  _vertex_finder->findVertices(&rave_vertices, gf_tracks);
189  }
190  catch (...)
191  {
192  if (Verbosity() > 1)
193  std::cout << PHWHERE << "GFRaveVertexFactory::findVertices failed!";
194  }
195  }
196  if (Verbosity() > 1) _t_rave->stop();
197  FillSvtxVertexMap(rave_vertices, gf_track_map);
198 
199  for (auto iter : gf_track_map) delete iter.first;
200 
201  if (Verbosity() > 1)
202  {
203  _vertexmap->identify();
204 
205  std::cout << "=============== Timers: ===============" << std::endl;
206  std::cout << "Event: " << _event << std::endl;
207  std::cout << "Translate: " << _t_translate->get_accumulated_time() / 1000. << " sec" << std::endl;
208  std::cout << "RAVE: " << _t_rave->get_accumulated_time() / 1000. << " sec" << std::endl;
209  std::cout << "=======================================" << std::endl;
210  }
211 
212  return Fun4AllReturnCodes::EVENT_OK;
213 }

232 int PHRaveVertexing::CreateNodes(PHCompositeNode* topNode)

245 
246  // Create the SVTX node

248  "PHCompositeNode", "SVTX"));
249  if (!tb_node)
250  {
251  tb_node = new PHCompositeNode("SVTX");
252  dstNode->addNode(tb_node);
253  if (Verbosity() > 0)
254  cout << "SVTX node added" << endl;
255  }

257  if (!(_over_write_svtxvertexmap))
258  {
259  _vertexmap_refit = new SvtxVertexMap_v1;
260  PHIODataNode<PHObject>* vertexes_node = new PHIODataNode<PHObject>(
261  _vertexmap_refit, _svtxvertexmaprefit_node_name.c_str(), "PHObject");
262  tb_node->addNode(vertexes_node);
263  if (Verbosity() > 0)
264  cout << "Svtx/SvtxVertexMapRefit node added" << endl;
265  }
266  else if (!findNode::getClass<SvtxVertexMap>(topNode, "SvtxVertexMap"))
267  {
268  _vertexmap = new SvtxVertexMap_v1;
269  PHIODataNode<PHObject>* vertexes_node = new PHIODataNode<PHObject>(
270  _vertexmap, "SvtxVertexMap", "PHObject");
271  tb_node->addNode(vertexes_node);
272  if (Verbosity() > 0)
273  cout << "Svtx/SvtxVertexMap node added" << endl;
274  }
275 
276  return Fun4AllReturnCodes::EVENT_OK;

333 bool PHRaveVertexing::FillSvtxVertexMap(
334  const std::vector<genfit::GFRaveVertex*>& rave_vertices,

342  // for (unsigned int ivtx = 0; ivtx < rave_vertices.size(); ++ivtx) {
343  // genfit::GFRaveVertex* rave_vtx = rave_vertices[ivtx];
344 
345  for (genfit::GFRaveVertex* rave_vtx : rave_vertices)
346  {
347  if (!rave_vtx)

349  cerr << PHWHERE << endl;
350  return false;
351  }
352 
353  std::shared_ptr<SvtxVertex> svtx_vtx(new SvtxVertex_v1());

355  svtx_vtx->set_chisq(rave_vtx->getChi2());
356  svtx_vtx->set_ndof(rave_vtx->getNdf());
357  svtx_vtx->set_position(0, rave_vtx->getPos().X());
358  svtx_vtx->set_position(1, rave_vtx->getPos().Y());
359  svtx_vtx->set_position(2, rave_vtx->getPos().Z());
360 
361  for (int i = 0; i < 3; i++)
362  for (int j = 0; j < 3; j++)
363  svtx_vtx->set_error(i, j, rave_vtx->getCov()[i][j]);
364 
365  for (unsigned int i = 0; i < rave_vtx->getNTracks(); i++)
366  {
367  //TODO improve speed
368  const genfit::Track* rave_track =
369  rave_vtx->getParameters(i)->getTrack();
370  // for(auto iter : gf_track_map) {

372  // svtx_vtx->insert_track(iter.first);
373  // }
374  auto iter = gf_track_map.find(rave_track);
375  if (iter != gf_track_map.end())
376  svtx_vtx->insert_track(iter->second);
377  }
378 
379  if (_over_write_svtxvertexmap)
380  {
381  if (_vertexmap)

387  LogError("!_vertexmap");
388  }
389  }
390  else
391  {
392  if (_vertexmap_refit)
393  {
394  _vertexmap_refit->insert_clone(svtx_vtx.get());
395  }
396  else
397  {
398  LogError("!_vertexmap_refit");
399  }
400  }
401 
402 #ifdef _DEBUG_
403  cout << __LINE__ << endl;
404  svtx_vtx->identify();
405 #endif
406 
407  } //loop over RAVE vertices
408 
409  return true;

412 genfit::Track* PHRaveVertexing::TranslateSvtxToGenFitTrack(SvtxTrack* svtx_track)
413 {
414  try
415  {
416  // The first state is extracted to PCA, second one is the one with measurement
417  SvtxTrackState* svtx_state(nullptr);
418  //SvtxTrackState* svtx_state = (svtx_track->begin_states())->second;
419 
420  if (svtx_track->begin_states() == svtx_track->end_states())
421  {
422  LogDebug("TranslateSvtxToGenFitTrack no state in track!");
423  return nullptr;
424  }
425  else if (++(svtx_track->begin_states()) == svtx_track->end_states())
426  {
427  // only one state in track
428  svtx_state = (svtx_track->begin_states())->second;
429  }
430  else
431  {
432  // multiple state in track
433  // The first state is extracted to PCA, second one is the one with measurement
434  svtx_state = (++(svtx_track->begin_states()))->second;
435  }
436 
437  if (!svtx_state)
438  {
439  LogDebug("TranslateSvtxToGenFitTrack invalid state found on track!");
440  return nullptr;
441  }
442 
443  TVector3 pos(svtx_state->get_x(), svtx_state->get_y(), svtx_state->get_z());
444  TVector3 mom(svtx_state->get_px(), svtx_state->get_py(), svtx_state->get_pz());
445  TMatrixDSym cov(6);
446  for (int i = 0; i < 6; ++i)
447  {
448  for (int j = 0; j < 6; ++j)
449  {
450  cov[i][j] = svtx_state->get_error(i, j);
451  }
452  }
453 
454 #ifdef _DEBUG_
455  {
456  cout << "DEBUG" << __LINE__ << endl;
457  cout << "path length: " << svtx_state->get_pathlength() << endl;
458  cout << "radius: " << pos.Perp() << endl;
459  cout << "DEBUG: " << __LINE__ << endl;
460  for (int i = 0; i < 6; ++i)
461  {
462  for (int j = 0; j < 6; ++j)
463  {
464  cout << svtx_state->get_error(i, j) << "\t";
465  }
466  cout << endl;
467  }
468 
469  cov.Print();
470  }
471 
472 #endif
473 
474  genfit::AbsTrackRep* rep = new genfit::RKTrackRep(_primary_pid_guess);
475  genfit::Track* genfit_track = new genfit::Track(rep, TVector3(0, 0, 0), TVector3(0, 0, 0));
476 
477  genfit::FitStatus* fs = new genfit::FitStatus();
478  fs->setCharge(svtx_track->get_charge());
479  fs->setChi2(svtx_track->get_chisq());
480  fs->setNdf(svtx_track->get_ndf());
481  fs->setIsFitted(true);
482  fs->setIsFitConvergedFully(true);
483 
484  genfit_track->setFitStatus(fs, rep);
485 
486  genfit::TrackPoint* tp = new genfit::TrackPoint(genfit_track);
487 
488  genfit::KalmanFitterInfo* fi = new genfit::KalmanFitterInfo(tp, rep);
489  tp->setFitterInfo(fi);
490 
491  genfit::MeasuredStateOnPlane* ms = new genfit::MeasuredStateOnPlane(rep);
492  ms->setPosMomCov(pos, mom, cov);
493 #ifdef _DEBUG_
494  {
495  cout << "DEBUG: " << __LINE__ << endl;
496  ms->Print();
497  cout << "Orig: " << __LINE__ << endl;
498  cov.Print();
499  cout << "Translate: " << __LINE__ << endl;
500  ms->get6DCov().Print();
501  }
502 #endif
503  genfit::KalmanFittedStateOnPlane* kfs = new genfit::KalmanFittedStateOnPlane(*ms, 1., 1.);
504 
505  //< Acording to the special order of using the stored states
506  fi->setForwardUpdate(kfs);
507 
508  genfit_track->insertPoint(tp);
509 
510 #ifdef _DEBUG_
511 // {
512 // cout << "DEBUG" << __LINE__ << endl;
513 // TVector3 pos, mom;
514 // TMatrixDSym cov;
515 // genfit_track->getFittedState().getPosMomCov(pos, mom, cov);
516 // pos.Print();
517 // mom.Print();
518 // cov.Print();
519 // }
520 #endif
521 
522  return genfit_track;
523  }
524  catch (...)
525  {
526  LogDebug("TranslateSvtxToGenFitTrack failed!");
527  }
528 
529  return nullptr;
