55 bool PHSartreParticleTrigger::Apply(Event *event)
56 {
57  if (Verbosity() > 2)
58  {
59  cout << "PHSartreParticleTrigger::Apply - event size: "
60  << event->particles.size() << endl;
61  }
62 
63  // Loop over all particles in the event
64  for (unsigned int i = 0; i < event->particles.size(); ++i)
65  {
66  //cout << i << " " << event->particles[i].pdgId << endl;
67 
68  if ((i < 7) && (i != 4)) continue; // only the VM, daughters OR the nuclear remnants
69 
70  const Particle &particle = event->particles[i];
71 
72  // loop over all the trigger particle criteria
73  for (int j = 0; j < int(_theParticles.size()); j++)
74  {
75  if ((particle.pdgId == _theParticles[j]) &&
76  ((particle.status == 1) || (i == 4)))
77  { //only stable particles OR the vector meson
78 
79  if (_doBothEtaCut && (particle.p.Eta() < _theEtaLow ||
80  particle.p.Eta() > _theEtaHigh)) continue;
81  if (_doEtaLowCut && particle.p.Eta() < _theEtaLow) continue;
82  if (_doEtaHighCut && particle.p.Eta() > _theEtaHigh) continue;
83 
84  if (_doBothAbsEtaCut && (abs(particle.p.Eta()) < _theEtaLow ||
85  abs(particle.p.Eta()) > _theEtaHigh)) continue;
86  if (_doAbsEtaLowCut && abs(particle.p.Eta()) < _theEtaLow) continue;
87  if (_doAbsEtaHighCut && abs(particle.p.Eta()) > _theEtaHigh) continue;
88 
89  if (_doBothPtCut && (particle.p.Pt() < _thePtLow ||
90  particle.p.Pt() > _thePtHigh)) continue;
91  if (_doPtHighCut && particle.p.Pt() > _thePtHigh) continue;
92  if (_doPtLowCut && particle.p.Pt() < _thePtLow) continue;
93 
94  if (_doBothPCut && (particle.p.P() < _thePLow ||
95  particle.p.P() > _thePHigh)) continue;
96  if (_doPHighCut && particle.p.P() > _thePHigh) continue;
97  if (_doPLowCut && particle.p.P() < _thePLow) continue;
98 
99  if (_doBothPzCut && (particle.p.Pz() < _thePzLow ||
100  particle.p.Pz() > _thePzHigh)) continue;
101  if (_doPzHighCut && particle.p.Pz() > _thePzHigh) continue;
102  if (_doPzLowCut && particle.p.Pz() < _thePzLow) continue;
103 
104  //If we made it here, success!
105  return true;
106 
107  } //if _theParticles
108  } //_theParticles for loop
109  }
110 
111  return false;
