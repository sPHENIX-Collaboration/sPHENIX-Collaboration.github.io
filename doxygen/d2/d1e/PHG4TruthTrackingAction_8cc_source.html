51  trackid = m_TruthInfoList->maxtrkindex() + 1;

54  // add the user id to the geant4 user info
55  PHG4TrackUserInfo::SetUserTrackId(const_cast<G4Track*>(track), trackid);

59  int pdgid = def->GetPDGEncoding();
60  double m = def->GetPDGMass();
61  double ke = track->GetVertexKineticEnergy();
62  double ptot = sqrt(ke * ke + 2.0 * m * ke);
63  G4ThreeVector pdir = track->GetVertexMomentumDirection();
64  pdir *= ptot;
65  PHG4Particle* ti = nullptr;
66  // create a new particle -----------------------------------------------------
67  if (def->IsGeneralIon()) // for ions save a and z in v3 of phg4particle
68  {
69  ti = new PHG4Particlev3();
70  ti->set_A(def->GetAtomicMass());
71  ti->set_Z(def->GetAtomicNumber());
72  }
73  else
74  {
75  ti = new PHG4Particlev2;
76  }
77  ti->set_px(pdir[0] / GeV);
78  ti->set_py(pdir[1] / GeV);
79  ti->set_pz(pdir[2] / GeV);
80  ti->set_track_id(trackid);
81 
82  ti->set_parent_id(track->GetParentID());
83  if (PHG4TrackUserInfoV1* p = dynamic_cast<PHG4TrackUserInfoV1*>(track->GetUserInformation()))
84  {
85  ti->set_parent_id(p->GetUserParentId());
86  }

89  if (PHG4TrackUserInfoV1* p = dynamic_cast<PHG4TrackUserInfoV1*>(track->GetUserInformation()))

92  {

108  // primary track - propagate the barcode information
109  PHG4UserPrimaryParticleInformation* userdata = static_cast<PHG4UserPrimaryParticleInformation*>(track->GetDynamicParticle()->GetPrimaryParticle()->GetUserInformation());
110  if (userdata) ti->set_barcode(userdata->get_user_barcode());
111  }

114  G4ThreeVector v = track->GetVertexPosition();
115  map<G4ThreeVector, int>::const_iterator viter = m_VertexMap.find(v);

117  if (viter != m_VertexMap.end())

173  if (userdata)
