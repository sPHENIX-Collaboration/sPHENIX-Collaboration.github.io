209 
210 
211  if ( Femc_clusterizer == kFemcTemplateClusterizer )
212  {
213  RawClusterBuilderTemplateFEMC *ClusterBuilder = new RawClusterBuilderTemplateFEMC("EmcRawClusterBuilderTemplateFEMC");
214  ClusterBuilder->Detector("FEMC");
215  ClusterBuilder->Verbosity(verbosity);
216  ClusterBuilder->set_threshold_energy(0.020); // This threshold should be the same as in FEMCprof_Thresh**.root file below
217  std::string femc_prof = getenv("CALIBRATIONROOT");
218  femc_prof += "/EmcProfile/FEMCprof_Thresh20MeV.root";
219  ClusterBuilder->LoadProfile(femc_prof.c_str());
220  se->registerSubsystem(ClusterBuilder);
221  }
222  else if ( Femc_clusterizer == kFemcGraphClusterizer )
223  {
224  RawClusterBuilderFwd* ClusterBuilder = new RawClusterBuilderFwd("FEMCRawClusterBuilderFwd");
225 
226 
227  ClusterBuilder->Detector("FEMC");
228  ClusterBuilder->Verbosity(verbosity);
229  ClusterBuilder->set_threshold_energy(0.010);
230  se->registerSubsystem( ClusterBuilder );
231  }
232  else
233  {
234  cout << "FEMC_Clusters - unknown clusterizer setting!"<<endl;
235  exit(1);
236  }
237 
238  return;
