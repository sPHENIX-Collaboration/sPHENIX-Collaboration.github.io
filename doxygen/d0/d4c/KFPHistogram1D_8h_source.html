65  void Fill(float value)
66  {
67  int iBin = floor(float(value - fMinBin)/float(fMaxBin - fMinBin) * float(fSize-2)) + 1;
68 
69  if(iBin > fSize-1)
70  iBin = fSize-1;
71  if(iBin < 1)
72  iBin = 0;
73 
74  if( !(iBin==iBin) || !(std::isfinite(iBin)) ) iBin = 0;
75 
76  fHistogram[iBin]++;
77  }
