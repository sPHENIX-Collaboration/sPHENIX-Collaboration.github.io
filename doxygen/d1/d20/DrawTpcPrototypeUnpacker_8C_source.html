54 void useLogBins(TAxis *axis)
55 {
56  assert(axis);
57  assert(axis->GetXmin() > 0);
58  assert(axis->GetXmax() > 0);
59 
60  const int bins = axis->GetNbins();
61 
62  Axis_t from = log10(axis->GetXmin());
63  Axis_t to = log10(axis->GetXmax());
64  Axis_t width = (to - from) / bins;
65  vector<Axis_t> new_bins(bins + 1);
66 
67  for (int i = 0; i <= bins; i++)
68  {
69  new_bins[i] = TMath::Power(10, from + i * width);
70  }
71  axis->Set(bins, new_bins.data());
72 }
