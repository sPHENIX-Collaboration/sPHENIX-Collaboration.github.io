250  inline static double DeltaPhi (double phi1, double phi2) {
251  phi1 = InTwoPi(phi1);
252  phi2 = InTwoPi(phi2);
253  double dphi = abs(phi1 - phi2);
254  while (dphi > TMath::Pi()) dphi = abs (dphi - 2*TMath::Pi());
255  return dphi;
